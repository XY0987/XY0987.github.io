与车相交的点

给你一个下标从 0 开始的二维整数数组 nums 表示汽车停放在数轴上的坐标。对于任意下标 i，nums[i] = [starti, endi] ，其中 starti 是第 i 辆车的起点，endi 是第 i 辆车的终点。

返回数轴上被车 任意部分 覆盖的整数点的数目。

/**
 * @param {number[][]} nums
 * @return {number}
 */
var numberOfPoints = function (nums) {
  nums.sort((a, b) => {
    return a[0] - b[0];
  });

  const set = new Set();
  // 将元素添加进set
  for (let i = 0; i < nums.length; i++) {
    for (let j = nums[i][0]; j <= nums[i][1]; j++) {
      set.add(j);
    }
  }
  return set.size;
};

最大二进制奇数

给你一个 二进制 字符串 s ，其中至少包含一个 ‘1’ 。

你必须按某种方式 重新排列 字符串中的位，使得到的二进制数字是可以由该组合生成的 最大二进制奇数 。

const s = "010";
const sArr = s.split("");
sArr.sort((a, b) => {
  return parseInt(b) - parseInt(a);
});
// 找到最后一个1
sArr[sArr.lastIndexOf("1")] = 0;
sArr[sArr.length - 1] = 1;
console.log(sArr.join(""));

美丽塔 I

给你一个长度为 n 下标从 0 开始的整数数组 maxHeights 。

/**
 * @param {number[]} maxHeights
 * @return {number}
 */
// 暴力解法，把当前节点当作中心，分别遍历左右两边满足要求的数据求和，最终取出最大值
var maximumSumOfHeights = function (maxHeights) {
  const res = [];
  for (let i = 0; i < maxHeights.length; i++) {
    let sum = 0;
    const currentValue = maxHeights[i];
    let temValue = currentValue;
    // 去遍历左边满足元素的情况，然后相加
    for (let j = i - 1; j >= 0; j--) {
      temValue = Math.min(temValue, maxHeights[j]);
      sum += temValue;
    }
    temValue = currentValue;
    for (let j = i + 1; j < maxHeights.length; j++) {
      temValue = Math.min(temValue, maxHeights[j]);
      sum += temValue;
    }
    res.push(currentValue + sum);
  }
  return Math.max(...res);
};
