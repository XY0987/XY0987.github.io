二叉树的最近公共祖先

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {TreeNode}
 */
var lowestCommonAncestor = function (root, p, q) {
  // 遍历到空节点，直接返回
  if (!root) {
    return null
  }
  // 遇到节点等于p或者等于q往上返回
  if (root === p || root === q) {
    return root
  }
  // 获取左树是否有满足条件的值
  let left = lowestCommonAncestor(root.left, p, q)
  // 获取右树是否有满足条件的值
  let right = lowestCommonAncestor(root.right, p, q)
  if (left && right) {
    return root
  }
  // 如果左为空，右不为空，表示答案在右子树里边
  if (!left && right) {
    return right
  }
  if (!right && left) {
    return left
  }
};

二叉搜索树的最近公共祖先

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */

/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {TreeNode}
 */
var lowestCommonAncestor = function (root, p, q) {
  /**
    // 默认方法
    if (!root) {
        return null
    }
    if (root === p || root === q) {
        return root
    }
    const left = lowestCommonAncestor(root.left, p, q)
    const right = lowestCommonAncestor(root.right, p, q)
    if (left && right) {
        return root
    }
    if (!left && right) {
        return right
    }
    if (!right && left) {
        return left
    }
    */
  // 利用二叉树特性，找到一个节点的值在两个节点值之间,就是最近的公共祖先
  if (!root) {
    return null
  }
  // 向左去遍历(答案在左子树中)
  if (root.val > p.val && root.val > q.val) {
    const left = lowestCommonAncestor(root.left, p, q)
    if (left) {
      return left
    }
  }
  // 向右去遍历(答案在右子树中)
  if (root.val < p.val && root.val < q.val) {
    const right = lowestCommonAncestor(root.right, p, q)
    if (right) {
      return right
    }
  }
  // 获取到结果了
  return root
};

二叉搜索树的插入操作

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} val
 * @return {TreeNode}
 */
var insertIntoBST = function (root, val) {
  function insert(node) {
    if (!node) {
      // 找到插入位置了
      return new TreeNode(val)
    }
    if (node.val > val) {
      // 应该插入左子树
      const left = insert(node.left)
      if (left) {
        node.left = left
      }
    }
    if (node.val < val) {
      // 应该插入右子树
      const right = insert(node.right)
      if (right) {
        node.right = right
      }
    }
  }
  if (!root) {
    return new TreeNode(val)
  }
  insert(root)
  return root
};

删除二叉搜索树的节点

  1. 没找到要删除的节点
  2. 删除的是叶子节点（左右都为空，不用改变二叉树的结构）
  3. 要删除的节点右为空，左不为空（直接让父节点指向节点的左孩子）
  4. 左空，右不空
  5. 左右都不为空（直接把左子树放到右子树中最小的数（右子树最左侧的位置）中）
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} key
 * @return {TreeNode}
 */
var deleteNode = function (root, key) {
  // 没有遍历到要删除的节点,返回空节点
  if (!root) {
    return null
  }
  // 找到要删除的节点了
  if (root.val === key) {
    // 情况一,删除叶子节点(左子树为空，右子树也为空)
    if (!root.left && !root.right) {
      return null
    } else if (!root.left && root.right) {
      // 情况二,节点左为空，右不为空
      return root.right
    } else if (root.left && !root.right) {
      // 情况三,节点左不为空，右为空
      return root.left
    } else {
      // 左右节点都不为空
      // 找到右子树的最小的节点(将左子树放到右子树的最小值哪里)
      let cur = root.right
      while (cur.left) {
        cur = cur.left
      }
      cur.left = root.left
      return root.right
    }
  }
  // 
  if (key < root.val) {
    // 向左去遍历
    root.left = deleteNode(root.left, key)
  }
  if (key > root.val) {
    // 向右去遍历
    root.right = deleteNode(root.right, key)
  }
  return root
};

修剪二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} low
 * @param {number} high
 * @return {TreeNode}
 */
var trimBST = function (root, low, high) {
  if (!root) {
    return null
  }
  if (root.val < low) {
    // 当前节点小于给定的最小值，应该往右遍历是否有符合结果的节点
    let right = trimBST(root.right, low, high)
    return right
  }
  if (root.val > high) {
    // 当前节点值大于给定的最大值，遍历右节点
    let left = trimBST(root.left, low, high)
    return left
  }
  // 处理左子树
  root.left = trimBST(root.left, low, high)
  // 处理右子树
  root.right = trimBST(root.right, low, high)
  return root
};

将有序数组转换为二叉搜索树

平衡二叉搜索树，选取中间节点开始构造，这样才能保证两边的节点数量相同，便于构造平衡二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} nums
 * @return {TreeNode}
 */
var sortedArrayToBST = function (nums) {
  // nums表示数组,left和right表示左右区间，左闭右闭
  function getRes(nums, left, right) {
    if (left > right) {
      return null
    }
    const mid = (left + right) >> 1
    const root = new TreeNode(nums[mid])
    // 左闭右闭，所以传mid-1
    root.left = getRes(nums, left, mid - 1)
    root.right = getRes(nums, mid + 1, right)
    return root
  }
  return getRes(nums, 0, nums.length - 1)
};

把二叉搜索树转化为累加树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var convertBST = function (root) {
  // 定义一个变量，用于记录之前节点的相加的结果
  let prev = 0
  function getRes(node) {
    if (!node) {
      return
    }
    // 因为要加上一个结果的值，所以倒序遍历(右中左)
    getRes(node.right)
    node.val += prev
    // 记录上一个节点的数值
    prev = node.val
    getRes(node.left)
  }
  getRes(root)
  return root
};

组合

/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
// 无剪枝
var combine = function (n, k) {
  // 不强调顺序,1,2与2,1是一个组合
  const res = [];
  const path = [];
  // startIndex用于记录开始的取值范围，避免重复取值
  function backtracking(n, k, startIndex) {
    if (path.length === k) {
      // 拷贝数组，否则是值引用，最后还是空数组
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i <= n; i++) {
      path.push(i);
      // 从之后位置开始搜索
      backtracking(n, k, i + 1);
      path.pop();
    }
  }
  backtracking(n, k, 1);
  return res
};

组合总和三

/**
 * @param {number} k
 * @param {number} n
 * @return {number[][]}
 */
var combinationSum3 = function (k, n) {
  const res = [];
  const path = [];
  let sum = 0;
  function backtracking(n, k, startIndex) {
    if (sum === n && path.length === k) {
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i <= 9; i++) {
      if (path.length > k) {
        return;
      }
      sum += i;
      path.push(i);
      backtracking(n, k, i + 1);
      sum -= i;
      path.pop();
    }
  }
  backtracking(n, k, 1);
  return res
};

电话号码的字母组合

/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
  if (digits === '') {
    return []
  }
  const map = new Map();
  map.set("1", "");
  map.set("2", "abc");
  map.set("3", "def");
  map.set("4", "ghi");
  map.set("5", "jkl");
  map.set("6", "mno");
  map.set("7", "pqrs");
  map.set("8", "tuv");
  map.set("9", "wxyz");
  const strArr = digits.split("");
  const res = [];
  const path = [];
  function backtracking(startIndex) {
    if (path.length === strArr.length) {
      res.push(path.join(""));
      return;
    }
    const str = map.get(strArr[startIndex]);
    for (let i = 0; i < str.length; i++) {
      path.push(str[i]);
      backtracking(startIndex + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res
};

组合总和

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum = function (candidates, target) {
  // 遍历之后的数组不能带前面的   2,5与5,2是同一个,所以后面的不用再选取前面的
  const res = [];
  const path = [];
  let sum = 0;
  /* 
    2,5和5,2是同一个，所以再之后的选取中，不能再选取该元素前面的元素了
    */
  function backtracking(startIndex) {
    if (sum > target) {
      return;
    }
    if (sum === target) {
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i < candidates.length; i++) {
      sum += candidates[i];
      path.push(candidates[i]);
      // 可以重复选取，所以从id开始
      backtracking(i);
      sum -= candidates[i];
      path.pop();
    }
  }
  backtracking(0);
  return res
};

组合总和二

去重做的是树层去重

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum2 = function (candidates, target) {
  candidates.sort((a, b) => a - b);
  const path = [];
  const res = new Set();
  let sum = 0;
  function backtracking(startIndex) {
    if (sum > target) {
      return;
    }
    if (sum === target) {
      const temPath = [...path];
      // 对原数组排序的话，回溯时会导致顺序错乱
      res.add(
        temPath
        .sort((a, b) => {
          return a - b;
        })
        .join(",")
      );
      return;
    }
    for (let i = startIndex; i < candidates.length; i++) {
      /* 
      数组已经排过序，再进入第二次搜索时，以该元素开头的结果已经搜索过了
      就不用再次搜索了
      */
      if (i > startIndex && candidates[i] === candidates[i - 1]) {
        continue;
      }
      sum += candidates[i];
      path.push(candidates[i]);
      backtracking(i + 1);
      sum -= candidates[i];
      path.pop();
    }
  }
  backtracking(0);
  const ans = Array.from(res).map((item) => {
    return item.split(",");
  });
  return ans
};

分割回文串

字符串回溯放的是一段，之前回溯放的是一个，其本质都是以不同起点开头，遍历到终点的一个过程，只不过字符串表示的是以这个为起点为分割线，前面的是一体的

/**
 * @param {string} s
 * @return {string[][]}
 */
var partition = function (s) {
  const path = [];
  const res = [];
  // 判断是否是回文串
  const isPalindrome = (s, l, r) => {
    for (let i = l, j = r; i < j; i++, j--) {
      if (s[i] !== s[j]) return false;
    }
    return true;
  };
  function backtracking(startIndex) {
    // path收集的是同一起点到终点之间的所有回文子串
    if (startIndex >= s.length) {
      res.push([...path]);
      return;
    }
    // 开始的for循环都是第一层，之后递归是之后的层
    // 每次遍历都是不同的起点，遍历到最后(0 到最后之间的回文串，1到最后之间的回文串.......)
    for (let i = startIndex; i < s.length; i++) {
      // 判断startIndex到i之间是否是回文串
      if (!isPalindrome(s, startIndex, i)) {
        continue;
      }
      // 放的是一段字符串
      path.push(s.slice(startIndex, i + 1));
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res;
};

复原IP地址

/**
 * @param {string} s
 * @return {string[]}
 */
var restoreIpAddresses = function (s) {
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    // 终止条件
    if (path.length > 4) {
      return;
    }
    if (path.length === 4 && startIndex === s.length) {
      res.push(path.join("."));
      return;
    }
    // 单层递归逻辑
    for (let i = startIndex; i < s.length; i++) {
      // 判断截取的字符串是否满足
      const str = s.slice(startIndex, i + 1);
      // 分割的字符串大于255（长度大于4的原因是避免有0）
      if (str.length > 4 || parseInt(str) > 255) {
        break;
      }
      // 不是一个数字，开头是以0开头的，也不行
      if (str.length > 1 && str[0] === "0") {
        break;
      }
      path.push(str);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res
};

/**

var subsets = function (nums) {
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    if (startIndex >= nums.length) {
      return;
    }
    for (let i = startIndex; i < nums.length; i++) {
      path.push(nums[i]);
      res.push([...path]);
      backtracking(i + 1);
      path.pop();
    }
  }

  backtracking(0);
  res.push([]);
  return res
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsetsWithDup = function (nums) {
  const path = [];
  const res = [];
  nums.sort((a, b) => {
    return a - b;
  });
  function backtracking(startIndex) {
    if (startIndex >= nums.length) {
      return;
    }
    for (let i = startIndex; i < nums.length; i++) {
      // 前面的元素已经收集过了，就不用再收集相同元素了(大于startIndex表示单层),树层去重
      if (i > startIndex && nums[i] === nums[i - 1]) {
        continue;
      }
      path.push(nums[i]);
      res.push([...path]);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  res.push([]);
  return res
};
