/**
 * @param {number[]} nums
 * @return {number}
 */
var findLengthOfLCIS = function (nums) {
  // res默认从0开始，因为数组从1开始
  let res = 1;
  // index用于记录连续开始的下标
  let index = 0;
  for (let i = 1; i < nums.length; i++) {
    // 小于前一个数，下标重新开始
    if (nums[i] <= nums[i - 1]) {
      index = i;
    }
    res = Math.max(i - index + 1, res);
  }
  // console.log(res);
  return res
};

暴力解法

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findLength = function (nums1, nums2) {
  let res = 0;
  for (let i = 0; i < nums1.length; i++) {
    for (let j = 0; j < nums2.length; j++) {
      let temNum = 0;
      while (
        i + temNum < nums1.length &&
        j + temNum < nums2.length &&
        nums1[i + temNum] === nums2[j + temNum]
      ) {
        temNum++;
      }
      res = Math.max(res, temNum);
    }
  }
  // console.log(res);
  return res;
};

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findLength = function (nums1, nums2) {
  const length1 = nums1.length,
    length2 = nums2.length;
  const dp = new Array(length1 + 1)
    .fill(0)
    .map(() => new Array(length2 + 1).fill(0));
  let resMax = 0;
  for (let i = 1; i <= length1; i++) {
    for (let j = 1; j <= length2; j++) {
      if (nums1[i - 1] === nums2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
        resMax = Math.max(resMax, dp[i][j]);
      }
    }
  }
  return resMax;
};

最长公共子序列

/**
 * @param {string} text1
 * @param {string} text2
 * @return {number}
 */
var longestCommonSubsequence = function (text1, text2) {
  let dp = Array.from(Array(text1.length + 1), () => Array(text2.length + 1).fill(0));
  for (let i = 1; i <= text1.length; i++) {
    for (let j = 1; j <= text2.length; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1])
      }
    }
  }
  return dp[text1.length][text2.length];
};

不相交的线

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var maxUncrossedLines = function (nums1, nums2) {
  // 两个数组长度
  const [m, n] = [nums1.length, nums2.length];
  // 创建dp数组并都初始化为0
  const dp = new Array(m + 1).fill(0).map(x => new Array(n + 1).fill(0));
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      // 根据两种情况更新dp[i][j]
      if (nums1[i - 1] === nums2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  // 返回dp数组中右下角的元素
  return dp[m][n];
};

/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
  // 如果和小于0，就从下一个开始
  let res = Number.MIN_SAFE_INTEGER;
  let cur = 0;
  if (nums.length === 1) {
    return nums[0]
  }
  for (let i = 0; i < nums.length; i++) {
    if (cur < 0) {
      cur = 0;
    }
    cur += nums[i]
    res = Math.max(res, cur)
  }
  return res;
};

判断子序列

/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isSubsequence = function (s, t) {
  // const strS = Array.from(s).sort().join("");
  // const strT = Array.from(t).sort().join("");
  // console.log(strT.includes(strS));

  /* 
  转换为两个数组，相同的arrS移除首元素，最后判断arrS的长度
  */

  const arrS = s.split("");
  const arrT = t.split("");

  if (arrS.length > arrT.length) {
    return false;
  }

  while (arrS.length && arrT.length) {
    if (arrS[0] === arrT[0]) {
      arrS.shift();
    }
    arrT.shift();
  }
  // console.log(arrS.length === 0 ? true : false);
  return arrS.length === 0 ? true : false;
};

不同的子序列

/**
 * @param {string} s
 * @param {string} t
 * @return {number}
 */
var numDistinct = function (s, t) {
  /* 
    不要求连续
    dp[i][j] 表示以i-1为结尾的s中有以j-1为结尾的t的个数为dp[i][j]
    */
  let dp = Array.from(Array(s.length + 1), () => Array(t.length + 1).fill(0));
  for (let i = 0; i <= s.length; i++) {
    dp[i][0] = 1;
  }
  for (let i = 1; i <= s.length; i++) {
    for (let j = 1; j <= t.length; j++) {
      if (s[i - 1] === t[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j];
      } else {
        dp[i][j] = dp[i - 1][j];
      }
    }
  }
  return dp[s.length][t.length];
};

/**
 * @param {string} word1
 * @param {string} word2
 * @return {number}
 */
var minDistance = function (word1, word2) {
  let dp = Array.from(new Array(word1.length + 1), () =>
    Array(word2.length + 1).fill(0)
                     );
  for (let i = 1; i <= word1.length; i++) {
    dp[i][0] = i;
  }
  for (let j = 1; j <= word2.length; j++) {
    dp[0][j] = j;
  }
  for (let i = 1; i <= word1.length; i++) {
    for (let j = 1; j <= word2.length; j++) {
      if (word1[i - 1] === word2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      } else {
        dp[i][j] = Math.min(
          dp[i - 1][j] + 1,
          dp[i][j - 1] + 1,
          dp[i - 1][j - 1] + 2
        );
      }
    }
  }
  return dp[word1.length][word2.length];
};

var countSubstrings = function (s) {

  //     for (let j = i + 1; j < s.length + 1; j++) {

  // return res;
  const strLen = s.length;
  let numOfPalindromicStr = 0;
  let dp = Array.from(Array(strLen), () => Array(strLen).fill(false));

  for (let j = 0; j < strLen; j++) {
    for (let i = 0; i <= j; i++) {
      if (s[i] === s[j]) {
        if ((j - i) < 2) {
          dp[i][j] = true;
        } else {
          dp[i][j] = dp[i + 1][j - 1];
        }
        numOfPalindromicStr += dp[i][j] ? 1 : 0;
      }
    }
  }
  return numOfPalindromicStr;
};

最长回文子序列

/**
 * @param {string} s
 * @return {number}
 */
var longestPalindromeSubseq = function (s) {
  /* 
    可以不连续
    */
  const strLen = s.length;
  let dp = Array.from(Array(strLen), () => Array(strLen).fill(0));
  for (let i = 0; i < strLen; i++) {
    dp[i][i] = 1;
  }
  for (let i = strLen - 1; i >= 0; i--) {
    for (let j = i + 1; j < strLen; j++) {
      if (s[i] === s[j]) {
        dp[i][j] = dp[i + 1][j - 1] + 2;
      } else {
        dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
      }
    }
  }
  return dp[0][strLen - 1];
};

/**
 * @param {number[]} temperatures
 * @return {number[]}
 */
var dailyTemperatures = function (temperatures) {
  const res = new Array(temperatures.length).fill(0);
  for (let i = 0; i < temperatures.length - 1; i++) {
    let index = 0;
    for (let j = i + 1; j < temperatures.length; j++) {
      if (temperatures[j] > temperatures[i]) {
        index = j - i;
        break;
      }
    }
    res[i] = index;
  }
  // console.log(res);
  return res;
};

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var nextGreaterElement = function (nums1, nums2) {
  const res = new Array(nums1.length).fill(-1);
  for (let i = 0; i < nums1.length; i++) {
    let index = nums2.indexOf(nums1[i]);
    for (let j = index; j < nums2.length; j++) {
      if (nums1[i] < nums2[j]) {
        // 要的是元素，不是元素对应的下标
        res[i] = nums2[j];
        break;
      }
    }
  }
  return res;
};

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var nextGreaterElements = function (nums) {
  const max = Math.max(...nums);
  const res = new Array(nums.length).fill(-1);
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === max) {
      continue;
    }
    let j = i;
    while (true) {
      j++;
      j = j % nums.length;
      if (nums[i] < nums[j]) {
        res[i] = nums[j];
        break;
      }
    }
  }
  return res;
};

/**
 * @param {number[][]} grid
 * @return {number}
 */
var findChampion = function(grid) {
  let numOne = 0;
  let res = 0;
  for (let i = 0; i < grid.length; i++) {
    const arr = [...grid[i]];
    arr.sort((a, b) => {
      return a - b;
    });
    if (arr.indexOf(1) !== -1) {
      let size = arr.lastIndexOf(1) - arr.indexOf(1) + 1;
      if (size > numOne) {
        numOne = size;
        res = i;
      }
    }
  }
  return res;
};

100116. 找到冠军 II

/**
 * @param {number} n
 * @param {number[][]} edges
 * @return {number}
 */
var findChampion = function(n, edges) {
  const res = new Set();
  if (n == 1) {
    return 0;
  }

  const set = new Set();
  for (let i = 0; i < edges.length; i++) {
    set.add(edges[i][0]);
    set.add(edges[i][1]);
  }
  if (set.size < n) {
    return -1;
  }
  for (let i = 0; i < edges.length; i++) {
    // 存储当前最强的
    let yn = true;
    const first = edges[i][0];
    for (let j = 0; j < edges.length; j++) {
      if (j == i) {
        continue;
      }
      const tem = edges[j][1];
      if (first == tem) {
        yn = false;
        break;
      }
    }
    if (yn) {
      res.add(first);
    }
  }
  const arr = Array.from(res);
  return arr.length == 1 ? arr[0] : -1;
};
