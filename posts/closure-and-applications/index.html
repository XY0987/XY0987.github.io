JavaScript闭包深入解析：原理、应用与最佳实践

发表于2023|更新于2026|前端基础
|字数总计:2.5k|阅读时长:11分钟

闭包（Closure）是 JavaScript 中最强大、最核心的特性之一，也是面试必考知识点。理解闭包对于编写高质量代码、掌握设计模式至关重要。本文将全面深入地解析闭包的原理与应用。

一、什么是闭包？

1.1 闭包的定义

闭包是指有权访问另一个函数作用域中变量的函数。简单来说，当一个函数能够”记住”并访问它被创建时所在的词法作用域时，就产生了闭包。

// 最简单的闭包示例
function outer() {
  const message = 'Hello';
  
  function inner() {
    console.log(message);  // 访问外部函数的变量
  }
  
  return inner;
}

const fn = outer();
fn();  // 'Hello'

1.2 闭包的形成条件

// 闭包形成的三个条件
function createClosure() {
  // 1. 存在函数嵌套
  const outerVar = 'I am from outer';
  
  // 2. 内部函数引用了外部函数的变量
  return function inner() {
    console.log(outerVar);  // 引用 outerVar
  };
  
  // 3. 内部函数被返回或传递到外部
}

const closure = createClosure();  // 3. 函数执行完毕后，内部函数仍被引用
closure();  // 'I am from outer'

1.3 闭包的本质

闭包的本质是：函数 + 函数能够访问的词法作用域

function outer() {
  const a = 1;
  const b = 2;
  
  function inner() {
    console.log(a);
  }
  
  return inner;
}

/*
inner 函数的作用域链：
[inner, outer, Global]

即使 outer 执行完毕，outer 仍然存在于 inner 的作用域链中
这就是闭包的本质
*/

二、闭包的原理

2.1 垃圾回收机制

JavaScript 有自动的垃圾回收机制，当一个值不再被引用时，会被回收。

// 没有闭包：函数执行完后变量被回收
function normalFn() {
  const a = 1;
  console.log(a);
}
normalFn();
// 函数执行完，a 被回收

// 有闭包：变量不会被回收
function closureFn() {
  const a = 1;
  return function() {
    console.log(a);  // 引用 a
  };
}
const fn = closureFn();
fn();  // 1
// a 不会被回收，因为 fn 仍然引用它

2.2 作用域链的作用

function createCounter() {
  let count = 0;
  
  return {
    increment: function() {
      count++;
      console.log(count);
    },
    decrement: function() {
      count--;
      console.log(count);
    },
    getCount: function() {
      return count;
    }
  };
}

const counter = createCounter();
counter.increment();  // 1
counter.increment();  // 2
counter.decrement();  // 1
console.log(counter.getCount());  // 1

/*
count 存在于 createCounter 中
每次调用 increment，都会沿着作用域链找到 count
*/

2.3 词法作用域与闭包

// 词法作用域：函数定义时确定作用域
function outer() {
  const x = 10;
  
  function inner() {
    console.log(x);  // 定义时确定：访问 outer 的 x
  }
  
  return inner;
}

const fn = outer();
fn();  // 10

// 即使在外部调用，仍然能访问 outer 的变量
// 这就是词法作用域的"记忆"特性

三、闭包的应用场景

3.1 数据私有化

闭包可以创建私有变量，实现数据封装：

// 创建私有变量
function createPerson(name) {
  let _age = 0;  // 私有变量
  
  return {
    getName: function() {
      return name;
    },
    getAge: function() {
      return _age;
    },
    setAge: function(age) {
      if (age > 0 && age < 150) {
        _age = age;
      }
    }
  };
}

const person = createPerson('张三');
console.log(person.getName());  // '张三'
console.log(person.getAge());   // 0
person.setAge(25);
console.log(person.getAge());   // 25
person.setAge(200);  // 无效
console.log(person.getAge());   // 25

// _age 无法直接访问
console.log(person._age);  // undefined

3.2 单例模式

利用闭包实现单例模式：

// 单例模式：确保一个类只有一个实例
const Singleton = (function() {
  let instance;  // 闭包保存实例
  
  function createInstance(name) {
    this.name = name;
  }
  
  return {
    getInstance: function(name) {
      if (!instance) {
        instance = new createInstance(name);
      }
      return instance;
    }
  };
})();

const instance1 = Singleton.getInstance('实例1');
const instance2 = Singleton.getInstance('实例2');

console.log(instance1 === instance2);  // true
console.log(instance1.name);  // '实例1'
console.log(instance2.name);  // '实例1'（第二次调用不会创建新实例）

3.3 柯里化函数

闭包实现函数柯里化：

// 柯里化：将多参数函数转换为单参数函数
function curry(fn) {
  const args = Array.prototype.slice.call(arguments, 1);
  
  return function() {
    const innerArgs = Array.prototype.slice.call(arguments);
    const finalArgs = args.concat(innerArgs);
    
    if (finalArgs.length >= fn.length) {
      return fn.apply(null, finalArgs);
    } else {
      return curry.apply(null, [fn].concat(finalArgs));
    }
  };
}

// 使用示例
function add(a, b, c) {
  return a + b + c;
}

const curriedAdd = curry(add);
console.log(curriedAdd(1)(2)(3));  // 6
console.log(curriedAdd(1, 2)(3));  // 6
console.log(curriedAdd(1)(2, 3));  // 6

// 简化版柯里化
function simpleCurry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function(...args2) {
      return curried.apply(this, args.concat(args2));
    };
  };
}

3.4 防抖与节流

利用闭包实现防抖和节流：

// 防抖：在一定时间内，只执行最后一次
function debounce(fn, delay) {
  let timer = null;  // 闭包保存 timer
  
  return function(...args) {
    if (timer) {
      clearTimeout(timer);
    }
    
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// 使用
const handleInput = debounce(function(value) {
  console.log('搜索：', value);
}, 500);

handleInput('h');
handleInput('he');
handleInput('hel');
handleInput('hello');  // 只有这次会执行（500ms 后）

// 节流：在一定时间内，只执行一次
function throttle(fn, delay) {
  let lastTime = 0;  // 闭包保存上次执行时间
  
  return function(...args) {
    const now = Date.now();
    
    if (now - lastTime >= delay) {
      fn.apply(this, args);
      lastTime = now;
    }
  };
}

// 使用
const handleScroll = throttle(function() {
  console.log('滚动中...');
}, 200);

window.addEventListener('scroll', handleScroll);

3.5 模块化

利用闭包实现模块化：

// 模块模式
const Module = (function() {
  // 私有变量和方法
  let privateVar = '私有变量';
  
  function privateMethod() {
    console.log('私有方法');
  }
  
  // 公开的接口
  return {
    publicVar: '公共变量',
    
    publicMethod: function() {
      console.log(privateVar);
      privateMethod();
    },
    
    getPrivateVar: function() {
      return privateVar;
    }
  };
})();

Module.publicMethod();  // '私有变量' '私有方法'
console.log(Module.publicVar);  // '公共变量'
console.log(Module.privateVar);  // undefined（无法访问）

// 现代模块模式（ES6+）
const ModernModule = (() => {
  const privateData = new WeakMap();
  
  class Module {
    constructor(data) {
      privateData.set(this, data);
    }
    
    getData() {
      return privateData.get(this);
    }
    
    setData(data) {
      privateData.set(this, data);
    }
  }
  
  return Module;
})();

四、经典闭包问题

4.1 循环中的闭包问题

// 经典问题
for (var i = 1; i <= 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 100);
}
// 输出：6, 6, 6, 6, 6

/*
原因：
1. var 声明的 i 是函数作用域，循环结束后 i = 6
2. setTimeout 的回调函数形成闭包，但引用的是同一个 i
3. 100ms 后执行时，i 已经变成 6
*/

// 解决方案一：IIFE（立即执行函数）
for (var i = 1; i <= 5; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j);
    }, 100);
  })(i);
}
// 输出：1, 2, 3, 4, 5

// 解决方案二：let（块级作用域）
for (let i = 1; i <= 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 100);
}
// 输出：1, 2, 3, 4, 5

// 解决方案三：使用 setTimeout 的第三个参数
for (var i = 1; i <= 5; i++) {
  setTimeout(function(j) {
    console.log(j);
  }, 100, i);
}
// 输出：1, 2, 3, 4, 5

五、闭包的内存管理

5.1 内存泄漏风险

闭包会导致外部函数的变量无法被回收，可能造成内存泄漏：

// 潜在的内存泄漏
function createHandlers() {
  const largeData = new Array(1000000).fill('x');
  
  return {
    method1: function() {
      console.log('method1');
    },
    method2: function() {
      console.log(largeData.length);  // 引用了 largeData
    }
  };
}

// 即使只使用 method1，largeData 也无法被回收
const handlers = createHandlers();
handlers.method1();  // largeData 仍存在于内存中

5.2 避免内存泄漏

// 方案一：只引用需要的变量
function createHandlers() {
  const largeData = new Array(1000000).fill('x');
  const length = largeData.length;  // 只保存需要的数据
  
  return {
    method1: function() {
      console.log('method1');
    },
    method2: function() {
      console.log(length);
    }
  };
}

// 方案二：手动释放
function createClosure() {
  let data = { name: 'test' };
  
  const closure = {
    getData: function() {
      return data;
    },
    clear: function() {
      data = null;  // 手动释放
    }
  };
  
  return closure;
}

const c = createClosure();
c.getData();
c.clear();  // 释放内存

六、实战案例

6.1 实现一个简单的状态管理

function createStore(reducer) {
  let state = reducer(undefined, {});
  const listeners = [];
  
  return {
    getState: function() {
      return state;
    },
    
    dispatch: function(action) {
      state = reducer(state, action);
      listeners.forEach(listener => listener());
    },
    
    subscribe: function(listener) {
      listeners.push(listener);
      return function() {
        const index = listeners.indexOf(listener);
        listeners.splice(index, 1);
      };
    }
  };
}

// 使用示例
function counterReducer(state = 0, action) {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
    default:
      return state;
  }
}

const store = createStore(counterReducer);

store.subscribe(() => {
  console.log('State changed:', store.getState());
});

store.dispatch({ type: 'INCREMENT' });  // State changed: 1
store.dispatch({ type: 'INCREMENT' });  // State changed: 2
store.dispatch({ type: 'DECREMENT' });  // State changed: 1

6.2 实现发布-订阅模式

const EventEmitter = (function() {
  const events = {};  // 私有事件存储
  
  return {
    on: function(event, callback) {
      if (!events[event]) {
        events[event] = [];
      }
      events[event].push(callback);
    },
    
    emit: function(event, ...args) {
      if (events[event]) {
        events[event].forEach(callback => callback.apply(null, args));
      }
    },
    
    off: function(event, callback) {
      if (events[event]) {
        events[event] = events[event].filter(cb => cb !== callback);
      }
    }
  };
})();

// 使用
EventEmitter.on('message', function(msg) {
  console.log('收到消息：', msg);
});

EventEmitter.emit('message', 'Hello!');  // 收到消息：Hello!

6.3 实现惰性函数

// 惰性载入：第一次执行后重写函数
function createXHR() {
  if (typeof XMLHttpRequest !== 'undefined') {
    createXHR = function() {
      return new XMLHttpRequest();
    };
  } else if (typeof ActiveXObject !== 'undefined') {
    createXHR = function() {
      return new ActiveXObject('Microsoft.XMLHTTP');
    };
  } else {
    createXHR = function() {
      throw new Error('No XHR object available.');
    };
  }
  
  return createXHR();
}

// 第一次调用会进行检测，后续调用直接返回缓存的函数
const xhr1 = createXHR();
const xhr2 = createXHR();  // 直接使用缓存的函数

七、总结

7.1 闭包的特点

特点 说明
函数嵌套 存在函数嵌套结构
引用外部变量 内部函数引用外部函数的变量
延长变量生命周期 变量在函数执行后仍然存在
私有化 可以创建私有变量和方法

7.2 闭包的应用

  1. 数据私有化：创建私有变量
  2. 单例模式：确保唯一实例
  3. 柯里化：函数式编程
  4. 防抖节流：性能优化
  5. 模块化：封装私有成员

7.3 注意事项

  1. 内存泄漏：及时释放不需要的引用
  2. this 指向：注意闭包中 this 的指向
  3. 性能考虑：闭包会占用内存，不宜过度使用
  4. 循环问题：注意循环中的闭包陷阱

7.4 面试高频问题

  1. 什么是闭包？ → 函数 + 词法作用域
  2. 闭包的应用场景？ → 私有变量、单例、柯里化等
  3. 闭包的缺点？ → 内存泄漏风险
  4. 如何解决循环闭包问题？ → let/IIFE
  5. 闭包与内存管理？ → 手动释放、避免不必要的引用
