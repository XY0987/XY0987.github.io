Ollama本地部署与私人知识库搭建

发表于2025|更新于2026|AI相关
|字数总计:2.5k|阅读时长:11分钟

前言：最近ollama比较火，都在本地部署大模型。所以这一篇，我想讲解一下如何在自己的电脑上跑大模型。

环境准备与 Ollama 安装

硬件要求

先说实话，本地跑大模型对硬件有要求：

模型规模 最低内存 推荐配置
1.5B-3B（如 DeepSeek-R1:1.5b） 4GB RAM 8GB RAM
7B-8B（如 LLaMA3:8b、Qwen2.5:7b） 8GB RAM 16GB RAM
14B（如 Qwen2.5:14b） 16GB RAM 32GB RAM
32B-70B 32GB+ RAM 64GB+ RAM / GPU

我的实际配置：MacBook Pro M2 16GB —— 跑 7B-8B 模型流畅，14B 模型可以接受，再大就吃力了。

安装 Ollama

# macOS —— 直接下载安装
# 访问 https://ollama.com/download 下载安装包，双击安装即可

# 或者用 Homebrew
brew install ollama

# Linux —— 一行命令
curl -fsSL https://ollama.com/install.sh | sh

# 验证安装
ollama --version
# 输出类似：ollama version is 0.6.x

下载并运行第一个模型

# 推荐从 Qwen2.5:7b 开始（中文能力强，性价比高）
ollama run qwen2.5:7b

# 下载完成后直接进入对话模式
>>> 你好，请用一句话介绍自己
我是通义千问，一个由阿里云开发的AI助手。

>>> 用JavaScript写一个快速排序
# 它会直接输出可运行的代码

# 退出对话
>>> /bye

常用模型推荐

# 中文场景首选
ollama pull qwen2.5:7b        # 阿里通义千问，中文很强
ollama pull deepseek-r1:8b     # DeepSeek推理模型，逻辑链清晰

# 轻量级（性能弱但速度快，适合测试）
ollama pull qwen2.5:1.5b
ollama pull deepseek-r1:1.5b

# 嵌入模型（构建知识库必需）
ollama pull shaw/dmeta-embedding-zh:latest   # 文本向量化模型

# 查看已下载的模型
ollama list

我的选型建议：

  • 日常中文对话：qwen2.5:7b（中文理解最好）
  • 代码生成：deepseek-r1:8b（推理链条清晰，代码质量高）
  • 知识库嵌入：shaw/dmeta-embedding-zh:latest（处理中文比较好的本地嵌入模型）

Ollama API 调用

Ollama 安装后自动启动一个 HTTP 服务（默认端口 11434），兼容 OpenAI API 格式：

// 基础对话调用 —— 和调 OpenAI 几乎一样
const response = await fetch('http://localhost:11434/api/chat', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    model: 'qwen2.5:7b',
    messages: [
      { role: 'system', content: '你是一个友好的助手' },
      { role: 'user', content: '用一句话解释什么是闭包' }
    ],
    stream: false  // 非流式，等待完整响应
  })
});

const data = await response.json();
console.log(data.message.content);
// 输出："闭包是指一个函数能够访问并记住其词法作用域中的变量，即使该函数在其词法作用域之外执行。"

// 流式调用 —— 逐字输出
async function streamChat(prompt) {
  const response = await fetch('http://localhost:11434/api/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'qwen2.5:7b',
      messages: [{ role: 'user', content: prompt }],
      stream: true  // 流式输出
    })
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let fullResponse = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    const json = JSON.parse(chunk);

    if (json.message?.content) {
      process.stdout.write(json.message.content);  // 逐字打印
      fullResponse += json.message.content;
    }
  }

  return fullResponse;
}

await streamChat('写一个Node.js的Hello World HTTP服务器');

文本向量化：知识库的基础

构建知识库的核心步骤是把文本转换为向量，这样才能进行语义检索。

// 使用 Ollama 的嵌入模型生成文本向量
async function getEmbedding(text) {
  const response = await fetch('http://localhost:11434/api/embed', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'shaw/dmeta-embedding-zh:latest',
      input: text
    })
  });

  const data = await response.json();
  return data.embeddings[0];  // 返回一个数字数组（向量）
}

// 测试
const vector = await getEmbedding('JavaScript闭包是什么？');
console.log(`向量维度: ${vector.length}`);  // 768维
console.log(`前5个分量: ${vector.slice(0, 5)}`);
// 输出类似: [0.0234, -0.0891, 0.0456, ...]

实战：构建知识库问答系统

现在我们把所有能力串起来，构建一个本地知识库问答系统。

项目结构

local-rag/
├── package.json
├── index.mjs        # 主程序
├── knowledge/       # 知识库文档
│   ├── react-hooks.md
│   └── nodejs-basics.md
└── db/              # 向量存储（简单JSON实现）
    └── vectors.json

完整代码实现

// index.mjs —— 完全本地化的知识库问答系统
import fs from 'fs/promises';
import path from 'path';
import readline from 'readline';

const OLLAMA_BASE = 'http://localhost:11434';
const CHAT_MODEL = 'qwen2.5:7b';
const EMBED_MODEL = 'shaw/dmeta-embedding-zh:latest';
const VECTOR_DB_PATH = './db/vectors.json';
const KNOWLEDGE_DIR = './knowledge';

// ========== 1. 向量化工具 ==========
async function getEmbedding(text) {
  const res = await fetch(`${OLLAMA_BASE}/api/embed`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ model: EMBED_MODEL, input: text })
  });
  const data = await res.json();
  return data.embeddings[0];
}

function cosineSimilarity(a, b) {
  let dot = 0, normA = 0, normB = 0;
  for (let i = 0; i < a.length; i++) {
    dot += a[i] * b[i];
    normA += a[i] * a[i];
    normB += b[i] * b[i];
  }
  return dot / (Math.sqrt(normA) * Math.sqrt(normB));
}

// ========== 2. 文档处理 ==========
function splitDocument(text, chunkSize = 500, overlap = 50) {
  // 优先按段落分割，保持语义完整
  const paragraphs = text.split(/\n\n+/);
  const chunks = [];
  let currentChunk = '';

  for (const para of paragraphs) {
    if ((currentChunk + para).length > chunkSize && currentChunk) {
      chunks.push(currentChunk.trim());
      // 保留overlap部分，避免语义断裂
      const words = currentChunk.split('');
      currentChunk = words.slice(-overlap).join('') + '\n\n' + para;
    } else {
      currentChunk += (currentChunk ? '\n\n' : '') + para;
    }
  }
  if (currentChunk.trim()) {
    chunks.push(currentChunk.trim());
  }

  return chunks;
}

// ========== 3. 知识库构建 ==========
async function buildKnowledgeBase() {
  console.log('📚 开始构建知识库...');

  const files = await fs.readdir(KNOWLEDGE_DIR);
  const mdFiles = files.filter(f => f.endsWith('.md'));
  const vectorDB = [];

  for (const file of mdFiles) {
    const content = await fs.readFile(path.join(KNOWLEDGE_DIR, file), 'utf-8');
    const chunks = splitDocument(content);

    console.log(`  📄 ${file} → ${chunks.length} 个文档块`);

    for (let i = 0; i < chunks.length; i++) {
      const embedding = await getEmbedding(chunks[i]);
      vectorDB.push({
        id: `${file}_chunk_${i}`,
        source: file,
        text: chunks[i],
        embedding
      });
    }
  }

  // 持久化存储
  await fs.mkdir('./db', { recursive: true });
  await fs.writeFile(VECTOR_DB_PATH, JSON.stringify(vectorDB));
  console.log(`✅ 知识库构建完成，共 ${vectorDB.length} 个文档块\n`);

  return vectorDB;
}

// ========== 4. 语义检索 ==========
async function searchKnowledge(query, vectorDB, topK = 3) {
  const queryEmbedding = await getEmbedding(query);

  const scored = vectorDB.map(doc => ({
    ...doc,
    score: cosineSimilarity(queryEmbedding, doc.embedding)
  }));

  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, topK);
}

// ========== 5. RAG 问答 ==========
async function ragChat(question, vectorDB) {
  // 步骤1: 检索相关文档
  const relevantDocs = await searchKnowledge(question, vectorDB);

  console.log('\n🔍 检索到相关文档:');
  relevantDocs.forEach((doc, i) => {
    console.log(`  [${i + 1}] ${doc.source} (相关度: ${doc.score.toFixed(3)})`);
  });

  // 步骤2: 构建带上下文的Prompt
  const context = relevantDocs.map(d => d.text).join('\n\n---\n\n');

  const messages = [
    {
      role: 'system',
      content: `你是一个基于知识库的问答助手。请严格根据提供的"参考资料"回答用户问题。
如果参考资料中没有相关信息，请明确说"根据现有知识库，我没有找到相关信息"。
不要编造参考资料中不存在的内容。回答要简洁准确。`
    },
    {
      role: 'user',
      content: `参考资料：
${context}

问题：${question}`
    }
  ];

  // 步骤3: 调用本地模型生成回答（流式输出）
  const response = await fetch(`${OLLAMA_BASE}/api/chat`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ model: CHAT_MODEL, messages, stream: true })
  });

  process.stdout.write('\n💬 回答: ');
  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    const lines = decoder.decode(value).split('\n').filter(Boolean);
    for (const line of lines) {
      try {
        const json = JSON.parse(line);
        if (json.message?.content) {
          process.stdout.write(json.message.content);
        }
      } catch {}
    }
  }
  console.log('\n');
}

// ========== 6. 主程序 ==========
async function main() {
  // 加载或构建知识库
  let vectorDB;
  try {
    const data = await fs.readFile(VECTOR_DB_PATH, 'utf-8');
    vectorDB = JSON.parse(data);
    console.log(`📂 已加载知识库，共 ${vectorDB.length} 个文档块\n`);
  } catch {
    vectorDB = await buildKnowledgeBase();
  }

  // 交互式问答
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  });

  console.log('🤖 本地知识库问答系统已启动（输入 exit 退出，输入 rebuild 重建知识库）\n');

  const ask = () => {
    rl.question('❓ 你的问题: ', async (input) => {
      const trimmed = input.trim();
      if (trimmed === 'exit') {
        rl.close();
        return;
      }
      if (trimmed === 'rebuild') {
        vectorDB = await buildKnowledgeBase();
        ask();
        return;
      }
      if (trimmed) {
        await ragChat(trimmed, vectorDB);
      }
      ask();
    });
  };

  ask();
}

main();

准备知识库文档

<!-- knowledge/react-hooks.md -->
# React Hooks 核心知识

## useState
useState 是 React 最基础的 Hook，用于在函数组件中添加状态。

使用方式：
const [state, setState] = useState(initialValue);

注意事项：
1. setState 是异步的，不要依赖更新后的值
2. 如果新状态依赖旧状态，使用函数式更新：setState(prev => prev + 1)
3. 对于对象和数组，必须返回新的引用才能触发重渲染

## useEffect
useEffect 用于处理副作用，如数据请求、订阅、DOM操作等。

清除副作用：
useEffect(() => {
  const timer = setInterval(() => console.log('tick'), 1000);
  return () => clearInterval(timer);  // 清除函数
}, []);

依赖数组：
- 空数组 []：只在挂载时执行一次
- [dep1, dep2]：依赖变化时重新执行
- 不传：每次渲染都执行（慎用）
