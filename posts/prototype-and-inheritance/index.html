JavaScript原型与原型链完全指南：从原理到继承实现

发表于2023|更新于2026|前端基础
|字数总计:3.2k|阅读时长:15分钟

原型与原型链是 JavaScript 面向对象编程的核心，也是面试必考知识点。理解原型机制对于掌握 JavaScript 继承、属性查找、类型判断等特性至关重要。本文将全面深入地解析原型与原型链。

一、原型对象

1.1 什么是原型？

每个 JavaScript 对象在创建时都会关联另一个对象，这个对象就是它的原型。对象从原型继承属性和方法。

// 每个函数都有 prototype 属性
function Person(name) {
  this.name = name;
}

Person.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

const person = new Person('张三');
person.sayHi();  // 'Hi, I am 张三'

// person 从 Person.prototype 继承了 sayHi 方法

1.2 prototype 与 __proto__

function Person(name) {
  this.name = name;
}

const person = new Person('张三');

// prototype：函数特有的属性，指向原型对象
console.log(Person.prototype);  // { constructor: Person, ... }

// __proto__：对象特有的属性，指向它的原型
console.log(person.__proto__);  // Person.prototype
console.log(person.__proto__ === Person.prototype);  // true

// 关系图：
// person → person.__proto__ → Person.prototype
// Person.prototype.constructor → Person

1.3 constructor 属性

原型对象有一个 constructor 属性，指向关联的构造函数：

function Person(name) {
  this.name = name;
}

console.log(Person.prototype.constructor === Person);  // true

const person = new Person('张三');
console.log(person.constructor === Person);  // true
// person 本身没有 constructor，从原型链继承

// 重写 prototype 会丢失 constructor
Person.prototype = {
  sayHi: function() {}
};
console.log(Person.prototype.constructor === Person);  // false
console.log(Person.prototype.constructor === Object);  // true

// 修复 constructor
Person.prototype = {
  constructor: Person,  // 手动指定
  sayHi: function() {}
};
console.log(Person.prototype.constructor === Person);  // true

二、原型链

2.1 什么是原型链？

原型链是由对象的原型层层连接形成的链式结构。当访问对象属性时，JavaScript 会沿着原型链向上查找。

function Person(name) {
  this.name = name;
}

Person.prototype.sayHi = function() {
  console.log('Hi');
};

const person = new Person('张三');

// 原型链：
// person → Person.prototype → Object.prototype → null

console.log(person.__proto__);  // Person.prototype
console.log(person.__proto__.__proto__);  // Object.prototype
console.log(person.__proto__.__proto__.__proto__);  // null

// 查找属性的过程：
console.log(person.name);  // '张三'（实例属性）
console.log(person.sayHi);  // function（原型属性）
console.log(person.toString);  // function（Object.prototype 属性）
console.log(person.xyz);  // undefined（不存在）

2.2 原型链查找规则

function Person(name) {
  this.name = name;
}

Person.prototype.name = '原型name';

const person = new Person('实例name');

// 查找规则：就近原则
console.log(person.name);  // '实例name'（实例属性优先）

delete person.name;  // 删除实例属性
console.log(person.name);  // '原型name'（查找到原型）

delete Person.prototype.name;  // 删除原型属性
console.log(person.name);  // undefined（继续向上查找）

// hasOwnProperty：判断是否为实例属性
person.name = '实例name';
console.log(person.hasOwnProperty('name'));  // true
console.log(Person.prototype.hasOwnProperty('name'));  // false

2.3 完整的原型链结构

function Person(name) {
  this.name = name;
}

const person = new Person('张三');

/*
完整的原型链：

person（实例）
  │
  ├── __proto__ → Person.prototype
  │                  │
  │                  ├── constructor → Person
  │                  ├── __proto__ → Object.prototype
  │                  │                  │
  │                  │                  ├── constructor → Object
  │                  │                  ├── toString()
  │                  │                  ├── hasOwnProperty()
  │                  │                  ├── __proto__ → null
  │                  │                  │
  │                  │                  └── ...
  │                  │
  │                  └── sayHi()（自定义方法）
  │
  └── name: '张三'

Person（构造函数）
  │
  ├── prototype → Person.prototype
  ├── __proto__ → Function.prototype
  │                  │
  │                  ├── __proto__ → Object.prototype
  │                  │
  │                  └── ...
  │
  └── ...

Function.prototype
  │
  ├── constructor → Function
  ├── __proto__ → Object.prototype
  └── ...
*/

console.log(Person.__proto__ === Function.prototype);  // true
console.log(Function.__proto__ === Function.prototype);  // true（Function 是自己的实例）
console.log(Object.__proto__ === Function.prototype);  // true
console.log(Object.prototype.__proto__);  // null（原型链终点）

三、获取原型的方法

3.1 三种获取原型的方式

function Person(name) {
  this.name = name;
}

const person = new Person('张三');

// 方法一：__proto__（不推荐，已废弃）
console.log(person.__proto__);

// 方法二：Object.getPrototypeOf()（推荐）
console.log(Object.getPrototypeOf(person));  // Person.prototype

// 方法三：constructor.prototype（仅适用于构造函数创建的对象）
console.log(person.constructor.prototype);  // Person.prototype

// 三者相等
console.log(person.__proto__ === Object.getPrototypeOf(person));  // true
console.log(person.__proto__ === person.constructor.prototype);  // true

3.2 设置原型的方法

// 方法一：Object.setPrototypeOf()（ES6，不推荐用于性能敏感场景）
const obj = { a: 1 };
const proto = { b: 2 };
Object.setPrototypeOf(obj, proto);
console.log(obj.b);  // 2

// 方法二：Object.create()（推荐）
const newObj = Object.create(proto);
newObj.a = 1;
console.log(newObj.b);  // 2

// 方法三：直接赋值 __proto__（不推荐）
const obj2 = { a: 1 };
obj2.__proto__ = proto;
console.log(obj2.b);  // 2

3.3 判断原型关系

function Person(name) {
  this.name = name;
}

const person = new Person('张三');

// instanceof：判断原型链上是否存在构造函数的 prototype
console.log(person instanceof Person);  // true
console.log(person instanceof Object);  // true
console.log(person instanceof Array);  // false

// isPrototypeOf：判断对象是否在另一个对象的原型链上
console.log(Person.prototype.isPrototypeOf(person));  // true
console.log(Object.prototype.isPrototypeOf(person));  // true

// Object.prototype.isPrototypeOf() 可以检测所有对象
console.log(Object.prototype.isPrototypeOf({}));  // true
console.log(Object.prototype.isPrototypeOf([]));  // true
console.log(Object.prototype.isPrototypeOf(function(){}));  // true
console.log(Object.prototype.isPrototypeOf(null));  // false

四、继承的实现方式

4.1 原型链继承

// 原型链继承：将父类实例作为子类原型
function Animal(name) {
  this.name = name;
  this.colors = ['black', 'white'];
}

Animal.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

function Dog(name, breed) {
  this.breed = breed;
}

// 关键：将 Animal 实例作为 Dog 的原型
Dog.prototype = new Animal('狗');
Dog.prototype.constructor = Dog;

const dog1 = new Dog('旺财', '金毛');
const dog2 = new Dog('来福', '哈士奇');

dog1.sayHi();  // 'Hi, I am 狗'
console.log(dog1.breed);  // '金毛'

// 问题一：引用类型属性被所有实例共享
dog1.colors.push('brown');
console.log(dog2.colors);  // ['black', 'white', 'brown']

// 问题二：无法向父类构造函数传参
console.log(dog1.name);  // '狗'（所有实例共享）

缺点：

  • 引用类型属性被所有实例共享
  • 无法向父类构造函数传参

4.2 借用构造函数继承

// 借用构造函数：在子类构造函数中调用父类构造函数
function Animal(name) {
  this.name = name;
  this.colors = ['black', 'white'];
}

Animal.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

function Dog(name, breed) {
  Animal.call(this, name);  // 关键：借用构造函数
  this.breed = breed;
}

const dog1 = new Dog('旺财', '金毛');
const dog2 = new Dog('来福', '哈士奇');

dog1.colors.push('brown');
console.log(dog1.colors);  // ['black', 'white', 'brown']
console.log(dog2.colors);  // ['black', 'white']

// 问题：无法继承父类原型上的方法
// dog1.sayHi();  // TypeError: dog1.sayHi is not a function

优点：

  • 解决了引用类型共享问题
  • 可以向父类传参

缺点：

  • 无法继承父类原型上的方法
  • 每个实例都有一份父类构造函数的副本，浪费内存

4.3 组合继承

// 组合继承：原型链继承 + 借用构造函数继承
function Animal(name) {
  this.name = name;
  this.colors = ['black', 'white'];
}

Animal.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

function Dog(name, breed) {
  Animal.call(this, name);  // 第二次调用 Animal
  this.breed = breed;
}

Dog.prototype = new Animal();  // 第一次调用 Animal
Dog.prototype.constructor = Dog;

const dog1 = new Dog('旺财', '金毛');
const dog2 = new Dog('来福', '哈士奇');

dog1.colors.push('brown');
console.log(dog1.colors);  // ['black', 'white', 'brown']
console.log(dog2.colors);  // ['black', 'white']

dog1.sayHi();  // 'Hi, I am 旺财'
dog2.sayHi();  // 'Hi, I am 来福'

// 问题：调用了两次父类构造函数

优点：

  • 既继承了原型方法，又解决了引用类型共享问题

缺点：

  • 调用了两次父类构造函数

4.4 寄生组合式继承（最优解）

// 寄生组合式继承：优化组合继承
function Animal(name) {
  this.name = name;
  this.colors = ['black', 'white'];
}

Animal.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

function Dog(name, breed) {
  Animal.call(this, name);  // 只调用一次父类构造函数
  this.breed = breed;
}

// 关键：使用 Object.create() 创建原型
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

const dog1 = new Dog('旺财', '金毛');
const dog2 = new Dog('来福', '哈士奇');

dog1.colors.push('brown');
console.log(dog1.colors);  // ['black', 'white', 'brown']
console.log(dog2.colors);  // ['black', 'white']

dog1.sayHi();  // 'Hi, I am 旺财'
dog2.sayHi();  // 'Hi, I am 来福'

console.log(dog1 instanceof Animal);  // true
console.log(dog1 instanceof Dog);  // true

封装继承函数：

function inherit(Child, Parent) {
  // 创建一个原型为 Parent.prototype 的对象
  Child.prototype = Object.create(Parent.prototype);
  // 修复 constructor
  Child.prototype.constructor = Child;
  // 保存对父类的引用
  Child.super = Parent;
  
  // 如果需要继承静态属性
  Object.setPrototypeOf(Child, Parent);
}

// 使用
function Animal(name) {
  this.name = name;
}

Animal.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

function Dog(name, breed) {
  Dog.super.call(this, name);
  this.breed = breed;
}

inherit(Dog, Animal);

4.5 ES6 class 继承

// ES6 class 继承（语法糖，底层仍是原型链）
class Animal {
  constructor(name) {
    this.name = name;
    this.colors = ['black', 'white'];
  }
  
  sayHi() {
    console.log('Hi, I am ' + this.name);
  }
}

class Dog extends Animal {
  constructor(name, breed) {
    super(name);  // 调用父类构造函数
    this.breed = breed;
  }
  
  // 重写父类方法
  sayHi() {
    super.sayHi();  // 调用父类方法
    console.log('I am a ' + this.breed);
  }
}

const dog = new Dog('旺财', '金毛');
dog.sayHi();  
// Hi, I am 旺财
// I am a 金毛

五、继承方式对比

继承方式 关键代码 优点 缺点
原型链继承 Child.prototype = new Parent() 简单 引用共享、无法传参
借用构造函数 Parent.call(this) 可传参、无引用共享 无法继承原型方法
组合继承 原型链 + 借用构造函数 兼具两者优点 调用两次父类构造函数
寄生组合式 Object.create(Parent.prototype) 最优方案 代码稍复杂
ES6 class extends + super 语法简洁、语义清晰 本质仍是原型链

六、实战应用

6.1 判断数组类型

const arr = [1, 2, 3];

// 方法一：instanceof
console.log(arr instanceof Array);  // true

// 方法二：constructor
console.log(arr.constructor === Array);  // true

// 方法三：Object.prototype.toString
console.log(Object.prototype.toString.call(arr));  // '[object Array]'

// 方法四：Array.isArray（推荐）
console.log(Array.isArray(arr));  // true

// 方法五：原型链判断
console.log(Array.prototype.isPrototypeOf(arr));  // true

6.2 实现 instanceof

function myInstanceof(obj, Constructor) {
  // 基本类型直接返回 false
  if (typeof obj !== 'object' || obj === null) {
    return false;
  }
  
  // 获取对象的原型
  let proto = Object.getPrototypeOf(obj);
  
  // 沿着原型链查找
  while (proto !== null) {
    if (proto === Constructor.prototype) {
      return true;
    }
    proto = Object.getPrototypeOf(proto);
  }
  
  return false;
}

// 测试
console.log(myInstanceof([], Array));  // true
console.log(myInstanceof([], Object));  // true
console.log(myInstanceof({}, Array));  // false

6.3 实现 new 操作符

function myNew(Constructor, ...args) {
  // 1. 创建一个新对象，原型指向构造函数的 prototype
  const obj = Object.create(Constructor.prototype);
  
  // 2. 执行构造函数，绑定 this
  const result = Constructor.apply(obj, args);
  
  // 3. 如果构造函数返回对象，则返回该对象；否则返回新对象
  return result !== null && typeof result === 'object' ? result : obj;
}

// 测试
function Person(name, age) {
  this.name = name;
  this.age = age;
}

Person.prototype.sayHi = function() {
  console.log('Hi, I am ' + this.name);
};

const person = myNew(Person, '张三', 18);
person.sayHi();  // 'Hi, I am 张三'
console.log(person instanceof Person);  // true

6.4 实现 Object.create()

function myCreate(proto) {
  if (proto === null || typeof proto !== 'object') {
    throw new TypeError('Object prototype may only be an Object or null');
  }
  
  function F() {}  // 创建临时构造函数
  F.prototype = proto;  // 原型指向传入的对象
  return new F();  // 返回新实例
}

// 测试
const proto = {
  sayHi: function() {
    console.log('Hi');
  }
};

const obj = myCreate(proto);
obj.sayHi();  // 'Hi'
console.log(Object.getPrototypeOf(obj) === proto);  // true

七、常见面试题

7.1 题目一：原型链输出

function Foo() {
  this.a = 1;
}

Foo.prototype.a = 2;
Foo.prototype.b = 3;

const foo = new Foo();

console.log(foo.a);  // 1（实例属性优先）
console.log(foo.b);  // 3（原型属性）
console.log(foo.c);  // undefined（不存在）

console.log(Foo.a);  // undefined（Foo 是函数，没有 a 实例属性）
console.log(Foo.b);  // undefined
console.log(Foo.prototype.a);  // 2
console.log(Foo.prototype.b);  // 3

7.2 题目二：构造函数与原型

function Person(name) {
  this.name = name;
}

Person.prototype.name = '原型name';

const person = new Person('实例name');

console.log(person.name);  // '实例name'
console.log(Person.prototype.name);  // '原型name'
console.log(person.__proto__.name);  // '原型name'

delete person.name;
console.log(person.name);  // '原型name'

delete Person.prototype.name;
console.log(person.name);  // undefined

7.3 题目三：原型链终点

console.log(Object.prototype.__proto__);  // null

function Fn() {}
console.log(Fn.prototype.__proto__ === Object.prototype);  // true
console.log(Fn.__proto__ === Function.prototype);  // true
console.log(Function.prototype.__proto__ === Object.prototype);  // true
console.log(Object.__proto__ === Function.prototype);  // true
console.log(Function.__proto__ === Function.prototype);  // true

7.4 题目四：修改原型的影响

function A() {}
function B() {}

A.prototype = new B();

const a = new A();

console.log(a instanceof A);  // true
console.log(a instanceof B);  // true

A.prototype = {};
console.log(a instanceof A);  // false（原型已改变）
console.log(a instanceof B);  // true（旧原型仍是 B 实例）

const a2 = new A();
console.log(a2 instanceof A);  // true
console.log(a2 instanceof B);  // false

八、总结

8.1 核心概念

概念 说明
prototype 函数特有的属性，指向原型对象
__proto__ 对象特有的属性，指向其原型
constructor 原型对象的属性，指向构造函数
原型链 由 __proto__ 连接形成的链式结构

8.2 原型链结构

实例
  ↓ __proto__
构造函数.prototype
  ↓ __proto__
Object.prototype
  ↓ __proto__
null

8.3 继承实现要点

  1. 原型链继承：Child.prototype = new Parent()
  2. 借用构造函数：Parent.call(this)
  3. 寄生组合式：Child.prototype = Object.create(Parent.prototype)
  4. ES6 class：extends + super

8.4 最佳实践

  1. 优先使用 ES6 class：语法简洁、语义清晰
  2. **避免直接使用 __proto__**：使用 Object.getPrototypeOf() 和 Object.create()
  3. 注意 constructor 指向：重写 prototype 后需要修复
  4. 理解原型链终点：Object.prototype.__proto__ === null
