栈与队列专题

发表于2023|更新于2026|算法
|字数总计:1.2k|阅读时长:6分钟

栈是一种后进先出(LIFO)的数据结构，队列是一种先进先出(FIFO)的数据结构，本文整理了栈与队列相关的经典算法题目。

用栈实现队列

var MyQueue = function () {
  this.stackIn = []
  this.stackOut = []
};

/** 
 * @param {number} x
 * @return {void}
 */
MyQueue.prototype.push = function (x) {
  this.stackIn.push(x)
};

/**
 * @return {number}
 */
MyQueue.prototype.pop = function () {
  // 判断出栈中是否有
  if (this.stackOut.length > 0) {
    return this.stackOut.pop()
  }
  while (this.stackIn.length > 0) {
    this.stackOut.push(this.stackIn.pop())
  }
  return this.stackOut.pop()
};

/**
 * @return {number}
 */
MyQueue.prototype.peek = function () {
  // 查看栈顶元素但是不用弹出
  const x = this.pop();
  this.stackOut.push(x);
  return x;
};

/**
 * @return {boolean}
 */
MyQueue.prototype.empty = function () {
  // 判断是否为空，空返回true
  return this.stackIn.length <= 0 && this.stackOut.length <= 0
};

 * var obj = new MyQueue()
 * obj.push(x)
 * var param_2 = obj.pop()
 * var param_3 = obj.peek()

用队列实现栈

将队列中的元素全部弹出加到队列的末尾

var MyStack = function () {
  this.qunue = []
};

/** 
 * @param {number} x
 * @return {void}
 */
MyStack.prototype.push = function (x) {
  this.qunue.push(x)
};

/**
 * @return {number}
 */
MyStack.prototype.pop = function () {
  // 将前面元素都弹出，加入到队列末尾，然后将最后一个元素弹出
  let size = this.qunue.length
  size--
  while (size--) {
    this.qunue.push(this.qunue.shift())
  }
  return this.qunue.shift()
};

/**
 * @return {number}
 */
MyStack.prototype.top = function () {
  // 查看栈顶元素，不弹出
  const x = this.pop();
  this.qunue.push(x);
  return x;
};

/**
 * @return {boolean}
 */
MyStack.prototype.empty = function () {
  return this.qunue.length <= 0
};

有效括号

/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function (s) {
  // 长度为奇数
  if (s.length % 2 !== 0) {
    return false
  }
  // 用栈去匹配
  const stack = []
  for (let i = 0; i < s.length; i++) {
    if (s[i] === '(') {
      stack.push(')')
    }
    if (s[i] === '[') {
      stack.push(']')
    }
    if (s[i] === '{') {
      stack.push('}')
    }
    if (s[i] === ')' || s[i] === ']' || s[i] === '}') {
      if (stack[stack.length - 1] !== s[i]) {
        return false
      } else {
        stack.pop()
      }
    }
  }
  return stack.length == 0
};

删除字符串中所有相邻重复项

每次匹配栈顶元素，如果匹配上就移除

/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
  const stack = []
  for (let i = 0; i < s.length; i++) {
    // 相同字母，移除
    if (stack.length > 0 && s[i] === stack[stack.length - 1]) {
      // 重复元素，栈顶弹出
      stack.pop()
    } else {
      stack.push(s[i])
    }
  }
  return stack.join('')
};

逆波兰表达式求值

遇见数字添加到栈里边，遇见运算符号从栈里边取出两个数进行运算，并把结果压入栈

/**
 * @param {string[]} tokens
 * @return {number}
 */
var evalRPN = function (tokens) {
  const stack = [];
  for (const token of tokens) {
    if (isNaN(Number(token))) { // 非数字
      // n2表示后边的数字
      const n2 = stack.pop(); // 出栈两个数字
      const n1 = stack.pop();
      switch (token) { // 判断运算符类型，算出新数入栈
        case "+":
          stack.push(n1 + n2);
          break;
        case "-":
          stack.push(n1 - n2);
          break;
        case "*":
          stack.push(n1 * n2);
          break;
        case "/":
          stack.push(n1 / n2 | 0);
          break;
      }
    } else { // 数字
      stack.push(Number(token));
    }
  }
  return stack[0]; // 因没有遇到运算符而待在栈中的结果
};

滑动窗口最大值

超时解法：

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var maxSlidingWindow = function (nums, k) {
  const res = [];
  for (let i = 0; i < nums.length - k + 1; i++) {
    const temNum = nums.slice(i, i + k);
    const tem = Math.max(...temNum);
    res.push(tem);
  }
  return res
};

单调队列解法：将数添加进队列中，如果添加的数比之前的大，就把之前的全部pop出去，出口一直保存着最大元素

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var maxSlidingWindow = function (nums, k) {
  class SignleStack {
    constructor() {
      this.queue = []
    }
    push(value) {
      let last = this.queue[this.queue.length - 1]
      while (last !== undefined && value > last) {
        this.queue.pop()
        last = this.queue[this.queue.length - 1]
      }
      this.queue.push(value)
    }
    refresh(value) {
      const max = this.getMax()
      if (max === value) {
        this.queue.shift()
      }
    }
    getMax() {
      return this.queue[0]
    }
  }
  const res = []
  const signleStack = new SignleStack()
  // 加入k个元素
  let i = 0
  while (i < k) {
    signleStack.push(nums[i])
    i++
  }
  let j = 0
  res.push(signleStack.getMax())
  while (i < nums.length) {
    signleStack.push(nums[i])
    // 保证单调栈中的元素永远是当前滑动窗口的中的值[1,-1],1的情况
    signleStack.refresh(nums[j])
    res.push(signleStack.getMax())
    i++
    j++
  }
  return res
};

前k个高频元素

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var topKFrequent = function (nums, k) {
  const map = new Map()
  for (let i = 0; i < nums.length; i++) {
    map.set(nums[i], (map.get(nums[i]) || 0) + 1)
  }
  const temArr = Array.from(map)
  temArr.sort((a, b) => {
    return b[1] - a[1]
  })
  const res = [];
  for (let i = 0; i < k; i++) {
    res.push(temArr[i][0]);
  }
  return res
};
