哈希表专题

发表于2023|更新于2026|算法
|字数总计:1.3k|阅读时长:6分钟

哈希表是一种根据关键码值(Key value)而直接进行访问的数据结构，本文整理了哈希表相关的经典算法题目。

哈希表基础

哈希表常用的数据结构有：数组、Set、Map

有效的字母异位词

方法一：用数组

定义一个可以存放26个字母的数组，第一个字符串做加法，第二个字符串做减法，最后判断数组中元素是不是全部是0

方法二：字符串排序

/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function (s, t) {
  if (s.length !== t.length || s === t) {
    return false;
  }
  const arrS = s.split('');
  const arrT = t.split('');
  arrS.sort();
  arrT.sort();
  for (let i = 0; i < arrS.length; i++) {
    if (arrS[i] !== arrT[i]) {
      return false;
    }
  }
  return true;
};

两个数组的交集

方法一：哈希表(set)

将第一个数组中的数据存放到哈希表中，遍历第二个数组然后出现过就加到结果数组中

方法二：set+排序

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersection = function (nums1, nums2) {
  nums1 = Array.from(new Set(nums1));
  nums2 = Array.from(new Set(nums2));
  const tem = [...nums1, ...nums2];
  tem.sort((a, b) => {
    return a - b;
  })
  const res = []
  tem.forEach((elem) => {
    if (tem.indexOf(elem) !== tem.lastIndexOf(elem) && res.indexOf(elem) === -1) {
      res.push(elem);
    }
  })
  return res;
};

两数之和

/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
  let map = new Map();
  for (let i = 0, len = nums.length; i < len; i++) {
    if (map.has(target - nums[i])) {
      return [map.get(target - nums[i]), i];
    } else {
      map.set(nums[i], i);
    }
  }
  return [];
};

四数之和

  1. 暴力解法，四重for循环
  2. 遍历两个数组之和相加，存储到map中，另外需要存储出现的次数，再遍历另外两个数组，去map中查看是否有需要的元素
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @param {number[]} nums3
 * @param {number[]} nums4
 * @return {number}
 */
var fourSumCount = function (nums1, nums2, nums3, nums4) {
  const map = new Map();
  // 遍历前两个数组将和存储到map中，并统计次数
  for (let i = 0; i < nums1.length; i++) {
    for (let j = 0; j < nums2.length; j++) {
      const sum = nums1[i] + nums2[j]
      map.set(sum, (map.get(sum) || 0) + 1)
    }
  }
  // 遍历剩余两个数组，判断map中是否具有目标元素(出现多少次就加多少个)
  let res = 0
  for (let i = 0; i < nums3.length; i++) {
    for (let j = 0; j < nums4.length; j++) {
      const sum = 0 - (nums3[i] + nums4[j]);
      res += map.get(sum) || 0
    }
  }
  return res
};

三数之和

一个数组中找到三个元素，三个元素之和为0

方法一：hash法

需要考虑去重，a、b、c都需要考虑去重

方法二：双指针

对数组进行排序
a值固定，b和c用双指针去找，如果三个数的值相加大了，就移动右指针，如果小了就移动左指针

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var threeSum = function (nums) {
  if (nums.length < 3) {
    return []
  }
  // 数组排序
  nums.sort((a, b) => {
    return a - b
  })
  // 目标结果是0，如果第一个元素都大于0，那么不可能获得结果
  if (nums[0] > 0) {
    return []
  }
  const res = []
  for (let i = 0; i < nums.length; i++) {
    /* 
        判断第一个数是否重复,如果判断后一个的话判断的是数组中是否有重复元素，
        [-1,-1,2]
        所有判断前一个是否用过
    */
    if (i > 0 && nums[i] === nums[i - 1]) {
      continue;
    }
    let left = i + 1
    let right = nums.length - 1
    while (right > left) {
      const sum = nums[i] + nums[right] + nums[left]
      if (sum > 0) {
        right--;
      } else if (sum < 0) {
        left++;
      } else if (sum == 0) {
        res.push([nums[i], nums[left], nums[right]])
        /* 
          收获结果之后再进行去重操作
          去重,如果数组中元素是[0,-1,-1,-1,-1,1,1,1,1]
          left指针和right指针移动后指针指向的值是相同的，需要进行去重操作
        */
        while (left < right && nums[left] == nums[left + 1]) {
          left++
        }
        while (left < right && nums[right] == nums[right - 1]) {
          right--
        }
        // 去重完成之和继续循环
        left++
        right--
      }
    }
  }
  return res
};

四数之和（数组）

/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[][]}
 */
var fourSum = function (nums, target) {
  if (nums.length < 4) {
    return []
  }
  nums.sort((a, b) => {
    return a - b
  })
  // 剪枝操作,这里需要加上target大于0
  if (nums[0] > target && target > 0) {
    return []
  }
  /*
    双重for循环+定义两指针去寻找
    */
  const res = []
  for (let k = 0; k < nums.length; k++) {
    // 去重,判断是否与前一个数相等相等表示已经用过了
    if (k > 0 && nums[k] === nums[k - 1]) {
      continue
    }
    for (let i = k + 1; i < nums.length; i++) {
      // 去重,判断是否与前一个数相等相等表示已经用过了
      if (i > k + 1 && nums[i] === nums[i - 1]) continue;
      let left = i + 1;
      let right = nums.length - 1;
      while (right > left) {
        // 获得结果
        const sum = nums[k] + nums[i] + nums[left] + nums[right]
        if (sum > target) {
          right--
        }
        if (sum < target) {
          left++
        }
        if (sum === target) {
          res.push([nums[k], nums[i], nums[left], nums[right]])
          // 去重
          while (right > left && nums[left] === nums[left + 1]) {
            left++
          }
          while (right > left && nums[right] === nums[right - 1]) {
            right--
          }
          left++
          right--
        }
      }
    }
  }
  return res
};
