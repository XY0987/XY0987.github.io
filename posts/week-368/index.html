元素和最小的山形三元组 II

const nums = [1, 3, 2, 2];

const dl = new Array(nums.length).fill(0);
const dr = new Array(nums.length).fill(0);
dl[0] = nums[0];
dr[nums.length - 1] = nums[nums.length - 1];

for (let i = 1; i < nums.length; i++) {
  dl[i] = Math.min(dl[i - 1], nums[i]);
}

for (let i = nums.length - 2; i >= 0; i--) {
  dr[i] = Math.min(dr[i + 1], nums[i]);
}

let res = Infinity;
for (let j = 1; j < nums.length - 1; j++) {
  if (dl[j - 1] < nums[j] && nums[j] > dr[j + 1]) {
    res = Math.min(res, nums[j] + dl[j - 1] + dr[j + 1]);
  }
}
return res === Infinity ? -1 : res;
