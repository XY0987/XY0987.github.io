二叉树

对称二叉树

只能使用后序遍历（左右中）
需要比对孩子节点的信息是否相等，返回给上一个节点

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function (root) {
  // 定义函数判断是否相等，传入根节点的左右节点
  function getRes(left, right) {
    // 左边节点为空，右边不为空
    if (!left && right) {
      return false;
    }
    // 左边不为空，右边为空
    if (left && !right) {
      return false;
    }
    // 左右节点都为空
    if (!left && !right) {
      return true;
    }
    // 左右节点都不为空，但是值不相等
    if (left && right && left.val !== right.val) {
      return false;
    }
    // 先比对外侧节点（左节点的左孩子和右节点的右孩子）
    const yn1 = getRes(left.left, right.right);
    // 比对内侧节点（左节点的右孩子，右节点的左孩子）
    const yn2 = getRes(left.right, right.left);
    return yn1 && yn2;
  }
  return getRes(root.left, root.right);
};

二叉树的最大深度

深度：二叉树任意一个节点到跟节点的距离
高度：二叉树种任意一个节点到叶子节点的距离

根节点的高度就是二叉树的最大深度

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function (root) {
  function getHeight(node) {
    if (!node) {
      return 0;
    }
    // 左节点
    const leftHeight = getHeight(node.left);
    // 右节点
    const rightHeight = getHeight(node.right);
    // 中节点(处理逻辑,1加上左右节点的最大值)
    const height = 1 + Math.max(leftHeight, rightHeight);
    return height;
  }
  return getHeight(root);
};

二叉树的最小深度

根节点到最近的叶子节点的距离
后续遍历

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var minDepth = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    const leftRes = getRes(node.left);
    const rightRes = getRes(node.right);
    // 如果左子树为空，但是右子树不为空(排除掉左边为空的错误情况)
    if (!node.left && node.right) {
      return 1 + rightRes;
    }
    if (!node.right && node.left) {
      return 1 + leftRes;
    }
    // 两边的子树都不为空，取最小值
    return 1 + Math.min(leftRes, rightRes);
  }
  return getRes(root);
};

完全二叉树节点的数量

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
// 普通二叉树的方式写(后序遍历)
var countNodes = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    const leftNum = getRes(node.left);
    const rightNum = getRes(node.right);
    // 左子树的数量加上右子树的数量加上本身节点的数量
    return leftNum + rightNum + 1;
  }
  return getRes(root);
};

判断平衡二叉树

任意节点它的左右子树的高度差不超过 1(后序遍历)

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isBalanced = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    // 后续遍历，如果不是一个平衡二叉树，就返回-1
    const leftHeight = getRes(node.left);
    if (leftHeight === -1) {
      return -1;
    }
    const rightHeight = getRes(node.right);
    if (rightHeight === -1) {
      return -1;
    }
    if (Math.abs(leftHeight - rightHeight) > 1) {
      return -1;
    }
    // 获取本层的高度(高度差不超过1)(左子树和右子树的高度的最大值+1)
    return Math.max(leftHeight, rightHeight) + 1;
  }
  return getRes(root) === -1 ? false : true;
};

二叉树的所有路径

前序遍历，中左右，前序遍历才有将父节点指向子节点
回溯

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {string[]}
 */
var binaryTreePaths = function (root) {
  // 用于存放结果
  const res = [];
  function getRes(node, path, res) {
    // 添加操作再这里的原因是遍历到叶子节点就结束了
    path.push(node.val);
    // 左右孩子都为空，遍历到叶子节点
    if (!node.left && !node.right) {
      res.push(path.join("->").toString());
      return;
    }
    if (node.left) {
      getRes(node.left, path, res);
      // 回溯过程
      path.pop();
    }
    if (node.right) {
      getRes(node.right, path, res);
      path.pop();
    }
  }
  getRes(root, [], res);
  return res;
};

左叶子之和

叶子节点（左右节点都为空）

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var sumOfLeftLeaves = function (root) {
  function getRes(node) {
    /**
        遍历到叶子节点的上一层节点，获取到值，叶子节点以及当前节点为空的都返回0
        叶子节点的上一层节点的左节点，就是要求的值
         */
    if (!node) {
      return 0;
    }
    if (!node.left && !node.right) {
      return 0;
    }
    let leftNum = getRes(node.left);
    // 叶子节点
    if (node.left && !node.left.left && !node.left.right) {
      leftNum = node.left.val;
    }
    let rightNum = getRes(node.right);
    return leftNum + rightNum;
  }
  return getRes(root);
};

找二叉树左下角的值

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
// 用层序遍历
var findBottomLeftValue = function (root) {
  // 使用队列去辅助，弹出原生
  const queue = [];
  const res = [];
  if (!root) {
    return res;
  }
  queue.push(root);
  // 遍历对列
  while (queue.length > 0) {
    // 保存本层节点的数量(因为之后遍历的时候会添加下一层的数据)
    let len = queue.length;
    const resItem = [];
    // 将本层数据弹出，并保存进数组
    while (len > 0) {
      const node = queue.shift();
      resItem.push(node.val);
      // 将下一层的左右节点压入
      if (node.left) {
        queue.push(node.left);
      }
      if (node.right) {
        queue.push(node.right);
      }
      len--;
    }
    res.push(resItem);
  }
  return res[res.length - 1][0];
};

路径总和

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} targetSum
 * @return {boolean}
 */
var hasPathSum = function (root, targetSum) {
  const res = [];
  // 遍历所有路径
  function getPath(node, path) {
    path.push(node.val);
    if (!node.left && !node.right) {
      res.push(
        path.reduce((a, b) => {
          return a + b;
        }, 0)
      );
    }
    // 去遍历左节点
    if (node.left) {
      getPath(node.left, path);
      path.pop();
    }
    if (node.right) {
      getPath(node.right, path);
      path.pop();
    }
  }
  if (!root) {
    return false;
  }
  getPath(root, []);
  return res.includes(targetSum);
};

从中序和后序遍历中构造二叉树

中序是左中右
后序是左右中
先从后序中找到中间节点(后序的最后一位就是二叉树的根节点)，然后再根据前序遍历中确定左子树和右子树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} inorder
 * @param {number[]} postorder
 * @return {TreeNode}
 */
var buildTree = function (inorder, postorder) {
  if (!inorder.length) return null;
  const rootVal = postorder.pop(); // 从后序遍历的数组中获取中间节点的值， 即数组最后一个值
  let rootIndex = inorder.indexOf(rootVal); // 获取中间节点在中序遍历中的下标
  const root = new TreeNode(rootVal); // 创建中间节点
  root.left = buildTree(
    inorder.slice(0, rootIndex),
    postorder.slice(0, rootIndex)
  ); // 创建左节点
  root.right = buildTree(
    inorder.slice(rootIndex + 1),
    postorder.slice(rootIndex)
  ); // 创建右节点
  return root;
};

最大二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} nums
 * @return {TreeNode}
 */
var constructMaximumBinaryTree = function (nums) {
  // 没有元素之间返回null
  if (nums.length === 0) {
    return null;
  }
  // 找到数组中的最大值
  const maxValue = Math.max(...nums);
  // 找到最大值的下标
  const index = nums.indexOf(maxValue);
  const root = new TreeNode(maxValue);
  // 递归处理左节点
  root.left = constructMaximumBinaryTree(nums.slice(0, index));
  // 递归处理右节点
  root.right = constructMaximumBinaryTree(nums.slice(index + 1));
  return root;
};

合并二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root1
 * @param {TreeNode} root2
 * @return {TreeNode}
 */
var mergeTrees = function (root1, root2) {
  // 遇到空节点返回另一个树的节点
  if (!root1) {
    return root2;
  }
  if (!root2) {
    return root1;
  }
  // 两个节点的值都不为空，合并(直接修改root1的值)
  root1.val += root2.val;
  root1.left = mergeTrees(root1.left, root2.left);
  root1.right = mergeTrees(root1.right, root2.right);
  return root1;
};

二叉搜索树的搜索

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} val
 * @return {TreeNode}
 */
var searchBST = function (root, val) {
  // 二叉搜索树的左子树都要比根节点小，右子树都要比根节点大
  if (root.val === val) {
    return root;
  }
  if (root.left && root.val > val) {
    return searchBST(root.left, val);
  }
  if (root.right && root.val < val) {
    return searchBST(root.right, val);
  }
  return null;
};

验证二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isValidBST = function (root) {
  /** 
    中序遍历，先遍历左节点，再遍历中节点，最后遍历右节点
    如果中序遍历出来的结果是单调递增的，就表示是二叉搜索树
    (同时需要验证是否有重复元素)
    */
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    getVal(node.left);
    res.push(node.val);
    getVal(node.right);
  }
  getVal(root);
  // 重复元素返回false
  const temArr = Array.from(new Set(res));
  if (temArr.length !== res.length) {
    return false;
  }
  const resStrOri = res.join("");
  const resStrSort = res
    .sort((a, b) => {
      return a - b;
    })
    .join("");
  return resStrOri === resStrSort;
};

二叉搜索树的最小绝对差

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var getMinimumDifference = function (root) {
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    res.push(node.val);
    getVal(node.left);
    getVal(node.right);
  }
  getVal(root);
  res.sort((a, b) => {
    return a - b;
  });
  let min = 0;
  let left = 1;
  for (let i = 0; i < res.length - 1; i++) {
    if (min === 0) {
      min = res[left] - res[i];
    } else {
      min = Math.min(min, res[left] - res[i]);
    }
    left++;
  }
  return min;
};

二叉搜索树中的众数

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var findMode = function (root) {
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    getVal(node.left);
    res.push(node.val);
    getVal(node.right);
  }
  getVal(root);
  if (res.length === 1) {
    return res;
  }
  const map = new Map();
  res.forEach((item) => {
    map.set(item, (map.get(item) | 0) + 1);
  });
  const arr = Array.from(map).sort((a, b) => {
    return b[1] - a[1];
  });
  const result = [];
  arr.forEach((value) => {
    if (value[1] === arr[0][1]) {
      result.push(value[0]);
    }
  });
  return result;
};

给你一个长度为 n 下标从 0 开始的整数数组 maxHeights 。
你的任务是在坐标轴上建 n 座塔。第 i 座塔的下标为 i ，高度为 heights[i] 。
如果以下条件满足，我们称这些塔是 美丽 的：

  1. 1 <= heights[i] <= maxHeights[i]
  2. heights 是一个 山状 数组。

如果存在下标 i 满足以下条件，那么我们称数组 heights 是一个 山状 数组：

  • 对于所有 0 < j <= i ，都有 heights[j - 1] <= heights[j]
  • 对于所有 i <= k < n - 1 ，都有 heights[k + 1] <= heights[k]

请你返回满足 美丽塔 要求的方案中，高度和的最大值 。
示例 1：
输入：maxHeights = [5,3,4,1,1] 输出：13 解释：和最大的美丽塔方案为 heights = [5,3,3,1,1] ，这是一个美丽塔方案，因为： - 1 <= heights[i] <= maxHeights[i] - heights 是个山状数组，峰值在 i = 0 处。 13 是所有美丽塔方案中的最大高度和。
示例 2：
输入：maxHeights = [6,5,3,9,2,7] 输出：22 解释： 和最大的美丽塔方案为 heights = [3,3,3,9,2,2] ，这是一个美丽塔方案，因为： - 1 <= heights[i] <= maxHeights[i] - heights 是个山状数组，峰值在 i = 3 处。 22 是所有美丽塔方案中的最大高度和。
示例 3：
输入：maxHeights = [3,2,5,5,2,3] 输出：18 解释：和最大的美丽塔方案为 heights = [2,2,5,5,2,2] ，这是一个美丽塔方案，因为： - 1 <= heights[i] <= maxHeights[i] - heights 是个山状数组，最大值在 i = 2 处。 注意，在这个方案中，i = 3 也是一个峰值。 18 是所有美丽塔方案中的最大高度和。

/**
 * @param {number[]} maxHeights
 * @return {number}
 */
// 暴力解法，把当前节点当作中心，分别遍历左右两边满足要求的数据求和，最终取出最大值
var maximumSumOfHeights = function (maxHeights) {
  const res = [];
  for (let i = 0; i < maxHeights.length; i++) {
    let sum = 0;
    const currentValue = maxHeights[i];
    let temValue = currentValue;
    // 去遍历左边满足元素的情况，然后相加
    for (let j = i - 1; j >= 0; j--) {
      // 左右两边的山状数组都不能大于中间的
      temValue = Math.min(temValue, maxHeights[j]);
      sum += temValue;
    }
    // 遍历右边重新置回当前变量
    temValue = currentValue;
    // 遍历右边
    for (let j = i + 1; j < maxHeights.length; j++) {
      temValue = Math.min(temValue, maxHeights[j]);
      sum += temValue;
    }
    res.push(currentValue + sum);
  }
  return Math.max(...res);
};
