/**
 * @param {number[]} ratings
 * @return {number}
 */
var candy = function (ratings) {
  const res = new Array(ratings.length).fill(1);
  for (let i = 1; i < ratings.length; i++) {
    if (ratings[i] > ratings[i - 1]) {
      res[i] = res[i - 1] + 1;
    }
  }
  for (let i = ratings.length - 2; i >= 0; i--) {
    if (ratings[i] > ratings[i + 1]) {
      res[i] = Math.max(res[i + 1] + 1, res[i]);
    }
  }
  return res.reduce((prev, next) => {
    return prev + next;
  }, 0);
};

柠檬水找零

/**
 * @param {number[]} bills
 * @return {boolean}
 */
var lemonadeChange = function (bills) {
  let five = 0;
  let ten = 0;
  let twenty = 0;
  for (let i = 0; i < bills.length; i++) {
    const item = bills[i]
    if (item === 5) {
      five++;
    }
    if (item === 10) {
      if (five === 0) {
        return false;
      }
      five--;
      ten++;
    }
    if (item === 20) {
      if (ten > 0 && five > 0) {
        ten--;
        five--;
        twenty++;
      } else if (five >= 3) {
        five -= 3;
        twenty++;
      } else {
        return false;
      }
    }
  }
  return true;
};

根据身高重建队列

/**
 * @param {number[][]} people
 * @return {number[][]}
 */
var reconstructQueue = function (people) {
  let queue = []
  // 根据身高进行排序，如果身高一样，就根据k来排序
  people.sort((a, b) => {
    if (b[0] !== a[0]) {
      return b[0] - a[0]
    } else {
      return a[1] - b[1]
    }

  })
  // 插入到指定位置
  for (let i = 0; i < people.length; i++) {
    queue.splice(people[i][1], 0, people[i])
  }
  return queue
};

用最少数量的箭引爆气球

/**
 * @param {number[][]} points
 * @return {number}
 */
var findMinArrowShots = function (points) {
  points.sort((a, b) => {
    return a[0] - b[0];
  });
  let result = 1;
  for (let i = 1; i < points.length; i++) {
    // 与前一个区间进行边界比较，如果不相邻就直接结果加一
    if (points[i][0] > points[i - 1][1]) {
      result++;
    } else {
      // 否则更新区间（取最小值）,每次比较都是与前一个区间比较
      points[i][1] = Math.min(points[i - 1][1], points[i][1]);
    }
  }
  return result
};

无重叠区间

/**
 * @param {number[][]} intervals
 * @return {number}
 */
var eraseOverlapIntervals = function (intervals) {
  /* 
    求出重叠区间
    以左边界排序
    
    如果区间的左端点大于等于前一个区间的右端点就表示没有重叠区间
    */
  intervals.sort((a, b) => {
    return a[0] - b[0];
  });
  let res = 0;
  for (let i = 1; i < intervals.length; i++) {
    if (intervals[i][0] >= intervals[i - 1][1]) {
    } else {
      res++;
      // 有重叠区间时更新当前区间的右边界，排除下一个区间与前一个区间重合，每次比较都是与前一个区间比较
      intervals[i][1] = Math.min(intervals[i - 1][1], intervals[i][1]);
    }
  }
  return res;
};

划分字母区间

/**
 * @param {string} s
 * @return {number[]}
 */
var partitionLabels = function (s) {
  /* 
    记录每个字母出现的最后的位置
    */
  let hash = {};
  for (let i = 0; i < s.length; i++) {
    hash[s[i]] = i;
  }
  let res = [];
  let left = 0;
  let right = 0;
  for (let i = 0; i < s.length; i++) {
    // 取出现过的最后的一个位置
    right = Math.max(right, hash[s[i]]);
    if (right === i) {
      // 存放结果(i从0开始，所以需要加一)
      res.push(right - left + 1);
      // left定义的是左区间
      left = i + 1;
    }
  }
  return res;
};

合并区间

/**
 * @param {number[][]} intervals
 * @return {number[][]}
 */
var merge = function (intervals) {
  let res = []
  // 根据左边界排序
  intervals.sort((a, b) => {
    return a[0] - b[0];
  });
  // 默认等于第一个区间
  let prev = intervals[0];
  for (let i = 1; i < intervals.length; i++) {
    let cur = intervals[i];
    // 每次都是与上一个区间相比较，如果有重复的，就更新值，直到没有重复的，再收入结果
    if (prev[1] >= cur[0]) { // 有重合
      prev[1] = Math.max(cur[1], prev[1]);
    } else {       // 不重合，prev推入res数组 
      res.push(prev);
      prev = cur;  // 更新 prev
    }
  }
  res.push(prev);
  return res;
};

单调自增的数字

从后往前遍历，去找到需要减1的地方，前边减一，然后后边的值统一改为9

/**
 * @param {number} n
 * @return {number}
 */
var monotoneIncreasingDigits = function (n) {
  n = String(n);
  n = n.split("").map((item) => {
    return +item;
  });
  // 初始值为最大值，避免后续赋值为9出现问题
  let flag = Infinity;
  for (let i = n.length - 1; i > 0; i--) {
    if (n[i - 1] > n[i]) {
      flag = i;
      n[i - 1]--;
    }
  }

  for (let i = flag; i < n.length; i++) {
    n[i] = 9;
  }
  n = n.join("");
  return +n
};

监督二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var minCameraCover = function (root) {
  let result = 0;
  function traversal(cur) {
    // 空节点
    if (cur === null) {
      return 2;
    }
    let left = traversal(cur.left);
    let right = traversal(cur.right);
    if (left === 2 && right === 2) {
      return 0;
    }
    if (left === 0 || right === 0) {
      result++;
      return 1;
    }
    if (left === 1 || right === 1) {
      return 2;
    }
    return -1;
  }
  if (traversal(root) === 0) {
    result++;
  }
  return result;
};

使用最小花费爬楼梯

/**
 * @param {number[]} cost
 * @return {number}
 */
var minCostClimbingStairs = function (cost) {
  /* 
    dp数组的含义是到达i位置花费为dp[i]
    跳到i阶之后所需要的最小花费从dp[i-1]+cost[i-1]和dp[i-2]+cost[i-2]中取
    第i阶可以从i-1或i-2中跳到
    */
  // 往上跳才花费体力（起点可以从0或者1开始）
  const dp = [0, 0];
  for (let i = 2; i <= cost.length; i++) {
    dp[i] = Math.min(dp[i - 1] + cost[i - 1], dp[i - 2] + cost[i - 2]);
  }
  return dp[cost.length]
};

不同路径

/**
 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var uniquePaths = function (m, n) {
  /* 
    dp数组的含义：
    从0 0开始到i j的路径总和
    
    dp[i][j] 只能从dp[i-1][j]或者dp[i][j-1]走到，二者相加就是该点的路径总和
    */

  // 全部初始化为1
  const dp = new Array(m).fill(1).map(() => Array(n).fill(1));
  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      dp[i][j] = dp[i][j - 1] + dp[i - 1][j];
    }
  }
  return dp[m - 1][n - 1]
};

不同路径二

/**
 * @param {number[][]} obstacleGrid
 * @return {number}
 */
var uniquePathsWithObstacles = function (obstacleGrid) {
  /* 
    dp数组表示 0 0位置到i j的路径和
    */

  // 初始化时，遇到障碍，之后的都应该时0(第一行和第一列)
  const m = obstacleGrid.length;
  const n = obstacleGrid[0].length;
  const dp = Array(m)
    .fill()
    .map(() => Array(n).fill(0));

  // 初始化行
  for (let i = 0; i < m && obstacleGrid[i][0] === 0; i++) {
    dp[i][0] = 1;
  }
  // 初始化列
  for (let i = 0; i < n && obstacleGrid[0][i] === 0; i++) {
    dp[0][i] = 1;
  }

  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      // 如果路径数组该位置有障碍物，不用计算
      if (obstacleGrid[i][j] === 0) {
        dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
      }
    }
  }
  return dp[m - 1][n - 1]
};
