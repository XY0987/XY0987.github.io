/**
 * @param {number[]} g
 * @param {number[]} s
 * @return {number}
 */
var findContentChildren = function (g, s) {
  g.sort((a, b) => {
    return a - b;
  });
  s.sort((a, b) => {
    return a - b;
  });
  let index = s.length - 1;
  let res = 0;
  for (let i = g.length - 1; i >= 0; i--) {
    // 先满足大胃口孩子
    if (index >= 0 && s[index] >= g[i]) {
      res++;
      index--;
    }
  }
  return res;
};

摆动序列

/**
 * @param {number[]} nums
 * @return {number}
 */
var wiggleMaxLength = function (nums) {
  /* 
1. 上下坡有平坡
2. 首尾元素
3. 单调坡度有平坡
*/

  if (nums.length === 1) {
    return 1;
  }
  // 相当于延长了前边
  let prediff = 0;
  let curdiff = 0;
  // 默认算上最后一个元素
  let res = 1;
  for (let i = 0; i < nums.length - 1; i++) {
    curdiff = nums[i + 1] - nums[i];
    if ((prediff >= 0 && curdiff < 0) || (prediff <= 0 && curdiff > 0)) {
      res++;
      prediff = curdiff;
    }
  }
  return res;
};

最大子序和

/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
  /* 
暴力解法：双重for循环，去遍历所有子数组，求和
贪心解法：不断的去求和，如果求得的连续和小于0了，那就从下一个数开始
*/
  let res = -Infinity;
  let count = 0;
  for (let i = 0; i < nums.length; i++) {
    count += nums[i];
    if (count > res) {
      res = count;
    }
    // 连续和为负数了，重新赋值为0，相当于从下一个数开始
    if (count < 0) {
      count = 0;
    }
  }
  return res;
};

买卖股票的最佳时机二

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /*
    尽可能的多卖，遇到正数相加，负数不要 
    算出每天的利润(前一天买入，后一天卖出)，然后只取正数
    */
  let res = 0;
  for (let i = 0; i < prices.length; i++) {
    const num = prices[i] - prices[i - 1];
    res += num > 0 ? num : 0;
  }
  return res;
};

跳跃游戏

/**
 * @param {number[]} nums
 * @return {boolean}
 */
var canJump = function (nums) {
  /* 
    查看每个点的覆盖范围，根据覆盖范围是否能到达终点去判断
  */
  let cover = 0;
  if (nums.length === 1) {
    return true;
  }
  // 只在覆盖范围内移动
  for (let i = 0; i <= cover; i++) {
    // 取最大的覆盖范围
    cover = Math.max(cover, i + nums[i]);
    // 覆盖范围能到达终点
    if (cover >= nums.length - 1) {
      return true;
    }
  }
  return false;
};

跳跃游戏二

/**
 * @param {number[]} nums
 * @return {number}
 */
var jump = function (nums) {
  /* 
        先记录当前值的覆盖范围，如果当前覆盖范围没有到终点
        就启用下一步的覆盖范围，每次都是取最大值
    */
  let curIndex = 0;
  let nextIndex = 0;
  let steps = 0;
  for (let i = 0; i < nums.length - 1; i++) {
    nextIndex = Math.max(nums[i] + i, nextIndex);
    // 等于时还没有到达终点，启用下一个(每次都是保存最大的值)
    if (i === curIndex) {
      // 如果之前就走到终点了，后边的不会走到这里
      curIndex = nextIndex;
      steps++;
    }
  }

  return steps;
};

k 次取反后最大化的数组和

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var largestSumAfterKNegations = function (nums, k) {
  for (let i = 0; i < k; i++) {
    nums.sort((a, b) => {
      return a - b;
    });
    nums[0] = -nums[0];
  }
  return nums.reduce((prev, next) => {
    return prev + next;
  }, 0);
};

加油站

/**
 * @param {number[]} gas
 * @param {number[]} cost
 * @return {number}
 */
var canCompleteCircuit = function (gas, cost) {
  /* 
    统计每个站点剩余的油量，如果剩余的为负数，就将下一个站点作为起始位置
    */
  let curSum = 0;
  let totalSum = 0;
  let start = 0;
  for (let i = 0; i < gas.length; i++) {
    curSum += gas[i] - cost[i];
    totalSum += gas[i] - cost[i];
    if (curSum < 0) {
      start = i + 1;
      curSum = 0;
    }
  }
  if (totalSum < 0) {
    return -1;
  }

  return start;
};

/**
 * @param {number[]} ratings
 * @return {number}
 */
var candy = function (ratings) {
  const res = new Array(ratings.length).fill(1);
  for (let i = 1; i < ratings.length; i++) {
    if (ratings[i] > ratings[i - 1]) {
      res[i] = res[i - 1] + 1;
    }
  }
  for (let i = ratings.length - 2; i >= 0; i--) {
    if (ratings[i] > ratings[i + 1]) {
      res[i] = Math.max(res[i + 1] + 1, res[i]);
    }
  }
  return res.reduce((prev, next) => {
    return prev + next;
  }, 0);
};

/**
 * @param {number[]} bills
 * @return {boolean}
 */
var lemonadeChange = function (bills) {
  let five = 0;
  let ten = 0;
  let twenty = 0;
  for (let i = 0; i < bills.length; i++) {
    const item = bills[i]
    if (item === 5) {
      five++;
    }
    if (item === 10) {
      if (five === 0) {
        return false;
      }
      five--;
      ten++;
    }
    if (item === 20) {
      if (ten > 0 && five > 0) {
        ten--;
        five--;
        twenty++;
      } else if (five >= 3) {
        five -= 3;
        twenty++;
      } else {
        return false;
      }
    }
  }
  return true;
};

/**
 * @param {number[][]} people
 * @return {number[][]}
 */
var reconstructQueue = function (people) {
  let queue = []
  // 根据身高进行排序，如果身高一样，就根据k来排序
  people.sort((a, b) => {
    if (b[0] !== a[0]) {
      return b[0] - a[0]
    } else {
      return a[1] - b[1]
    }

  })
  // 插入到指定位置
  for (let i = 0; i < people.length; i++) {
    queue.splice(people[i][1], 0, people[i])
  }
  return queue
};

/**
 * @param {number[][]} points
 * @return {number}
 */
var findMinArrowShots = function (points) {
  points.sort((a, b) => {
    return a[0] - b[0];
  });
  let result = 1;
  for (let i = 1; i < points.length; i++) {
    // 与前一个区间进行边界比较，如果不相邻就直接结果加一
    if (points[i][0] > points[i - 1][1]) {
      result++;
    } else {
      // 否则更新区间（取最小值）,每次比较都是与前一个区间比较
      points[i][1] = Math.min(points[i - 1][1], points[i][1]);
    }
  }
  return result
};

无重叠区间

/**
 * @param {number[][]} intervals
 * @return {number}
 */
var eraseOverlapIntervals = function (intervals) {
  /* 
    求出重叠区间
    以左边界排序
    
    如果区间的左端点大于等于前一个区间的右端点就表示没有重叠区间
    */
  intervals.sort((a, b) => {
    return a[0] - b[0];
  });
  let res = 0;
  for (let i = 1; i < intervals.length; i++) {
    if (intervals[i][0] >= intervals[i - 1][1]) {
    } else {
      res++;
      // 有重叠区间时更新当前区间的右边界，排除下一个区间与前一个区间重合，每次比较都是与前一个区间比较
      intervals[i][1] = Math.min(intervals[i - 1][1], intervals[i][1]);
    }
  }
  return res;
};

划分字母区间

/**
 * @param {string} s
 * @return {number[]}
 */
var partitionLabels = function (s) {
  /* 
    记录每个字母出现的最后的位置
    */
  let hash = {};
  for (let i = 0; i < s.length; i++) {
    hash[s[i]] = i;
  }
  let res = [];
  let left = 0;
  let right = 0;
  for (let i = 0; i < s.length; i++) {
    // 取出现过的最后的一个位置
    right = Math.max(right, hash[s[i]]);
    if (right === i) {
      // 存放结果(i从0开始，所以需要加一)
      res.push(right - left + 1);
      // left定义的是左区间
      left = i + 1;
    }
  }
  return res;
};

合并区间

/**
 * @param {number[][]} intervals
 * @return {number[][]}
 */
var merge = function (intervals) {
  let res = []
  // 根据左边界排序
  intervals.sort((a, b) => {
    return a[0] - b[0];
  });
  // 默认等于第一个区间
  let prev = intervals[0];
  for (let i = 1; i < intervals.length; i++) {
    let cur = intervals[i];
    // 每次都是与上一个区间相比较，如果有重复的，就更新值，直到没有重复的，再收入结果
    if (prev[1] >= cur[0]) { // 有重合
      prev[1] = Math.max(cur[1], prev[1]);
    } else {       // 不重合，prev推入res数组 
      res.push(prev);
      prev = cur;  // 更新 prev
    }
  }
  res.push(prev);
  return res;
};

单调自增的数字

从后往前遍历，去找到需要减1的地方，前边减一，然后后边的值统一改为9

/**
 * @param {number} n
 * @return {number}
 */
var monotoneIncreasingDigits = function (n) {
  n = String(n);
  n = n.split("").map((item) => {
    return +item;
  });
  // 初始值为最大值，避免后续赋值为9出现问题
  let flag = Infinity;
  for (let i = n.length - 1; i > 0; i--) {
    if (n[i - 1] > n[i]) {
      flag = i;
      n[i - 1]--;
    }
  }

  for (let i = flag; i < n.length; i++) {
    n[i] = 9;
  }
  n = n.join("");
  return +n
};
