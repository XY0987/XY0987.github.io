二叉树专题

发表于2023|更新于2026|算法
|字数总计:5.1k|阅读时长:24分钟

二叉树是每个节点最多有两个子树的树结构，本文整理了二叉树相关的经典算法题目。

二叉树基础

二叉树的种类：

  1. 满二叉树，节点数量为2^k-1 (k为节点深度)
  2. 完全二叉树，（其他层都是满的，底层是从左到右连续的二叉树，两个节点之间没有断层）
  3. 搜索二叉树，（左边的所有节点都小于中间节点，右边的所有节点都大于中间节点）
  4. 平衡搜索二叉树，（在搜索二叉树的基础上，左子树和右子树高度差不超过1）

二叉树遍历

前序遍历

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
//前序遍历
var preorderTraversal = function (root) {
  const res = []
  function getValue(node) {
    if (!node) {
      return
    }
    res.push(node.val)
    // 遍历左节点
    getValue(node.left)
    // 遍历右节点
    getValue(node.right)
  }
  getValue(root)
  return res
};

二叉树的层序遍历

借助队列，保存每一层的数据

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
var levelOrder = function (root) {
  // 使用队列去辅助，弹出原生
  const queue = []
  const res = []
  if (!root) {
    return res
  }
  queue.push(root)
  // 遍历对列
  while (queue.length > 0) {
    // 保存本层节点的数量(因为之后遍历的时候会添加下一层的数据)
    let len = queue.length
    const resItem = []
    // 将本层数据弹出，并保存进数组
    while (len > 0) {
      const node = queue.shift()
      resItem.push(node.val)
      // 将下一层的左右节点压入
      if (node.left) {
        queue.push(node.left)
      }
      if (node.right) {
        queue.push(node.right)
      }
      len--
    }
    res.push(resItem)
  }
  return res
};

反转二叉树

可以使用前序或者后序，中序比较麻烦

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var invertTree = function (root) {
  // 交换左右节点
  function invert(node) {
    if (!node) {
      return null;
    }
    // 交换两个节点
    const left = node.left
    node.left = node.right
    node.right = left
    invert(node.left)
    invert(node.right)
  }
  invert(root)
  return root
};

对称二叉树

只能使用后序遍历（左右中）
需要比对孩子节点的信息是否相等，返回给上一个节点

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function (root) {
  // 定义函数判断是否相等，传入根节点的左右节点
  function getRes(left, right) {
    // 左边节点为空，右边不为空
    if (!left && right) {
      return false;
    }
    // 左边不为空，右边为空
    if (left && !right) {
      return false;
    }
    // 左右节点都为空
    if (!left && !right) {
      return true;
    }
    // 左右节点都不为空，但是值不相等
    if (left && right && left.val !== right.val) {
      return false;
    }
    // 先比对外侧节点（左节点的左孩子和右节点的右孩子）
    const yn1 = getRes(left.left, right.right);
    // 比对内侧节点（左节点的右孩子，右节点的左孩子）
    const yn2 = getRes(left.right, right.left);
    return yn1 && yn2;
  }
  return getRes(root.left, root.right);
};

二叉树的最大深度

深度：二叉树任意一个节点到跟节点的距离
高度：二叉树种任意一个节点到叶子节点的距离
根节点的高度就是二叉树的最大深度

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function (root) {
  function getHeight(node) {
    if (!node) {
      return 0;
    }
    // 左节点
    const leftHeight = getHeight(node.left);
    // 右节点
    const rightHeight = getHeight(node.right);
    // 中节点(处理逻辑,1加上左右节点的最大值)
    const height = 1 + Math.max(leftHeight, rightHeight);
    return height;
  }
  return getHeight(root);
};

二叉树的最小深度

根节点到最近的叶子节点的距离，后续遍历

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var minDepth = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    const leftRes = getRes(node.left);
    const rightRes = getRes(node.right);
    // 如果左子树为空，但是右子树不为空(排除掉左边为空的错误情况)
    if (!node.left && node.right) {
      return 1 + rightRes;
    }
    if (!node.right && node.left) {
      return 1 + leftRes;
    }
    // 两边的子树都不为空，取最小值
    return 1 + Math.min(leftRes, rightRes);
  }
  return getRes(root);
};

完全二叉树节点的数量

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
// 普通二叉树的方式写(后序遍历)
var countNodes = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    const leftNum = getRes(node.left);
    const rightNum = getRes(node.right);
    // 左子树的数量加上右子树的数量加上本身节点的数量
    return leftNum + rightNum + 1;
  }
  return getRes(root);
};

判断平衡二叉树

任意节点它的左右子树的高度差不超过 1(后序遍历)

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isBalanced = function (root) {
  function getRes(node) {
    if (!node) {
      return 0;
    }
    // 后续遍历，如果不是一个平衡二叉树，就返回-1
    const leftHeight = getRes(node.left);
    if (leftHeight === -1) {
      return -1;
    }
    const rightHeight = getRes(node.right);
    if (rightHeight === -1) {
      return -1;
    }
    if (Math.abs(leftHeight - rightHeight) > 1) {
      return -1;
    }
    // 获取本层的高度(高度差不超过1)(左子树和右子树的高度的最大值+1)
    return Math.max(leftHeight, rightHeight) + 1;
  }
  return getRes(root) === -1 ? false : true;
};

二叉树的所有路径

前序遍历，中左右，前序遍历才有将父节点指向子节点
回溯

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {string[]}
 */
var binaryTreePaths = function (root) {
  // 用于存放结果
  const res = [];
  function getRes(node, path, res) {
    // 添加操作再这里的原因是遍历到叶子节点就结束了
    path.push(node.val);
    // 左右孩子都为空，遍历到叶子节点
    if (!node.left && !node.right) {
      res.push(path.join("->").toString());
      return;
    }
    if (node.left) {
      getRes(node.left, path, res);
      // 回溯过程
      path.pop();
    }
    if (node.right) {
      getRes(node.right, path, res);
      path.pop();
    }
  }
  getRes(root, [], res);
  return res;
};

左叶子之和

叶子节点（左右节点都为空）

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var sumOfLeftLeaves = function (root) {
  function getRes(node) {
    /**
        遍历到叶子节点的上一层节点，获取到值，叶子节点以及当前节点为空的都返回0
        叶子节点的上一层节点的左节点，就是要求的值
         */
    if (!node) {
      return 0;
    }
    if (!node.left && !node.right) {
      return 0;
    }
    let leftNum = getRes(node.left);
    // 叶子节点
    if (node.left && !node.left.left && !node.left.right) {
      leftNum = node.left.val;
    }
    let rightNum = getRes(node.right);
    return leftNum + rightNum;
  }
  return getRes(root);
};

找二叉树左下角的值

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
// 用层序遍历
var findBottomLeftValue = function (root) {
  // 使用队列去辅助，弹出原生
  const queue = [];
  const res = [];
  if (!root) {
    return res;
  }
  queue.push(root);
  // 遍历对列
  while (queue.length > 0) {
    // 保存本层节点的数量(因为之后遍历的时候会添加下一层的数据)
    let len = queue.length;
    const resItem = [];
    // 将本层数据弹出，并保存进数组
    while (len > 0) {
      const node = queue.shift();
      resItem.push(node.val);
      // 将下一层的左右节点压入
      if (node.left) {
        queue.push(node.left);
      }
      if (node.right) {
        queue.push(node.right);
      }
      len--;
    }
    res.push(resItem);
  }
  return res[res.length - 1][0];
};

路径总和

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} targetSum
 * @return {boolean}
 */
var hasPathSum = function (root, targetSum) {
  const res = [];
  // 遍历所有路径
  function getPath(node, path) {
    path.push(node.val);
    if (!node.left && !node.right) {
      res.push(
        path.reduce((a, b) => {
          return a + b;
        }, 0)
      );
    }
    // 去遍历左节点
    if (node.left) {
      getPath(node.left, path);
      path.pop();
    }
    if (node.right) {
      getPath(node.right, path);
      path.pop();
    }
  }
  if (!root) {
    return false;
  }
  getPath(root, []);
  return res.includes(targetSum);
};

从中序和后序遍历中构造二叉树

中序是左中右
后序是左右中
先从后序中找到中间节点(后序的最后一位就是二叉树的根节点)，然后再根据前序遍历中确定左子树和右子树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} inorder
 * @param {number[]} postorder
 * @return {TreeNode}
 */
var buildTree = function (inorder, postorder) {
  if (!inorder.length) return null;
  const rootVal = postorder.pop(); // 从后序遍历的数组中获取中间节点的值， 即数组最后一个值
  let rootIndex = inorder.indexOf(rootVal); // 获取中间节点在中序遍历中的下标
  const root = new TreeNode(rootVal); // 创建中间节点
  root.left = buildTree(
    inorder.slice(0, rootIndex),
    postorder.slice(0, rootIndex)
  ); // 创建左节点
  root.right = buildTree(
    inorder.slice(rootIndex + 1),
    postorder.slice(rootIndex)
  ); // 创建右节点
  return root;
};

最大二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} nums
 * @return {TreeNode}
 */
var constructMaximumBinaryTree = function (nums) {
  // 没有元素之间返回null
  if (nums.length === 0) {
    return null;
  }
  // 找到数组中的最大值
  const maxValue = Math.max(...nums);
  // 找到最大值的下标
  const index = nums.indexOf(maxValue);
  const root = new TreeNode(maxValue);
  // 递归处理左节点
  root.left = constructMaximumBinaryTree(nums.slice(0, index));
  // 递归处理右节点
  root.right = constructMaximumBinaryTree(nums.slice(index + 1));
  return root;
};

合并二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root1
 * @param {TreeNode} root2
 * @return {TreeNode}
 */
var mergeTrees = function (root1, root2) {
  // 遇到空节点返回另一个树的节点
  if (!root1) {
    return root2;
  }
  if (!root2) {
    return root1;
  }
  // 两个节点的值都不为空，合并(直接修改root1的值)
  root1.val += root2.val;
  root1.left = mergeTrees(root1.left, root2.left);
  root1.right = mergeTrees(root1.right, root2.right);
  return root1;
};

二叉搜索树的搜索

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} val
 * @return {TreeNode}
 */
var searchBST = function (root, val) {
  // 二叉搜索树的左子树都要比根节点小，右子树都要比根节点大
  if (root.val === val) {
    return root;
  }
  if (root.left && root.val > val) {
    return searchBST(root.left, val);
  }
  if (root.right && root.val < val) {
    return searchBST(root.right, val);
  }
  return null;
};

验证二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isValidBST = function (root) {
  /** 
    中序遍历，先遍历左节点，再遍历中节点，最后遍历右节点
    如果中序遍历出来的结果是单调递增的，就表示是二叉搜索树
    (同时需要验证是否有重复元素)
    */
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    getVal(node.left);
    res.push(node.val);
    getVal(node.right);
  }
  getVal(root);
  // 重复元素返回false
  const temArr = Array.from(new Set(res));
  if (temArr.length !== res.length) {
    return false;
  }
  const resStrOri = res.join("");
  const resStrSort = res
    .sort((a, b) => {
      return a - b;
    })
    .join("");
  return resStrOri === resStrSort;
};

二叉搜索树的最小绝对差

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var getMinimumDifference = function (root) {
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    res.push(node.val);
    getVal(node.left);
    getVal(node.right);
  }
  getVal(root);
  res.sort((a, b) => {
    return a - b;
  });
  let min = 0;
  let left = 1;
  for (let i = 0; i < res.length - 1; i++) {
    if (min === 0) {
      min = res[left] - res[i];
    } else {
      min = Math.min(min, res[left] - res[i]);
    }
    left++;
  }
  return min;
};

二叉搜索树中的众数

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var findMode = function (root) {
  const res = [];
  function getVal(node) {
    if (!node) {
      return;
    }
    getVal(node.left);
    res.push(node.val);
    getVal(node.right);
  }
  getVal(root);
  if (res.length === 1) {
    return res;
  }
  const map = new Map();
  res.forEach((item) => {
    map.set(item, (map.get(item) | 0) + 1);
  });
  const arr = Array.from(map).sort((a, b) => {
    return b[1] - a[1];
  });
  const result = [];
  arr.forEach((value) => {
    if (value[1] === arr[0][1]) {
      result.push(value[0]);
    }
  });
  return result;
};

二叉树的最近公共祖先

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {TreeNode}
 */
var lowestCommonAncestor = function (root, p, q) {
  // 遍历到空节点，直接返回
  if (!root) {
    return null
  }
  // 遇到节点等于p或者等于q往上返回
  if (root === p || root === q) {
    return root
  }
  // 获取左树是否有满足条件的值
  let left = lowestCommonAncestor(root.left, p, q)
  // 获取右树是否有满足条件的值
  let right = lowestCommonAncestor(root.right, p, q)
  if (left && right) {
    return root
  }
  // 如果左为空，右不为空，表示答案在右子树里边
  if (!left && right) {
    return right
  }
  if (!right && left) {
    return left
  }
};

二叉搜索树的最近公共祖先

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */

/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {TreeNode}
 */
var lowestCommonAncestor = function (root, p, q) {
  // 利用二叉树特性，找到一个节点的值在两个节点值之间,就是最近的公共祖先
  if (!root) {
    return null
  }
  // 向左去遍历(答案在左子树中)
  if (root.val > p.val && root.val > q.val) {
    const left = lowestCommonAncestor(root.left, p, q)
    if (left) {
      return left
    }
  }
  // 向右去遍历(答案在右子树中)
  if (root.val < p.val && root.val < q.val) {
    const right = lowestCommonAncestor(root.right, p, q)
    if (right) {
      return right
    }
  }
  // 获取到结果了
  return root
};

二叉搜索树的插入操作

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} val
 * @return {TreeNode}
 */
var insertIntoBST = function (root, val) {
  function insert(node) {
    if (!node) {
      // 找到插入位置了
      return new TreeNode(val)
    }
    if (node.val > val) {
      // 应该插入左子树
      const left = insert(node.left)
      if (left) {
        node.left = left
      }
    }
    if (node.val < val) {
      // 应该插入右子树
      const right = insert(node.right)
      if (right) {
        node.right = right
      }
    }
  }
  if (!root) {
    return new TreeNode(val)
  }
  insert(root)
  return root
};

删除二叉搜索树的节点

  1. 没找到要删除的节点
  2. 删除的是叶子节点（左右都为空，不用改变二叉树的结构）
  3. 要删除的节点右为空，左不为空（直接让父节点指向节点的左孩子）
  4. 左空，右不空
  5. 左右都不为空（直接把左子树放到右子树中最小的数（右子树最左侧的位置）中）
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} key
 * @return {TreeNode}
 */
var deleteNode = function (root, key) {
  // 没有遍历到要删除的节点,返回空节点
  if (!root) {
    return null
  }
  // 找到要删除的节点了
  if (root.val === key) {
    // 情况一,删除叶子节点(左子树为空，右子树也为空)
    if (!root.left && !root.right) {
      return null
    } else if (!root.left && root.right) {
      // 情况二,节点左为空，右不为空
      return root.right
    } else if (root.left && !root.right) {
      // 情况三,节点左不为空，右为空
      return root.left
    } else {
      // 左右节点都不为空
      // 找到右子树的最小的节点(将左子树放到右子树的最小值哪里)
      let cur = root.right
      while (cur.left) {
        cur = cur.left
      }
      cur.left = root.left
      return root.right
    }
  }
  // 
  if (key < root.val) {
    // 向左去遍历
    root.left = deleteNode(root.left, key)
  }
  if (key > root.val) {
    // 向右去遍历
    root.right = deleteNode(root.right, key)
  }
  return root
};

修剪二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} low
 * @param {number} high
 * @return {TreeNode}
 */
var trimBST = function (root, low, high) {
  if (!root) {
    return null
  }
  if (root.val < low) {
    // 当前节点小于给定的最小值，应该往右遍历是否有符合结果的节点
    let right = trimBST(root.right, low, high)
    return right
  }
  if (root.val > high) {
    // 当前节点值大于给定的最大值，遍历右节点
    let left = trimBST(root.left, low, high)
    return left
  }
  // 处理左子树
  root.left = trimBST(root.left, low, high)
  // 处理右子树
  root.right = trimBST(root.right, low, high)
  return root
};

将有序数组转换为二叉搜索树

平衡二叉搜索树，选取中间节点开始构造，这样才能保证两边的节点数量相同，便于构造平衡二叉搜索树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {number[]} nums
 * @return {TreeNode}
 */
var sortedArrayToBST = function (nums) {
  // nums表示数组,left和right表示左右区间，左闭右闭
  function getRes(nums, left, right) {
    if (left > right) {
      return null
    }
    const mid = (left + right) >> 1
    const root = new TreeNode(nums[mid])
    // 左闭右闭，所以传mid-1
    root.left = getRes(nums, left, mid - 1)
    root.right = getRes(nums, mid + 1, right)
    return root
  }
  return getRes(nums, 0, nums.length - 1)
};

把二叉搜索树转化为累加树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var convertBST = function (root) {
  // 定义一个变量，用于记录之前节点的相加的结果
  let prev = 0
  function getRes(node) {
    if (!node) {
      return
    }
    // 因为要加上一个结果的值，所以倒序遍历(右中左)
    getRes(node.right)
    node.val += prev
    // 记录上一个节点的数值
    prev = node.val
    getRes(node.left)
  }
  getRes(root)
  return root
};

打家劫舍三

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var rob = function (root) {
    /* 
    dp数组是一维数组，只有两个元素，分别表示偷该节点和不偷该节点的最大值
    dp[0] 表示不偷
    dp[1] 表示偷

    后序遍历
    */
  
  // 后序遍历函数
  const postOrder = (node) => {
    // 递归出口
    if (!node) return [0, 0];
    // 遍历左子树
    const left = postOrder(node.left);
    // 遍历右子树
    const right = postOrder(node.right);
    // 不偷当前节点，左右子节点都可以偷或不偷，取最大值
    const DoNot = Math.max(left[0], left[1]) + Math.max(right[0], right[1]);
    // 偷当前节点，左右子节点只能不偷
    const Do = node.val + left[0] + right[0];
    // [不偷，偷]
    return [DoNot, Do];
  };
  const res = postOrder(root);
  // 返回最大值
  return Math.max(...res);
};

监控二叉树

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var minCameraCover = function (root) {
  let result = 0;
  function traversal(cur) {
    // 空节点
    if (cur === null) {
      return 2;
    }
    let left = traversal(cur.left);
    let right = traversal(cur.right);
    if (left === 2 && right === 2) {
      return 0;
    }
    if (left === 0 || right === 0) {
      result++;
      return 1;
    }
    if (left === 1 || right === 1) {
      return 2;
    }
    return -1;
  }
  if (traversal(root) === 0) {
    result++;
  }
  return result;
};
