TypeScript类型体操：手写内置类型工具原理与实战

发表于2025|更新于2026|前端基础
|字数总计:4.3k|阅读时长:21分钟

前言

TypeScript 的类型系统可以让我们用类型来完成复杂的逻辑运算。类型体操就是利用 TypeScript 的类型系统来实现各种复杂的类型转换。本文将深入剖析 TypeScript 内置类型工具的实现原理，并带你手写这些工具类型，真正理解类型编程的本质。

一、类型体操基础

1.1 什么是类型体操？

类型体操是指在 TypeScript 类型层面进行的编程操作，包括：

  • 映射类型：从旧类型创建新类型
  • 条件类型：根据条件判断类型
  • 类型推断：使用 infer 关键字推断类型
  • 递归类型：类型递归调用自身
// 简单的类型体操示例
type MyPick<T, K extends keyof T> = {
  [P in K]: T[P];
};

interface User {
  name: string;
  age: number;
  email: string;
}

// 使用自定义的 Pick
type UserName = MyPick<User, 'name'>; // { name: string }

1.2 核心概念

1.2.1 keyof 操作符

keyof 用于获取类型的所有键组成的联合类型：

interface Person {
  name: string;
  age: number;
}

type PersonKeys = keyof Person; // "name" | "age"

// 实际应用：约束参数类型
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const person: Person = { name: '张三', age: 25 };
const name = getProperty(person, 'name'); // 类型安全
// const invalid = getProperty(person, 'gender'); // 错误：属性不存在

1.2.2 索引访问类型

使用 T[K] 访问类型的属性类型：

interface User {
  id: number;
  name: string;
  address: {
    city: string;
    street: string;
  };
}

type UserId = User['id']; // number
type UserAddress = User['address']; // { city: string; street: string; }
type CityType = User['address']['city']; // string

1.2.3 映射类型

映射类型用于从旧类型创建新类型：

// 基础映射
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

// 使用示例
interface Todo {
  title: string;
  description: string;
}

type ReadonlyTodo = Readonly<Todo>;
// {
//   readonly title: string;
//   readonly description: string;
// }

二、工具类型实现原理

2.1 Partial - 可选属性

Partial<T> 将类型 T 的所有属性变为可选。

内置定义：

type Partial<T> = {
  [P in keyof T]?: T[P];
};

手写实现：

type MyPartial<T> = {
  [P in keyof T]?: T[P];
};

// 测试
interface User {
  name: string;
  age: number;
  email: string;
}

type PartialUser = MyPartial<User>;
// {
//   name?: string;
//   age?: number;
//   email?: string;
// }

// 实际应用
function updateUser(user: User, updates: MyPartial<User>): User {
  return { ...user, ...updates };
}

const user: User = {
  name: '张三',
  age: 25,
  email: 'zhangsan@example.com'
};

const updatedUser = updateUser(user, { age: 26 });

进阶：递归 Partial

type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object 
    ? DeepPartial<T[P]> 
    : T[P];
};

interface Config {
  server: {
    host: string;
    port: number;
  };
  database: {
    name: string;
    credentials: {
      username: string;
      password: string;
    };
  };
}

type PartialConfig = DeepPartial<Config>;
// 所有嵌套属性都变为可选
const config: PartialConfig = {
  server: {
    port: 3000
    // host 可以不提供
  }
};

2.2 Required - 必选属性

Required<T> 将类型 T 的所有属性变为必选。

内置定义：

type Required<T> = {
  [P in keyof T]-?: T[P];
};

手写实现：

type MyRequired<T> = {
  [P in keyof T]-?: T[P]; // -? 移除可选标记
};

// 测试
interface User {
  name: string;
  age?: number;
  email?: string;
}

type RequiredUser = MyRequired<User>;
// {
//   name: string;
//   age: number;    // 变为必选
//   email: string;  // 变为必选
// }

// 实际应用
function validateUser(user: MyRequired<User>): void {
  console.log(`Name: ${user.name}, Age: ${user.age}, Email: ${user.email}`);
}

注意： -? 是 TypeScript 2.8 引入的修饰符语法，用于移除可选标记。类似的还有 +?（添加可选，默认行为）和 readonly 的 +/- 修饰。

2.3 Readonly - 只读属性

Readonly<T> 将类型 T 的所有属性变为只读。

内置定义：

type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

手写实现：

type MyReadonly<T> = {
  readonly [P in keyof T]: T[P];
};

// 测试
interface Todo {
  title: string;
  description: string;
}

type ReadonlyTodo = MyReadonly<Todo>;
const todo: ReadonlyTodo = {
  title: '学习TypeScript',
  description: '掌握类型体操'
};

// todo.title = '新标题'; // 错误：无法赋值，因为是只读属性

进阶：深度 Readonly

type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? DeepReadonly<T[P]>
    : T[P];
};

interface AppState {
  user: {
    name: string;
    preferences: {
      theme: string;
      language: string;
    };
  };
}

type ImmutableState = DeepReadonly<AppState>;
const state: ImmutableState = {
  user: {
    name: '张三',
    preferences: {
      theme: 'dark',
      language: 'zh-CN'
    }
  }
};

// state.user.name = '李四'; // 错误：只读属性
// state.user.preferences.theme = 'light'; // 错误：只读属性

2.4 Pick - 选取属性

Pick<T, K> 从类型 T 中选取部分属性 K 组成新类型。

内置定义：

type Pick<T, K extends keyof T> = {
  [P in K]: T[P];
};

手写实现：

type MyPick<T, K extends keyof T> = {
  [P in K]: T[P];
};

// 测试
interface User {
  id: number;
  name: string;
  email: string;
  password: string;
  createdAt: Date;
}

// 只选取公开信息
type PublicUser = MyPick<User, 'id' | 'name' | 'email'>;
// {
//   id: number;
//   name: string;
//   email: string;
// }

// 实际应用：API 响应类型
function getPublicUser(user: User): PublicUser {
  return {
    id: user.id,
    name: user.name,
    email: user.email
  };
}

进阶：按类型选取属性

// 选取所有某种类型的属性
type PickByType<T, U> = {
  [P in keyof T as T[P] extends U ? P : never]: T[P];
};

interface SystemConfig {
  apiUrl: string;
  timeout: number;
  retryTimes: number;
  debug: boolean;
  apiKey: string;
}

type StringConfig = PickByType<SystemConfig, string>;
// {
//   apiUrl: string;
//   apiKey: string;
// }

type NumberConfig = PickByType<SystemConfig, number>;
// {
//   timeout: number;
//   retryTimes: number;
// }

2.5 Omit - 排除属性

Omit<T, K> 从类型 T 中排除部分属性 K。

内置定义：

type Omit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

手写实现：

type MyOmit<T, K extends keyof any> = Pick<T, Exclude<keyof T, K>>;

// 或者更直观的实现
type MyOmit2<T, K extends keyof T> = {
  [P in keyof T as P extends K ? never : P]: T[P];
};

// 测试
interface User {
  id: number;
  name: string;
  email: string;
  password: string;
}

// 排除敏感信息
type SafeUser = MyOmit<User, 'password'>;
// {
//   id: number;
//   name: string;
//   email: string;
// }

// 实际应用：创建用户时不包含 id
type CreateUser = MyOmit<User, 'id'>;
// {
//   name: string;
//   email: string;
//   password: string;
// }

进阶：按条件排除属性

// 排除所有某种类型的属性
type OmitByType<T, U> = {
  [P in keyof T as T[P] extends U ? never : P]: T[P];
};

interface SystemConfig {
  apiUrl: string;
  timeout: number;
  retryTimes: number;
  debug: boolean;
  apiKey: string;
}

type NonStringConfig = OmitByType<SystemConfig, string>;
// {
//   timeout: number;
//   retryTimes: number;
//   debug: boolean;
// }

2.6 Record - 对象类型

Record<K, T> 创建键类型为 K，值类型为 T 的对象类型。

内置定义：

type Record<K extends keyof any, T> = {
  [P in K]: T;
};

手写实现：

type MyRecord<K extends keyof any, T> = {
  [P in K]: T;
};

// 测试
type UserRole = 'admin' | 'user' | 'guest';
type RolePermissions = MyRecord<UserRole, string[]>;

const permissions: RolePermissions = {
  admin: ['read', 'write', 'delete'],
  user: ['read', 'write'],
  guest: ['read']
};

// 实际应用：路由配置
type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';
type RouteHandlers = MyRecord<HttpMethod, Function>;

const handlers: RouteHandlers = {
  GET: () => { /* ... */ },
  POST: () => { /* ... */ },
  PUT: () => { /* ... */ },
  DELETE: () => { /* ... */ }
};

2.7 Exclude - 排除联合类型

Exclude<T, U> 从类型 T 中排除可以赋值给 U 的类型。

内置定义：

type Exclude<T, U> = T extends U ? never : T;

手写实现：

type MyExclude<T, U> = T extends U ? never : T;

// 测试
type AllTypes = 'a' | 'b' | 'c' | 'd';
type Excluded = MyExclude<AllTypes, 'a' | 'c'>; // 'b' | 'd'

// 实际应用：过滤联合类型
type EventNames = 'click' | 'scroll' | 'mousemove' | 'keypress';
type MouseEvents = MyExclude<EventNames, 'keypress'>;
// 'click' | 'scroll' | 'mousemove'

原理解析：

当 T 是联合类型时，条件类型会自动分发：

type MyExclude<T, U> = T extends U ? never : T;

// 对于 Exclude<'a' | 'b', 'a'>
// 等价于：
// ('a' extends 'a' ? never : 'a') | ('b' extends 'a' ? never : 'b')
// = never | 'b'
// = 'b'

2.8 Extract - 提取联合类型

Extract<T, U> 从类型 T 中提取可以赋值给 U 的类型。

内置定义：

type Extract<T, U> = T extends U ? T : never;

手写实现：

type MyExtract<T, U> = T extends U ? T : never;

// 测试
type AllTypes = 'a' | 'b' | 'c' | 'd';
type Extracted = MyExtract<AllTypes, 'a' | 'c'>; // 'a' | 'c'

// 实际应用：提取函数类型
type MixedTypes = string | number | (() => void) | (() => string);
type FunctionTypes = MyExtract<MixedTypes, Function>;
// (() => void) | (() => string)

2.9 NonNullable - 排除 null 和 undefined

NonNullable<T> 从类型 T 中排除 null 和 undefined。

内置定义：

type NonNullable<T> = T extends null | undefined ? never : T;

手写实现：

type MyNonNullable<T> = T extends null | undefined ? never : T;

// 测试
type MaybeString = string | null | undefined;
type DefiniteString = MyNonNullable<MaybeString>; // string

// 实际应用
function processValue<T>(value: T): MyNonNullable<T> {
  if (value === null || value === undefined) {
    throw new Error('Value cannot be null or undefined');
  }
  return value;
}

2.10 ReturnType - 获取函数返回类型

ReturnType<T> 获取函数类型 T 的返回类型。

内置定义：

type ReturnType<T extends (...args: any) => any> = T extends (...args: any) => infer R ? R : any;

手写实现：

type MyReturnType<T extends (...args: any) => any> = 
  T extends (...args: any) => infer R ? R : any;

// 测试
function getUser() {
  return {
    id: 1,
    name: '张三',
    email: 'zhangsan@example.com'
  };
}

type User = MyReturnType<typeof getUser>;
// {
//   id: number;
//   name: string;
//   email: string;
// }

// 实际应用：推断 API 响应类型
async function fetchUser(id: number) {
  const response = await fetch(`/api/users/${id}`);
  return response.json() as Promise<{
    id: number;
    name: string;
    email: string;
  }>;
}

type ApiResponse = MyReturnType<typeof fetchUser>;
// Promise<{ id: number; name: string; email: string; }>

2.11 Parameters - 获取函数参数类型

Parameters<T> 获取函数类型 T 的参数类型组成的元组。

内置定义：

type Parameters<T extends (...args: any) => any> = T extends (...args: infer P) => any ? P : never;

手写实现：

type MyParameters<T extends (...args: any) => any> = 
  T extends (...args: infer P) => any ? P : never;

// 测试
function createUser(name: string, age: number, email: string) {
  return { name, age, email };
}

type UserParams = MyParameters<typeof createUser>;
// [string, number, string]

// 实际应用：转发函数参数
function wrapper<T extends (...args: any) => any>(
  fn: T,
  ...args: MyParameters<T>
): ReturnType<T> {
  console.log('Function called with:', args);
  return fn(...args);
}

2.12 InstanceType - 获取构造函数实例类型

InstanceType<T> 获取构造函数类型 T 的实例类型。

内置定义：

type InstanceType<T extends new (...args: any) => any> = 
  T extends new (...args: any) => infer R ? R : any;

手写实现：

type MyInstanceType<T extends new (...args: any) => any> = 
  T extends new (...args: any) => infer R ? R : any;

// 测试
class User {
  constructor(
    public name: string,
    public age: number
  ) {}
  
  greet() {
    return `Hello, I'm ${this.name}`;
  }
}

type UserInstance = MyInstanceType<typeof User>;
// User

// 实际应用：工厂模式
function createInstance<T extends new (...args: any) => any>(
  ctor: T,
  ...args: ConstructorParameters<T>
): MyInstanceType<T> {
  return new ctor(...args);
}

const user = createInstance(User, '张三', 25);

三、高级类型体操实战

3.1 Awaited - 深层解包 Promise

type MyAwaited<T> = T extends Promise<infer U>
  ? U extends Promise<any>
    ? MyAwaited<U>
    : U
  : T;

// 测试
type Result1 = MyAwaited<Promise<string>>; // string
type Result2 = MyAwaited<Promise<Promise<number>>>; // number
type Result3 = MyAwaited<Promise<Promise<Promise<boolean>>>>; // boolean

// 实际应用
async function getData() {
  return Promise.resolve(Promise.resolve(42));
}

type DataType = MyAwaited<ReturnType<typeof getData>>; // number

3.2 UnionToIntersection - 联合类型转交叉类型

type UnionToIntersection<U> = 
  (U extends any ? (k: U) => void : never) extends ((k: infer I) => void) ? I : never;

// 测试
type Union = { name: string } | { age: number };
type Intersection = UnionToIntersection<Union>;
// { name: string } & { age: number }

// 实际应用：合并配置
type BaseConfig = { apiUrl: string };
type AuthConfig = { apiKey: string };
type FullConfig = UnionToIntersection<BaseConfig | AuthConfig>;
// { apiUrl: string } & { apiKey: string }

3.3 RequiredKeys - 获取必选键

type RequiredKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? never : K;
}[keyof T];

// 测试
interface User {
  name: string;
  age?: number;
  email: string;
  phone?: string;
}

type UserRequiredKeys = RequiredKeys<User>; // "name" | "email"

原理解析：

  • {} 表示空对象，是所有对象类型的子类型
  • {} extends Pick<T, K> 判断是否可以赋值，从而判断属性是否可选
  • 如果可以赋值，说明属性是可选的（返回 never）
  • 如果不可以赋值，说明属性是必选的（返回键名）

3.4 GetRequired - 获取必选属性组成的类型

type GetRequired<T> = {
  [K in keyof T as {} extends Pick<T, K> ? never : K]: T[K];
};

// 测试
interface User {
  name: string;
  age?: number;
  email: string;
  phone?: string;
}

type RequiredUser = GetRequired<User>;
// {
//   name: string;
//   email: string;
// }

3.5 GetOptional - 获取可选属性组成的类型

type GetOptional<T> = {
  [K in keyof T as {} extends Pick<T, K> ? K : never]: T[K];
};

// 测试
interface User {
  name: string;
  age?: number;
  email: string;
  phone?: string;
}

type OptionalUser = GetOptional<User>;
// {
//   age?: number;
//   phone?: string;
// }

3.6 Mutable - 移除只读修饰

type MyMutable<T> = {
  -readonly [P in keyof T]: T[P];
};

// 测试
interface Todo {
  readonly title: string;
  readonly description: string;
}

type EditableTodo = MyMutable<Todo>;
// {
//   title: string;
//   description: string;
// }

const todo: EditableTodo = {
  title: '学习',
  description: 'TypeScript'
};

todo.title = '更新标题'; // 可以修改

3.7 Chainable - 链式调用类型

type Chainable<T = {}> = {
  option<K extends string, V>(key: K extends keyof T ? never : K, value: V): Chainable<T & { [P in K]: V }>;
  get(): T;
};

// 测试
declare const config: Chainable;

const result = config
  .option('apiUrl', 'https://api.example.com')
  .option('timeout', 5000)
  .option('retry', 3)
  .get();

// result 的类型：
// {
//   apiUrl: string;
//   timeout: number;
//   retry: number;
// }

3.8 Get - 深层属性访问

type Get<T, K extends string> = 
  K extends `${infer First}.${infer Rest}`
    ? First extends keyof T
      ? Get<T[First], Rest>
      : never
    : K extends keyof T
      ? T[K]
      : never;

// 测试
interface Config {
  server: {
    host: string;
    port: number;
  };
  database: {
    name: string;
    credentials: {
      username: string;
      password: string;
    };
  };
}

type HostType = Get<Config, 'server.host'>; // string
type PortType = Get<Config, 'server.port'>; // number
type UsernameType = Get<Config, 'database.credentials.username'>; // string

3.9 Diff - 差集类型

type Diff<T, U> = Omit<T & U, keyof (T | U)>;

// 或者更简单的实现
type Diff2<T, U> = {
  [K in keyof (T | U) as K extends keyof (T & U) ? never : K]: (T & U)[K];
};

// 测试
type Foo = {
  name: string;
  age: number;
};

type Bar = {
  name: string;
  email: string;
};

type FooDiffBar = Diff<Foo, Bar>;
// { age: number }

type BarDiffFoo = Diff<Bar, Foo>;
// { email: string }

3.10 Intersection - 交集类型

type Intersection<T, U> = Pick<T, keyof T & keyof U>;

// 测试
type Foo = {
  name: string;
  age: number;
};

type Bar = {
  name: string;
  email: string;
};

type FooIntersectBar = Intersection<Foo, Bar>;
// { name: string }

四、实战案例：构建完整的类型工具库

4.1 深度工具类型集合

// 深度 Partial
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

// 深度 Required
type DeepRequired<T> = {
  [P in keyof T]-?: T[P] extends object ? DeepRequired<T[P]> : T[P];
};

// 深度 Readonly
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object ? DeepReadonly<T[P]> : T[P];
};

// 测试
interface ComplexConfig {
  server: {
    host: string;
    port?: number;
    ssl: {
      enabled?: boolean;
      cert?: string;
    };
  };
}

type PartialConfig = DeepPartial<ComplexConfig>;
// 所有嵌套属性都变为可选

type RequiredConfig = DeepRequired<ComplexConfig>;
// 所有嵌套属性都变为必选

type ReadonlyConfig = DeepReadonly<ComplexConfig>;
// 所有嵌套属性都变为只读

4.2 类型守卫工具

// 判断类型是否为 any
type IsAny<T> = 0 extends (1 & T) ? true : false;

// 判断类型是否为 never
type IsNever<T> = [T] extends [never] ? true : false;

// 判断类型是否为 unknown
type IsUnknown<T> = IsNever<T> extends false
  ? T extends unknown
    ? unknown extends T
      ? IsAny<T> extends false
        ? true
        : false
      : false
    : false
  : false;

// 判断两个类型是否相等
type Equals<X, Y> = (<T>() => T extends X ? 1 : 2) extends (<T>() => T extends Y ? 1 : 2)
  ? true
  : false;

// 测试
type Test1 = IsAny<any>; // true
type Test2 = IsAny<string>; // false
type Test3 = IsNever<never>; // true
type Test4 = IsNever<string>; // false
type Test5 = IsUnknown<unknown>; // true
type Test6 = IsUnknown<string>; // false
type Test7 = Equals<{ a: 1 }, { a: 1 }>; // true
type Test8 = Equals<{ a: 1 }, { b: 1 }>; // false

4.3 数组工具类型

// 获取数组元素类型
type ElementType<T extends readonly unknown[]> = T extends readonly (infer E)[] ? E : never;

// 获取元组第一个元素
type Head<T extends any[]> = T extends [infer H, ...any[]] ? H : never;

// 获取元组最后一个元素
type Last<T extends any[]> = T extends [...any[], infer L] ? L : never;

// 获取元组除第一个外的元素
type Tail<T extends any[]> = T extends [any, ...infer Rest] ? Rest : never;

// 测试
type ArrayElement = ElementType<string[]>; // string
type First = Head<[1, 2, 3]>; // 1
type LastElement = Last<[1, 2, 3]>; // 3
type Rest = Tail<[1, 2, 3]>; // [2, 3]

五、总结

5.1 核心要点回顾

  1. 映射类型：使用 [P in keyof T] 遍历类型的所有键
  2. 条件类型：使用 T extends U ? X : Y 进行类型判断
  3. 类型推断：使用 infer 关键字推断类型
  4. 修饰符操作：使用 +/- 添加或移除 readonly 和 ? 修饰符
  5. 联合类型分发：条件类型会自动对联合类型进行分发

5.2 类型体操的价值

  • 类型安全：在编译时捕获更多错误
  • 代码提示：提供更好的 IDE 自动补全
  • 文档化：类型即文档，提高代码可读性
  • 重构友好：类型系统辅助安全重构

5.3 学习建议

  1. 从简单开始：先掌握基础的 Partial、Required、Pick、Omit
  2. 理解原理：深入理解映射类型、条件类型、infer 的原理
  3. 多做练习：通过 TypeScript Challenge 等平台进行练习
  4. 阅读源码：阅读流行库的类型定义，学习最佳实践

类型体操是 TypeScript 高级特性的核心，掌握它能够让你写出更加健壮、类型安全的代码。希望本文能够帮助你深入理解 TypeScript 的类型系统，在实际项目中灵活运用这些技巧。
