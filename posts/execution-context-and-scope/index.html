JavaScript执行上下文与作用域深度解析

发表于2023|更新于2026|前端基础
|字数总计:3.2k|阅读时长:13分钟

执行上下文和作用域是 JavaScript 最核心的概念之一，理解它们对于掌握闭包、this 指向、变量提升等特性至关重要。本文将深入剖析执行上下文与作用域的原理。

一、执行上下文

1.1 什么是执行上下文？

执行上下文（Execution Context）是 JavaScript 代码执行时的环境抽象。可以理解为代码执行时的”舞台”，它决定了代码可以访问哪些变量、函数和对象。

// 全局执行上下文
var globalVar = 'global';

function foo() {
  // 函数执行上下文
  var localVar = 'local';
  console.log(globalVar);  // 可以访问
  console.log(localVar);   // 可以访问
}

foo();

1.2 执行上下文的类型

类型 说明 创建时机
全局执行上下文 代码执行的基础环境 代码执行前创建，只有 1 个
函数执行上下文 函数调用时创建的环境 每次函数调用时创建
Eval 执行上下文 eval() 函数执行时的环境 eval() 执行时创建

1.3 执行上下文的生命周期

创建阶段 → 执行阶段 → 回收阶段

// 示例：执行上下文的创建和执行
var a = 10;
function foo(b) {
  var c = 20;
  console.log(a, b, c);
}
foo(30);

/*
执行过程：
1. 创建全局执行上下文
   - 变量对象：{ a: undefined, foo: function }
   - 作用域链：[Global]
   - this: window

2. 执行全局代码
   - a = 10
   - 调用 foo(30)

3. 创建 foo 函数执行上下文
   - 变量对象：{ arguments: { 0: 30, length: 1 }, b: 30, c: undefined }
   - 作用域链：[foo, Global]
   - this: window

4. 执行 foo 函数代码
   - c = 20
   - console.log(10, 30, 20)
*/

二、执行上下文栈

2.1 什么是执行上下文栈？

执行上下文栈（Execution Context Stack，也叫调用栈）是一种后进先出（LIFO）的数据结构，用于管理执行上下文的创建和销毁。

// 调用栈示例
function first() {
  console.log('first start');
  second();
  console.log('first end');
}

function second() {
  console.log('second start');
  third();
  console.log('second end');
}

function third() {
  console.log('third');
}

first();

/*
执行过程：
1. 全局执行上下文入栈
2. first() 调用，first 执行上下文入栈
   → 输出 'first start'
3. second() 调用，second 执行上下文入栈
   → 输出 'second start'
4. third() 调用，third 执行上下文入栈
   → 输出 'third'
5. third 执行完毕，third 执行上下文出栈
6. 回到 second，输出 'second end'
7. second 执行完毕，second 执行上下文出栈
8. 回到 first，输出 'first end'
9. first 执行完毕，first 执行上下文出栈

输出结果：
first start
second start
third
second end
first end
*/

2.2 栈溢出错误

当调用栈超出了最大容量时，会抛出栈溢出错误：

// 无限递归导致栈溢出
function recursion() {
  recursion();
}

recursion();  // RangeError: Maximum call stack size exceeded

三、执行上下文的组成部分

3.1 变量对象（Variable Object）

变量对象是执行上下文的数据作用域，存储了上下文中定义的变量和函数声明。

// 变量提升示例
console.log(a);  // undefined（变量提升，但未赋值）
console.log(b);  // undefined
console.log(c);  // 报错：c is not defined

var a = 1;
var b = 2;

// 创建阶段的变量对象：
// window = {
//   arguments: { length: 0 },
//   a: undefined,
//   b: undefined
// }

// 执行阶段：
// window = {
//   arguments: { length: 0 },
//   a: 1,
//   b: 2
// }

3.2 活动对象（Activation Object）

在函数执行上下文中，变量对象被称为活动对象，它包含了：

function foo(a, b) {
  var c = 10;
  function bar() {}
  var d = function() {};
}

foo(1, 2);

3.3 作用域链（Scope Chain）

作用域链是由当前执行上下文的变量对象和所有父级执行上下文的变量对象组成的链表。

var a = 'global';

function outer() {
  var b = 'outer';
  
  function inner() {
    var c = 'inner';
    console.log(a, b, c);  // 'global' 'outer' 'inner'
  }
  
  inner();
}

outer();

// inner 的作用域链：
// [inner, outer, Global]

// 查找变量的过程：
// 1. 在 inner 中查找 a → 未找到
// 2. 在 outer 中查找 a → 未找到
// 3. 在 Global 中查找 a → 找到：'global'

3.4 this 绑定

执行上下文还会绑定 this 的值：

// this 绑定规则
var name = 'window';

const obj = {
  name: 'obj',
  foo: function() {
    console.log(this.name);
  },
  bar: () => {
    console.log(this.name);
  }
};

obj.foo();  // 'obj'（隐式绑定）
obj.bar();  // 'window'（箭头函数继承外层 this）

const foo = obj.foo;
foo();  // 'window'（默认绑定）

四、变量提升与函数提升

4.1 变量提升

// 变量提升
console.log(a);  // undefined
var a = 1;
console.log(a);  // 1

// 等价于：
var a;  // 声明提升
console.log(a);  // undefined
a = 1;  // 赋值保留在原位
console.log(a);  // 1

4.2 函数提升

// 函数声明提升
foo();  // 'foo'（可以调用）

function foo() {
  console.log('foo');
}

// 函数表达式不会提升
bar();  // TypeError: bar is not a function

var bar = function() {
  console.log('bar');
};

// 实际执行：
var bar;  // 变量提升
bar();    // undefined() → 报错
bar = function() {};

4.3 提升优先级

函数声明提升优先级高于变量声明：

console.log(foo);  // [Function: foo]

function foo() {}
var foo = 1;

console.log(foo);  // 1

// 等价于：
function foo() {}  // 函数声明提升（优先）
var foo;           // 变量声明提升（被忽略）
console.log(foo);  // [Function: foo]
foo = 1;           // 赋值
console.log(foo);  // 1

五、作用域

5.1 什么是作用域？

作用域（Scope）是变量的可访问范围，它决定了代码中变量和其他资源的可见性。

// 全局作用域
var globalVar = 'global';

function outer() {
  // 函数作用域
  var outerVar = 'outer';
  
  function inner() {
    // 函数作用域
    var innerVar = 'inner';
    console.log(globalVar, outerVar, innerVar);
  }
  
  inner();
  // console.log(innerVar);  // 报错：innerVar is not defined
}

outer();
// console.log(outerVar);  // 报错

5.2 作用域类型

类型 说明 ES 版本
全局作用域 在任何地方都能访问 ES5-
函数作用域 只在函数内部可访问 ES5-
块级作用域 只在代码块内可访问 ES6+ 
// ES6 块级作用域
{
  let a = 1;
  const b = 2;
  var c = 3;
}

// console.log(a);  // 报错
// console.log(b);  // 报错
console.log(c);     // 3（var 没有块级作用域）

// 函数作用域
function foo() {
  var x = 1;
  if (true) {
    var x = 2;  // 同一个 x
    console.log(x);  // 2
  }
  console.log(x);  // 2
}

// 块级作用域
function bar() {
  let x = 1;
  if (true) {
    let x = 2;  // 不同的 x
    console.log(x);  // 2
  }
  console.log(x);  // 1
}

5.3 词法作用域

JavaScript 采用词法作用域（也叫静态作用域），函数的作用域在函数定义时就确定了，而不是调用时。

// 词法作用域示例
var value = 1;

function foo() {
  console.log(value);
}

function bar() {
  var value = 2;
  foo();  // 输出 1，而不是 2
}

bar();

/*
解释：
foo 函数定义在全局作用域中，
它的作用域链在定义时就确定了：[foo, Global]
查找 value 时，会沿着作用域链查找，找到全局的 value = 1
而不是调用处的 value = 2
*/

5.4 作用域链的形成

var a = 1;

function outer() {
  var b = 2;
  
  function middle() {
    var c = 3;
    
    function inner() {
      var d = 4;
      console.log(a, b, c, d);
    }
    
    inner();
  }
  
  middle();
}

outer();

/*
作用域链：

inner 执行上下文：
[[Scope]] = [
  inner,
  middle,
  outer,
  Global
]

middle 执行上下文：
[[Scope]] = [
  middle,
  outer,
  Global
]

outer 执行上下文：
[[Scope]] = [
  outer,
  Global
]

全局执行上下文：
[[Scope]] = [
  Global
]
*/

六、执行上下文的实现原理

6.1 ES5- 规范（传统模型）

ExecutionContext = {
  VO: 变量对象,
  Scope: 作用域链,
  this: this 值
}

6.2 ES5+ 规范（现代模型）

ES5 之后，执行上下文的结构发生了变化：

ExecutionContext = {
  LexicalEnvironment: 词法环境,
  VariableEnvironment: 变量环境,
  this: this 值
}

词法环境（Lexical Environment）

词法环境包含环境记录和对外部词法环境的引用：

// 词法环境结构
LexicalEnvironment = {
  EnvironmentRecord: {
    // 变量、函数声明、let/const 声明
  },
  outer: <reference to outer LexicalEnvironment>
}

// 示例
let a = 1;
const b = 2;

function foo() {
  let c = 3;
  console.log(a, b, c);
}

foo();

/*
全局词法环境：
{
  EnvironmentRecord: {
    a: 1,
    b: 2,
    foo: function
  },
  outer: null
}

foo 函数词法环境：
{
  EnvironmentRecord: {
    c: 3
  },
  outer: <Global LexicalEnvironment>
}
*/

变量环境（Variable Environment）

变量环境专门用于存储 var 声明的变量：

let a = 1;
var b = 2;

// 词法环境：{ a: 1 }
// 变量环境：{ b: 2 }

6.3 暂时性死区（Temporal Dead Zone）

// TDZ 示例
{
  // TDZ 开始
  console.log(a);  // 报错：Cannot access 'a' before initialization
  let a = 1;  // TDZ 结束
}

// var 不会产生 TDZ
{
  console.log(b);  // undefined
  var b = 2;
}

// 函数参数也有 TDZ
function foo(a = b, b = 2) {
  console.log(a, b);
}
foo();  // 报错：Cannot access 'b' before initialization

function bar(a = 2, b = a) {
  console.log(a, b);  // 2, 2
}
bar();

七、执行上下文与作用域的关系

7.1 区别与联系

概念 说明 动态/静态
执行上下文 代码执行时的环境抽象 动态（运行时创建）
作用域 变量的可访问范围 静态（定义时确定）
作用域链 执行上下文的一部分 静态 
// 执行上下文是动态的
function foo() {
  var a = 1;
  console.log(a);
}

foo();  // 创建 foo 执行上下文
foo();  // 创建新的 foo 执行上下文

// 作用域是静态的
function outer() {
  var a = 1;
  function inner() {
    console.log(a);  // a 的作用域在定义时确定
  }
  return inner;
}

const innerFn = outer();
innerFn();  // 1（即使在外部调用，仍然能访问 a）

7.2 作用域链的构建过程

// 函数定义时：保存作用域链
function foo() {
  var a = 1;
}

// foo.[[Scope]] = [Global.VO]

// 函数调用时：
// 1. 创建执行上下文
// 2. 复制 [[Scope]] 到作用域链
// 3. 创建活动对象，压入作用域链前端
// ExecutionContext = {
//   AO: { a: undefined },
//   Scope: [AO, Global.VO]  // AO + [[Scope]]
// }

八、实战应用

8.1 理解闭包

// 闭包的本质：函数 + 函数定义时的作用域
function createCounter() {
  let count = 0;
  
  return {
    increment: function() {
      count++;
      console.log(count);
    },
    decrement: function() {
      count--;
      console.log(count);
    }
  };
}

const counter = createCounter();
counter.increment();  // 1
counter.increment();  // 2
counter.decrement();  // 1

// 即使 createCounter 执行完毕
// count 仍然存在于 increment/decrement 的作用域链中

8.2 模拟块级作用域

// ES5 模拟块级作用域
for (var i = 0; i < 5; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j);
    }, 100);
  })(i);
}
// 输出：0, 1, 2, 3, 4

// ES6 块级作用域
for (let i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 100);
}
// 输出：0, 1, 2, 3, 4

8.3 变量污染问题

// 问题：var 导致的变量污染
for (var i = 0; i < 3; i++) {
  setTimeout(function() {
    console.log(i);
  }, 100);
}
// 输出：3, 3, 3

// 解决方案一：IIFE
for (var i = 0; i < 3; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j);
    }, 100);
  })(i);
}
// 输出：0, 1, 2

// 解决方案二：let
for (let i = 0; i < 3; i++) {
  setTimeout(function() {
    console.log(i);
  }, 100);
}
// 输出：0, 1, 2

九、面试题解析

9.1 题目一：变量提升

function foo() {
  console.log(a);
  var a = 1;
  console.log(a);
  
  function a() {}
  console.log(a);
}

foo();

/*
执行过程：
1. 创建阶段 AO：
   {
     a: function a() {}  // 函数声明提升优先
   }
   
2. 执行阶段：
   console.log(a);  // function a() {}
   a = 1;           // 变量赋值
   console.log(a);  // 1
   function a() {}  // 函数声明（已被提升，忽略）
   console.log(a);  // 1
   
输出：
function a() {}
1
1
*/

9.2 题目二：作用域链

var a = 1;

function foo() {
  console.log(a);
  var a = 2;
  console.log(a);
}

foo();
console.log(a);

/*
输出：
undefined
2
1

解析：
1. foo 函数内部有 var a = 2，变量提升
2. 第一个 console.log(a) 访问的是提升后的 undefined
3. 赋值后 a = 2
4. 全局的 a 不受影响
*/

9.3 题目三：词法作用域

var value = 1;

function foo() {
  console.log(value);
}

function bar() {
  var value = 2;
  foo();
}

bar();

/*
输出：1

解析：
词法作用域：函数定义时确定作用域
foo 定义在全局，作用域链为 [foo.AO, Global.VO]
查找 value 时，沿着作用域链找到 Global.VO 中的 value = 1
而不是 bar 中的 value = 2
*/

十、总结

10.1 核心概念

概念 说明
执行上下文 代码执行的环境抽象
执行上下文栈 管理执行上下文的 LIFO 栈
变量对象 存储变量和函数声明
作用域链 变量查找的路径
词法作用域 函数定义时确定的作用域

10.2 执行过程

1. 创建执行上下文
   - 创建变量对象/词法环境
   - 建立作用域链
   - 确定 this 值

2. 执行代码
   - 变量赋值
   - 函数执行
   - 访问变量沿作用域链查找

3. 回收执行上下文
   - 函数执行完毕，上下文出栈
   - 闭包情况下，变量继续存在于作用域链中

10.3 最佳实践

  1. 优先使用 let/const：避免变量提升带来的困惑
  2. 理解词法作用域：函数定义时确定作用域
  3. 注意 TDZ：let/const 声明前不可访问
  4. 合理使用闭包：注意内存泄漏
  5. 避免全局变量污染：使用模块化或块级作用域
