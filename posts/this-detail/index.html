JavaScript this 全面解析

发表于2023|更新于2026|前端基础
|字数总计:2.3k|阅读时长:10分钟

this 是 JavaScript 中最核心的概念之一，无论是日常开发还是面试，都占据着重要地位。本文将全面深入地解析 this 的指向问题。

一、this 是什么？

this 是函数执行时的上下文对象，它的值取决于函数的调用方式，而非定义位置。简单来说：

this 永远指向最后调用它的那个对象

理解 this 的关键在于分析函数的调用位置和调用方式，JavaScript 中有四种绑定规则决定了 this 的指向。

二、this 绑定的四种规则

2.1 默认绑定

当函数独立调用（不作为对象方法、不使用 call/apply/bind、不是构造函数）时，this 指向全局对象。

// 全局作用域中的 this
console.log(this) // window

// 函数独立调用
function foo() {
  console.log(this)
}
foo() // window

// 匿名函数调用
;(function () {
  console.log(this) // window
})()

// 嵌套函数中的 this
function outer() {
  function inner() {
    console.log(this) // window（独立调用）
  }
  inner()
}
outer()

2.2 隐式绑定

当函数作为对象的方法调用时，this 指向该对象。

function foo() {
  console.log(this.name)
}

const obj = {
  name: 'obj',
  foo: foo
}

obj.foo() // 'obj'（obj 调用了 foo）

隐式丢失

隐式绑定的函数在某些情况下会丢失绑定对象，回退到默认绑定：

// 情况一：赋值后调用
function foo() {
  console.log(this)
}

const obj = {
  foo: foo
}

const bar = obj.foo // 只是赋值，没有调用
bar() // window（独立调用，隐式绑定丢失）

// 情况二：回调函数
function doCallback(fn) {
  fn() // 这里是独立调用
}

const obj = {
  name: 'obj',
  sayHi: function () {
    console.log(this.name)
  }
}

doCallback(obj.sayHi) // undefined（隐式绑定丢失）

// 情况三：间接引用
const obj1 = {
  name: 'obj1',
  foo: function () {
    console.log(this.name)
  }
}

const obj2 = {
  name: 'obj2'
}

obj2.foo = obj1.foo
obj2.foo() // 'obj2'（obj2 调用）
;(obj2.foo = obj1.foo)() // undefined（赋值表达式的返回值是函数本身，独立调用）

2.3 显式绑定

使用 call、apply、bind 方法可以显式指定函数的 this。

const obj1 = {
  name: 'obj1',
  foo: function () {
    console.log(this.name)
  }
}

const obj2 = {
  name: 'obj2'
}

obj1.foo.call(obj2) // 'obj2'
obj1.foo.apply(obj2) // 'obj2'
obj1.foo.bind(obj2)() // 'obj2'

call、apply、bind 的区别

方法 参数 返回值 特点
call (thisArg, arg1, arg2, ...) 函数执行结果 立即执行
apply (thisArg, [arg1, arg2, ...]) 函数执行结果 立即执行，参数为数组
bind (thisArg, arg1, arg2, ...) 新函数 返回绑定后的函数 
const person = {
  name: '张三'
}

function greet(greeting, punctuation) {
  console.log(greeting + ', ' + this.name + punctuation)
}

// call：逐个传参
greet.call(person, '你好', '！') // 你好, 张三！

// apply：数组传参
greet.apply(person, ['你好', '！']) // 你好, 张三！

// bind：返回新函数
const greetPerson = greet.bind(person, '你好')
greetPerson('！') // 你好, 张三！

2.4 new 绑定

使用 new 关键字调用构造函数时，this 指向新创建的实例对象。

function Person(name, age) {
  this.name = name
  this.age = age
}

const person1 = new Person('张三', 18)
console.log(person1) // Person { name: '张三', age: 18 }

// new 做了什么？
// 1. 创建一个新对象
// 2. 将新对象的 __proto__ 指向构造函数的 prototype
// 3. 将 this 绑定到新对象，执行构造函数
// 4. 返回新对象（如果构造函数返回对象，则返回该对象）

三、this 绑定优先级

四种绑定规则的优先级从高到低：

new 绑定 > 显式绑定 > 隐式绑定 > 默认绑定

// 显式绑定 > 隐式绑定
const obj = {
  name: 'obj',
  foo: function () {
    console.log(this.name)
  }
}

const obj2 = { name: 'obj2' }
obj.foo.call(obj2) // 'obj2'（显式绑定优先）

// new 绑定 > 显式绑定
function foo(name) {
  this.name = name
}

const boundFoo = foo.bind({ name: 'bound' })
const obj = new boundFoo('new') // new 绑定优先
console.log(obj.name) // 'new'

四、箭头函数的 this

4.1 箭头函数的特点

箭头函数没有自己的 this，它会捕获定义时所在上下文的 this 值（词法作用域）。

const obj = {
  name: 'obj',
  // 普通函数
  sayHi: function () {
    console.log(this.name) // 'obj'
  },
  // 箭头函数
  sayHello: () => {
    console.log(this.name) // undefined（继承外层作用域的 this，即 window）
  }
}

obj.sayHi() // 'obj'
obj.sayHello() // undefined

4.2 箭头函数的应用场景

箭头函数非常适合用于回调函数，可以避免 this 丢失的问题：

// 问题：普通函数的 this 丢失
const obj = {
  name: 'obj',
  sayHi: function () {
    setTimeout(function () {
      console.log(this.name) // undefined（回调函数独立调用）
    }, 1000)
  }
}

// 解决方案一：保存 this
const obj1 = {
  name: 'obj1',
  sayHi: function () {
    const self = this
    setTimeout(function () {
      console.log(self.name) // 'obj1'
    }, 1000)
  }
}

// 解决方案二：箭头函数（推荐）
const obj2 = {
  name: 'obj2',
  sayHi: function () {
    setTimeout(() => {
      console.log(this.name) // 'obj2'（继承 sayHi 的 this）
    }, 1000)
  }
}

4.3 箭头函数的 this 不可改变

箭头函数的 this 在定义时就已确定，无法通过 call/apply/bind 改变：

const obj = {
  name: 'obj'
}

const arrowFunc = () => {
  console.log(this)
}

arrowFunc() // window
arrowFunc.call(obj) // window（无法改变）

五、严格模式下的 this

严格模式下，全局函数中的 this 不再指向 window，而是 undefined：

// 非严格模式
function foo() {
  console.log(this) // window
}
foo()

// 严格模式
function bar() {
  'use strict'
  console.log(this) // undefined
}
bar()

// 但全局作用域的 this 仍然是 window
;(function () {
  'use strict'
  console.log(this) // window
})()

六、手写实现

6.1 手写 call

Function.prototype.myCall = function (context, ...args) {
  // 1. 判断调用者是否为函数
  if (typeof this !== 'function') {
    throw new TypeError('Error')
  }

  // 2. 处理 context
  context = context || window

  // 3. 创建唯一的 key，避免覆盖原属性
  const fnKey = Symbol()

  // 4. 将函数设为对象的方法
  context[fnKey] = this

  // 5. 执行函数
  const result = context[fnKey](...args)

  // 6. 删除添加的属性
  delete context[fnKey]

  // 7. 返回结果
  return result
}

// 测试
function greet(greeting) {
  console.log(greeting + ', ' + this.name)
}

greet.myCall({ name: '张三' }, '你好') // 你好, 张三

6.2 手写 apply

Function.prototype.myApply = function (context, args) {
  if (typeof this !== 'function') {
    throw new TypeError('Error')
  }

  context = context || window
  const fnKey = Symbol()
  context[fnKey] = this

  // args 可能是 undefined 或 null
  const result = args ? context[fnKey](...args) : context[fnKey]()

  delete context[fnKey]
  return result
}

// 测试
function sum(a, b) {
  console.log(this.name, a + b)
}

sum.myApply({ name: '计算结果' }, [1, 2]) // 计算结果 3

6.3 手写 bind

Function.prototype.myBind = function (context, ...outerArgs) {
  if (typeof this !== 'function') {
    throw new TypeError('Error')
  }

  const fn = this

  const boundFn = function (...innerArgs) {
    // 如果是 new 调用，this 指向新创建的实例
    // 否则使用绑定的 context
    return fn.apply(
      this instanceof boundFn ? this : context,
      [...outerArgs, ...innerArgs]
    )
  }

  // 维持原型链
  boundFn.prototype = Object.create(fn.prototype)

  return boundFn
}

// 测试
const person = { name: '张三' }

function greet(greeting, punctuation) {
  console.log(greeting + ', ' + this.name + punctuation)
}

const greetPerson = greet.myBind(person, '你好')
greetPerson('！') // 你好, 张三！

// new 调用测试
function Person(name) {
  this.name = name
}

const BoundPerson = Person.myBind({ name: '默认' })
const p = new BoundPerson('李四')
console.log(p.name) // '李四'（new 调用时忽略绑定的 this）

七、实战练习

练习一：基础绑定

var obj = {
  a: 1,
  say: function () {
    console.log('this1:', this)

    var sayA = function () {
      console.log('this2:', this)
    }

    function sayB() {
      console.log('this3:', this)
    }

    sayA() // window（独立调用）
    sayB() // window（独立调用）
  }
}

obj.say() // this1: obj（obj 调用）
// 输出：this1: obj, this2: window, this3: window

练习二：构造函数

function Person(name, age) {
  this.name = name
  this.age = age
}

var person1 = new Person('张三', 18)
var person2 = Person('李四', 12) // 忘记 new，this 指向 window

console.log(person1) // { name: '张三', age: 18 }
console.log(person2) // undefined（函数没有返回值）
console.log(window.name, window.age) // '李四', 12（污染全局）

练习三：箭头函数

const obj = {
  func() {
    const arrowFunc = () => {
      console.log(this)
    }
    return arrowFunc
  },
  _name: 'obj'
}

// 箭头函数的 this 继承自 func
// func 的 this 取决于调用方式

obj.func()() // obj（obj 调用 func）

const func = obj.func
func()() // window（独立调用 func）

obj.func.bind({ _name: 'bindObj' })()() // { _name: 'bindObj' }
obj.func.bind()()() // window（bind 无参数，this 为 window）

// bind 返回的函数 this 不可再改变
obj.func.bind({ _name: 'bindObj' }).apply({ _name: 'applyObj' })()
// 输出：{ _name: 'bindObj' }（bind 的 this 不可改变）

练习四：arguments 中的 this

var length = 10

function fn() {
  console.log(this.length)
}

var obj = {
  length: 5,
  method: function (fn) {
    fn() // 10（独立调用，this 为 window）
    arguments[0]() // 2（arguments 调用，this 为 arguments）
  }
}

obj.method(fn, 1) // arguments = { 0: fn, 1: 1, length: 2 }
// 输出：10, 2

练习五：综合题

var name = 'window'

var obj1 = {
  name: 'obj1',
  foo: function () {
    console.log(this.name)
  }
}

var obj2 = {
  name: 'obj2',
  foo: () => {
    console.log(this.name)
  }
}

obj1.foo() // 'obj1'（obj1 调用）
obj2.foo() // 'window'（箭头函数继承外层 this）

obj1.foo.call(obj2) // 'obj2'（显式绑定）
obj2.foo.call(obj1) // 'window'（箭头函数 this 不可改变）

new obj1.foo() // undefined（new 调用，this 指向新实例，没有 name 属性）
// new obj2.foo() // 报错（箭头函数不能作为构造函数）

八、总结

记住判断 this 指向的四个步骤：

  1. 是否是箭头函数？ → 继承定义时外层的 this
  2. 是否使用 new 调用？ → this 指向新创建的实例
  3. 是否使用 call/apply/bind？ → this 指向绑定的对象
  4. 是否作为对象方法调用？ → this 指向该对象
  5. 以上都不是？ → 严格模式为 undefined，非严格模式为 window
