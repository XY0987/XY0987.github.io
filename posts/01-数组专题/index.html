数组专题

发表于2023|更新于2026|算法
|字数总计:1.2k|阅读时长:5分钟

数组是数据结构中最基础也是最重要的一种结构，本文整理了数组相关的经典算法题目。

二分查找

二分法，是定义一个中间值变量，比对两端点的值，然后更换搜索区间。

/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
  let left = 0;
  let right = nums.length - 1;
  // 搜索区间是左闭右闭，while循环值以及里边的判断值要判断区间的合法性
  while (left <= right) {
    const middle = (left + right) >> 1;
    if (nums[middle] > target) {
      // 中间值大于目标值，搜索区间中不用包含该值
      right = middle - 1;
    } else if (nums[middle] < target) {
      // 中间值小于目标值，搜索区间也不用包含该值
      left = middle + 1;
    } else {
      // 相等，获得目标结果
      return middle;
    }
  }
  // 没有找到
  return -1;
};

删除元素

/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */

/*
如果匹配到后边的值会前移覆盖掉，i指针会指向下一个值，后续
如果有符合要求的话会继续覆盖值
*/
var removeElement = function (nums, val) {
  // 记录有几个符合规则的数，后边值进行覆盖
  let i = 0;
  for (let j = 0; j < nums.length; j++) {
    // 符合规矩直接修改值
    if (nums[j] !== val) {
      nums[i] = nums[j];
      i++;
    }
  }
  return i;
};

长度最小的子数组（滑动窗口）

/**
 * @param {number} target
 * @param {number[]} nums
 * @return {number}
 */
var minSubArrayLen = function (target, nums) {
  let sum = 0;
  let left = 0;
  // 将结果值初始定为最大值(因为结果值最大值也不会超过num.length+1)
  let res = nums.length + 1;
  // 一重个for循环，循环里边使用while语句去找滑动窗口的值
  for (let i = 0; i < nums.length; i++) {
    sum += nums[i];
    // 如果大于目标值，移动左指针，并还原sum值(大于等于所以包含结果未1的情况)
    while (sum >= target) {
      sum -= nums[left];
      // 因为i从0开始所以加1
      res = Math.min(res, i - left + 1);
      left++;
    }
  }
  // res值是否未更新(未更新表示所有和都达不到)
  return res > nums.length ? 0 : res
};

/**
 * @param {number} n
 * @return {number[][]}
 */
var generateMatrix = function (n) {
  // 定义四个边界值
  let l = 0, r = n - 1, t = 0, b = n - 1;
  let res = Array(n).fill().map(() => Array(n));
  let num = 1, tar = n * n;
  while (num <= tar) {
    for (let i = l; i <= r; i++) res[t][i] = num++; // 从左到右
    t++;
    for (let i = t; i <= b; i++) res[i][r] = num++; // 从上到下
    r--;
    for (let i = r; i >= l; i--) res[b][i] = num++; // 从右到左
    b--;
    for (let i = b; i >= t; i--) res[i][l] = num++; // 从下到上
    l++;
  }
  return res;
};

找出满足差值条件的下标 I

const nums = [5, 1, 4, 1],
  indexDifference = 2,
  valueDifference = 4;

for (let i = 0; i < nums.length; i++) {
  for (let j = 0; j < nums.length; j++) {
    const value1 = Math.abs(i - j);
    const value2 = Math.abs(nums[i] - nums[j]);
    if (value1 >= indexDifference && value2 >= valueDifference) {
      return [i, j];
    }
  }
}
return [-1, -1];

找出满足差值条件的下标 II

const nums = [5, 1, 4, 1],
  indexDifference = 2,
  valueDifference = 4;

let minIndex = 0;
let maxIndex = 0;
for (let i = indexDifference; i < nums.length; i++) {
  let j = i - indexDifference;
  /* 
  这里并不是一直取正好差值为indexDifference，而是取最小的值,如果大于最大值才会切换值，如果小于最小值才会切换值
  */
  if (nums[j] > nums[maxIndex]) {
    maxIndex = j;
  }
  if (nums[j] < nums[minIndex]) {
    minIndex = j;
  }
  // 判断是否有满足条件的结果
  if (nums[maxIndex] - nums[i] >= valueDifference) {
    return [i, maxIndex];
  }
  if (nums[i] - nums[minIndex] >= valueDifference) {
    return [minIndex, i];
  }
}
return [-1, -1];

元素和最小的山形三元组 II

const nums = [1, 3, 2, 2];

/* 
dl数组用于存储小于该下标的最小值，左边的值
dr数组用于存储大于该下标的最小值，右边的值
*/

const dl = new Array(nums.length).fill(0);
const dr = new Array(nums.length).fill(0);
dl[0] = nums[0];
dr[nums.length - 1] = nums[nums.length - 1];

for (let i = 1; i < nums.length; i++) {
  dl[i] = Math.min(dl[i - 1], nums[i]);
}

for (let i = nums.length - 2; i >= 0; i--) {
  dr[i] = Math.min(dr[i + 1], nums[i]);
}

let res = Infinity;
for (let j = 1; j < nums.length - 1; j++) {
  if (dl[j - 1] < nums[j] && nums[j] > dr[j + 1]) {
    res = Math.min(res, nums[j] + dl[j - 1] + dr[j + 1]);
  }
}
return res === Infinity ? -1 : res;

dl数组保存从左到右的最小值
dr数组保存从右到左的最小值
