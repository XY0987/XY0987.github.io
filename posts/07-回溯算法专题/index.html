回溯算法专题

发表于2023|更新于2026|算法
|字数总计:2.5k|阅读时长:12分钟

回溯与递归相辅相成，回溯是一种纯暴力的搜索算法。

回溯理论基础

解决以下问题：

  1. 组合问题（强调没有顺序）
  2. 切割问题
  3. 子集问题
  4. 排列问题（强调元素的顺序比如1，2；2，1两种排列）
  5. 棋盘问题

模板：一般是无返回值，函数名一般叫做backtracking

void backtracking(){
  if(终止条件){
    收集结果
    return;
  }
  for(集合的元素集){
    处理节点
    递归函数
    回溯
  }
}

组合

/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
// 无剪枝
var combine = function (n, k) {
  // 不强调顺序,1,2与2,1是一个组合
  const res = [];
  const path = [];
  // startIndex用于记录开始的取值范围，避免重复取值
  function backtracking(n, k, startIndex) {
    if (path.length === k) {
      // 拷贝数组，否则是值引用，最后还是空数组
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i <= n; i++) {
      path.push(i);
      // 从之后位置开始搜索
      backtracking(n, k, i + 1);
      path.pop();
    }
  }
  backtracking(n, k, 1);
  return res
};

剪枝：判断剩余长度，如果小于，则直接返回

组合总和三

/**
 * @param {number} k
 * @param {number} n
 * @return {number[][]}
 */
var combinationSum3 = function (k, n) {
  const res = [];
  const path = [];
  let sum = 0;
  function backtracking(n, k, startIndex) {
    if (sum === n && path.length === k) {
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i <= 9; i++) {
      if (path.length > k) {
        return;
      }
      sum += i;
      path.push(i);
      backtracking(n, k, i + 1);
      sum -= i;
      path.pop();
    }
  }
  backtracking(n, k, 1);
  return res
};

电话号码的字母组合

/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
  if (digits === '') {
    return []
  }
  const map = new Map();
  map.set("1", "");
  map.set("2", "abc");
  map.set("3", "def");
  map.set("4", "ghi");
  map.set("5", "jkl");
  map.set("6", "mno");
  map.set("7", "pqrs");
  map.set("8", "tuv");
  map.set("9", "wxyz");
  const strArr = digits.split("");
  const res = [];
  const path = [];
  function backtracking(startIndex) {
    if (path.length === strArr.length) {
      res.push(path.join(""));
      return;
    }
    const str = map.get(strArr[startIndex]);
    for (let i = 0; i < str.length; i++) {
      path.push(str[i]);
      backtracking(startIndex + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res
};

组合总和

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum = function (candidates, target) {
  // 遍历之后的数组不能带前面的   2,5与5,2是同一个,所以后面的不用再选取前面的
  const res = [];
  const path = [];
  let sum = 0;
  /* 
    2,5和5,2是同一个，所以再之后的选取中，不能再选取该元素前面的元素了
    */
  function backtracking(startIndex) {
    if (sum > target) {
      return;
    }
    if (sum === target) {
      res.push([...path]);
      return;
    }
    for (let i = startIndex; i < candidates.length; i++) {
      sum += candidates[i];
      path.push(candidates[i]);
      // 可以重复选取，所以从id开始
      backtracking(i);
      sum -= candidates[i];
      path.pop();
    }
  }
  backtracking(0);
  return res
};

组合总和二

去重做的是树层去重

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum2 = function (candidates, target) {
  candidates.sort((a, b) => a - b);
  const path = [];
  const res = new Set();
  let sum = 0;
  function backtracking(startIndex) {
    if (sum > target) {
      return;
    }
    if (sum === target) {
      const temPath = [...path];
      // 对原数组排序的话，回溯时会导致顺序错乱
      res.add(
        temPath
        .sort((a, b) => {
          return a - b;
        })
        .join(",")
      );
      return;
    }
    for (let i = startIndex; i < candidates.length; i++) {
      /* 
      数组已经排过序，再进入第二次搜索时，以该元素开头的结果已经搜索过了
      就不用再次搜索了
      */
      if (i > startIndex && candidates[i] === candidates[i - 1]) {
        continue;
      }
      sum += candidates[i];
      path.push(candidates[i]);
      backtracking(i + 1);
      sum -= candidates[i];
      path.pop();
    }
  }
  backtracking(0);
  const ans = Array.from(res).map((item) => {
    return item.split(",");
  });
  return ans
};

分割回文串

字符串回溯放的是一段，之前回溯放的是一个，其本质都是以不同起点开头，遍历到终点的一个过程，只不过字符串表示的是以这个为起点为分割线，前面的是一体的

/**
 * @param {string} s
 * @return {string[][]}
 */
var partition = function (s) {
  const path = [];
  const res = [];
  // 判断是否是回文串
  const isPalindrome = (s, l, r) => {
    for (let i = l, j = r; i < j; i++, j--) {
      if (s[i] !== s[j]) return false;
    }
    return true;
  };
  function backtracking(startIndex) {
    // path收集的是同一起点到终点之间的所有回文子串
    if (startIndex >= s.length) {
      res.push([...path]);
      return;
    }
    // 开始的for循环都是第一层，之后递归是之后的层
    // 每次遍历都是不同的起点，遍历到最后(0 到最后之间的回文串，1到最后之间的回文串.......)
    for (let i = startIndex; i < s.length; i++) {
      // 判断startIndex到i之间是否是回文串
      if (!isPalindrome(s, startIndex, i)) {
        continue;
      }
      // 放的是一段字符串
      path.push(s.slice(startIndex, i + 1));
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res;
};

复原IP地址

/**
 * @param {string} s
 * @return {string[]}
 */
var restoreIpAddresses = function (s) {
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    // 终止条件
    if (path.length > 4) {
      return;
    }
    if (path.length === 4 && startIndex === s.length) {
      res.push(path.join("."));
      return;
    }
    // 单层递归逻辑
    for (let i = startIndex; i < s.length; i++) {
      // 判断截取的字符串是否满足
      const str = s.slice(startIndex, i + 1);
      // 分割的字符串大于255（长度大于4的原因是避免有0）
      if (str.length > 4 || parseInt(str) > 255) {
        break;
      }
      // 不是一个数字，开头是以0开头的，也不行
      if (str.length > 1 && str[0] === "0") {
        break;
      }
      path.push(str);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res
};

子集

与组合问题不同的是收集结果的位置

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function (nums) {
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    if (startIndex >= nums.length) {
      return;
    }
    for (let i = startIndex; i < nums.length; i++) {
      path.push(nums[i]);
      res.push([...path]);
      backtracking(i + 1);
      path.pop();
    }
  }

  backtracking(0);
  res.push([]);
  return res
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsetsWithDup = function (nums) {
  const path = [];
  const res = [];
  nums.sort((a, b) => {
    return a - b;
  });
  function backtracking(startIndex) {
    if (startIndex >= nums.length) {
      return;
    }
    for (let i = startIndex; i < nums.length; i++) {
      // 前面的元素已经收集过了，就不用再收集相同元素了(大于startIndex表示单层),树层去重
      if (i > startIndex && nums[i] === nums[i - 1]) {
        continue;
      }
      path.push(nums[i]);
      res.push([...path]);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  res.push([]);
  return res
};

递增子序列

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var findSubsequences = function (nums) {
  // 不能进行排序,因为排序回导致数据改变,导致一些不必要的答案
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    if (path.length >= 2) {
      res.push([...path]);
    }
    if (startIndex >= nums.length) {
      return;
    }
    // 去重
    const set = new Set();
    for (let i = startIndex; i < nums.length; i++) {
      if (
        set.has(nums[i]) ||
        (path.length > 0 && nums[i] < path[path.length - 1])
      ) {
        continue;
      }
      set.add(nums[i]);
      path.push(nums[i]);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res;
};

全排列

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function (nums) {
  const res = [];
  const path = [];
  const used = new Array(nums.length).fill(0);

  function backtracking() {
    if (path.length === nums.length) {
      res.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) {
        continue;
      }
      path.push(nums[i]);
      used[i] = 1;
      backtracking();
      path.pop();
      used[i] = 0;
    }
  }
  backtracking();
  return res;
};

全排列二

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permuteUnique = function (nums) {
  const res = [];
  const path = [];
  const used = new Array(nums.length).fill(0);
  nums.sort((a, b) => {
    return a - b;
  });
  function backtracking() {
    if (path.length === nums.length) {
      res.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      /* 
    树层去重(前一个数没有用过，才需要进行树层去重)，否则不会拿到结果
    被用过了相当于树枝，树枝不用去重
    以该元素开头的，前边以该元素开头的一定包含该元素开头的所有情况
    */
      if (i > 0 && !used[i - 1] && nums[i] === nums[i - 1]) {
        continue;
      }
      // 当前数没有被用
      if (!used[i]) {
        path.push(nums[i]);
        used[i] = 1;
        backtracking();
        path.pop();
        used[i] = 0;
      }
    }
  }
  backtracking();
  return res;
};

N 皇后

/**
 * @param {number} n
 * @return {string[][]}
 */
var solveNQueens = function (n) {
  /* 
      每一层递归对应的是不同的行，循环种遍历该行的不同列
    */
  const result = [];
  // 判断是否符合规则（同一行，同一列，斜45度是否有皇后，能不能放皇后）
  function isValid(row, col, chessBoard, n) {
    for (let i = 0; i < row; i++) {
      if (chessBoard[i][col] === "Q") {
        return false;
      }
    }
    for (let i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
      if (chessBoard[i][j] === "Q") {
        return false;
      }
    }
    for (let i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
      if (chessBoard[i][j] === "Q") {
        return false;
      }
    }
    return true;
  }
  // 将棋盘转化为最终结果
  function transformChessBoard(chessBoard) {
    let chessBoardBack = [];
    chessBoard.forEach((row) => {
      let rowStr = "";
      row.forEach((value) => {
        rowStr += value;
      });
      chessBoardBack.push(rowStr);
    });

    return chessBoardBack;
  }
  function backtracing(row, chessBoard) {
    if (row === n) {
      result.push(transformChessBoard(chessBoard));
      return;
    }
    // 遍历该行的每一列
    for (let col = 0; col < n; col++) {
      // 判断是否符合规范
      if (isValid(row, col, chessBoard, n)) {
        chessBoard[row][col] = "Q";
        backtracing(row + 1, chessBoard);
        chessBoard[row][col] = ".";
      }
    }
  }
  let chessBoard = new Array(n).fill([]).map(() => new Array(n).fill("."));
  backtracing(0, chessBoard);
  return result;
};

解数独

/**
 * @param {character[][]} board
 * @return {void} Do not return anything, modify board in-place instead.
 */
var solveSudoku = function (board) {
  // 判断值是否符合规范
  function isValid(row, col, val, board) {
    let len = board.length;
    // 行不能重复
    for (let i = 0; i < len; i++) {
      if (board[row][i] === val) {
        return false;
      }
    }
    // 列不能重复
    for (let i = 0; i < len; i++) {
      if (board[i][col] === val) {
        return false;
      }
    }
    let startRow = Math.floor(row / 3) * 3;
    let startCol = Math.floor(col / 3) * 3;
    for (let i = startRow; i < startRow + 3; i++) {
      for (let j = startCol; j < startCol + 3; j++) {
        if (board[i][j] === val) {
          return false;
        }
      }
    }
    return true;
  }
  // 回溯
  function backTracking() {
    // 循环每一行
    for (let i = 0; i < board.length; i++) {
      // 循环每一列
      for (let j = 0; j < board[0].length; j++) {
        // 只处理值是点的
        if (board[i][j] !== ".") continue;
        // 将该位置存放1到9，是否符合规则
        for (let val = 1; val <= 9; val++) {
          if (isValid(i, j, `${val}`, board)) {
            board[i][j] = `${val}`;
            // 遇到结果，立即返回
            if (backTracking()) {
              return true;
            }
            board[i][j] = `.`;
          }
        }
        return false;
      }
    }
    return true;
  }
  backTracking(board);
  return board;
};
