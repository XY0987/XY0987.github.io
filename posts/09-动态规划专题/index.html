动态规划是通过把原问题分解为相对简单的子问题的方式求解复杂问题的方法。

动态规划理论基础

动规基础、背包问题、打家劫舍、股票问题、子序列问题

解题五部曲：

  1. dp数组的含义，以及下标的含义
  2. 递推公式
  3. dp数组如何初始化
  4. 遍历顺序
  5. 打印dp数组(查看是否有问题)

var fib = function (n) {
  let dp = [0, 1];
  for (let i = 2; i <= n; i++) {
    dp[i] = dp[i - 1] + dp[i - 2];
  }

  return dp[n];
};

使用最小花费爬楼梯

/**
 * @param {number[]} cost
 * @return {number}
 */
var minCostClimbingStairs = function (cost) {
  /* 
    dp数组的含义是到达i位置花费为dp[i]
    跳到i阶之后所需要的最小花费从dp[i-1]+cost[i-1]和dp[i-2]+cost[i-2]中取
    第i阶可以从i-1或i-2中跳到
    */
  // 往上跳才花费体力（起点可以从0或者1开始）
  const dp = [0, 0];
  for (let i = 2; i <= cost.length; i++) {
    dp[i] = Math.min(dp[i - 1] + cost[i - 1], dp[i - 2] + cost[i - 2]);
  }
  return dp[cost.length]
};

不同路径

/**
 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var uniquePaths = function (m, n) {
  /* 
    dp数组的含义：
    从0 0开始到i j的路径总和
    
    dp[i][j] 只能从dp[i-1][j]或者dp[i][j-1]走到，二者相加就是该点的路径总和
    */

  // 全部初始化为1
  const dp = new Array(m).fill(1).map(() => Array(n).fill(1));
  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      dp[i][j] = dp[i][j - 1] + dp[i - 1][j];
    }
  }
  return dp[m - 1][n - 1]
};

不同路径二

/**
 * @param {number[][]} obstacleGrid
 * @return {number}
 */
var uniquePathsWithObstacles = function (obstacleGrid) {
  /* 
    dp数组表示 0 0位置到i j的路径和
    */

  // 初始化时，遇到障碍，之后的都应该时0(第一行和第一列)
  const m = obstacleGrid.length;
  const n = obstacleGrid[0].length;
  const dp = Array(m)
    .fill()
    .map(() => Array(n).fill(0));

  // 初始化行
  for (let i = 0; i < m && obstacleGrid[i][0] === 0; i++) {
    dp[i][0] = 1;
  }
  // 初始化列
  for (let i = 0; i < n && obstacleGrid[0][i] === 0; i++) {
    dp[0][i] = 1;
  }

  for (let i = 1; i < m; i++) {
    for (let j = 1; j < n; j++) {
      // 如果路径数组该位置有障碍物，不用计算
      if (obstacleGrid[i][j] === 0) {
        dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
      }
    }
  }
  return dp[m - 1][n - 1]
};

整数拆分

/**
 * @param {number} n
 * @return {number}
 */
var integerBreak = function (n) {
  /* 
    数值近似相等，最终乘积才会尽可能的大
    
    dp[i] i拆的乘积最大的数
    
    乘积最大分 i*n-i或i*dp[n-i] 或者dp[i]
    */

  let dp = new Array(n + 1).fill(0);
  dp[2] = 1;

  for (let i = 3; i <= n; i++) {
    for (let j = 1; j <= i / 2; j++) {
      // 求最大值(和为i)
      dp[i] = Math.max(dp[i], dp[i - j] * j, (i - j) * j);
    }
  }
  return dp[n];
};

n 为 3 的节点，可以从 n 为 2 和 n 为 1 推导出来，分别以 1 为根节点、2 为根节点和 3 为根节点

/**
 * @param {number} n
 * @return {number}
 */
var numTrees = function (n) {
  const dp = new Array(n + 1).fill(0);
  dp[0] = 1;
  dp[1] = 1;
  // i表示节点总数
  for (let i = 2; i <= n; i++) {
    /*
    j-1（根节点）表示左子树有多少个也表示以不同值为头节点的情况，剩余的是右子树的数量
    */
    for (let j = 1; j <= i; j++) {
      /*
      以j为头节点左子树有j-1个节点
      右子树有i-j个节点
      dp[3]需要以1为头节点加上2为头节点加上3为头节点，j就等于这1为头节点
      */
      dp[i] += dp[j - 1] * dp[i - j];
    }
  }
  return dp[n];
};

背包问题理论基础

01背包（二维数组）

dp[i][j] 表示[0,i]物品，任取放入容量为 j 的背包中的最大价值
第 i 个物品有两种状态，放入物品 i 以及不放入物品 i

不放物品 i 最大价值为：dp[i-1][j] （最大值就是上一个的最大值，物品重量不变）
放物品 i 的最大价值为: dp[i-1]j-weight[i]]+value[i]（j-weight[i]减去当前重量剩余的重量）（上一个最大值，物品重量为减去当前重量）

二维数组双重 for 循环是可以颠倒顺序的，先遍历背包或先遍历物品都是可以的，只要保证上方和左上方有值就可以推出当前的价值

01背包（一维数组）

把二维数组降为一维数组
dp[j] 表示容量为 j 的背包所有装的最大价值
不能颠倒遍历顺序
另外遍历背包需要倒叙遍历，否则会出现重复添加一个物品的情况

完全背包理论基础

01 背包一个物品只能使用一次，完全背包一个物品可以使用无数次
把第二层循环中的遍历倒叙改为正序
第一层遍历物品，第二层遍历背包，两层 for 循环可以颠倒

分割等和子集

/**
 * @param {number[]} nums
 * @return {boolean}
 */
var canPartition = function (nums) {
  /* 
    物品的价值和重量是相同的
    能凑成一半值，就返回true
    相当于一个背包问题，容量是11，价值也是11
    
    dp[j] 表示容量为j最大价值为dp[j]
    
    dp[j]=Math.max(dp[j],dp[j-nums[i]]+nums[i])
    */
  const sum = nums.reduce((prev, next) => {
    return prev + next;
  }, 0);
  if (sum % 2 === 1) {
    return false;
  }
  const target = sum / 2;
  const dp = new Array(target + 1).fill(0);

  // 第一层遍历物品
  for (let i = 0; i < nums.length; i++) {
    // 倒叙遍历,遍历背包
    for (let j = target; j >= nums[i]; j--) {
      dp[j] = Math.max(dp[j], dp[j - nums[i]] + nums[i]);
      if (dp[j] === target) {
        return true;
      }
    }
  }
  return false;
};

最后一块石头的重量二

/**
 * @param {number[]} stones
 * @return {number}
 */
var lastStoneWeightII = function (stones) {
  /* 
    重量为stones[i]价值也为stones[i]
    
    尽量把重量总和相近的分在一起
    
    dp[j] 表示装满容量为j的最大重量为dp[j]
    
    dp[j]=Math.max(dp[j],dp[j-stones[i]]+stones[i])
    */

  // 石头重量最多为3000，初始化为1510，多出来的10是边界情况
  const dp = new Array(1510).fill(0);
  // 凑sum/2,尽量凑总重量的一半,这样的话才能使剩的石头重量最小
  const sum = stones.reduce((prev, next) => {
    return prev + next;
  }, 0);
  const target = parseInt(sum / 2);
  for (let i = 0; i < stones.length; i++) {
    //物品
    // 如果背包容量比石头重量还小，就没有意义了
    for (let j = target; j >= stones[i]; j--) {
      //背包
      dp[j] = Math.max(dp[j], dp[j - stones[i]] + stones[i]);
    }
  }
  return sum - dp[target] - dp[target];
};

目标和

/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var findTargetSumWays = function (nums, target) {
  /* 
构造出两种集合，一个是正数集合，一种四负数集合
left表示正数集合
right表示负数集合
left+right=sum
left-right=target

right=sum-left
left-(sum-left)=target
left=(target+sum)/2 不能整除，就表示没有对应组合

dp[j] 表示装满背包容量为j有dp[j]种方法
*/

  const sum = nums.reduce((prev, next) => {
    return prev + next;
  }, 0);

  // 绝对值大于sum无法凑出来
  if (Math.abs(target) > sum) {
    return 0;
  }

  // 无法组成结果的情况
  if ((target + sum) % 2 !== 0) {
    return 0;
  }
  let left = (target + sum) / 2;

  const dp = new Array(left + 1).fill(0);
  dp[0] = 1;
  for (let i = 0; i < nums.length; i++) {
    for (let j = left; j >= nums[i]; j--) {
      dp[j] += dp[j - nums[i]];
    }
  }

  return dp[left];
};

一和零

/**
 * @param {string[]} strs
 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var findMaxForm = function (strs, m, n) {
  /* 
    dp[i][j] 表示 i个0 j个1 最大装dp[i][j]个物品
  */
  const dp = Array.from(Array(m + 1), () => Array(n + 1).fill(0));
  let numOfZeros, numOfOnes;

  for (let str of strs) {
    numOfZeros = 0;
    numOfOnes = 0;

    for (let c of str) {
      if (c === "0") {
        numOfZeros++;
      } else {
        numOfOnes++;
      }
    }

    for (let i = m; i >= numOfZeros; i--) {
      for (let j = n; j >= numOfOnes; j--) {
        dp[i][j] = Math.max(dp[i][j], dp[i - numOfZeros][j - numOfOnes] + 1);
      }
    }
  }
  return dp[m][n];
};

零钱兑换二

/**
 * @param {number} amount
 * @param {number[]} coins
 * @return {number}
 */
var change = function (amount, coins) {
  /* 
    dp[j]表示装满容量为j的背包有dp[j]种方法

    本题不能颠倒遍历顺序，因为要组合数（元素的顺序无关）
    颠倒顺序的话是排列数
    */

  const dp = new Array(amount + 1).fill(0);
  dp[0] = 1;
  for (let i = 0; i < coins.length; i++) {
    for (let j = coins[i]; j <= amount; j++) {
      dp[j] += dp[j - coins[i]];
    }
  }
  return dp[amount];
};

/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var combinationSum4 = function (nums, target) {
  /* 
    强调元素顺序
    
    dp[j] 表示装满容量为j的背包有dp[j]种方法
    
    先遍历背包在遍历物品
    */

  const dp = new Array(target + 1).fill(0);
  dp[0] = 1;
  for (let i = 0; i <= target; i++) {
    for (let j = 0; j < nums.length; j++) {
      if (i >= nums[j]) {
        dp[i] += dp[i - nums[j]];
      }
    }
  }

  return dp[target];
};

零钱兑换

/**
 * @param {number[]} coins
 * @param {number} amount
 * @return {number}
 */
var coinChange = function (coins, amount) {
  /* 
    dp[j] 表示装满容量为j的背包所需要的最少钱币数
    */

  const dp = new Array(amount + 1).fill(Number.MAX_SAFE_INTEGER);
  dp[0] = 0;

  for (let i = 0; i < coins.length; i++) {
    for (let j = coins[i]; j <= amount; j++) {
      // 减去当前钱币值之后去最小值
      dp[j] = Math.min(dp[j - coins[i]] + 1, dp[j]);
    }
  }
  return dp[amount] === Number.MAX_SAFE_INTEGER ? -1 : dp[amount];
};

完全平方数

/**
 * @param {number} n
 * @return {number}
 */
var numSquares = function (n) {
  /* 
    dp[j] 表示装满容量为j的最少完全平方数为dp[j]
    */
  const dp = new Array(n + 1).fill(Number.MAX_SAFE_INTEGER);
  dp[0] = 0;
  for (let i = 1; i * i <= n; i++) {
    const num = i * i;
    for (let j = num; j <= n; j++) {
      dp[j] = Math.min(dp[j - num] + 1, dp[j]);
    }
  }
  return dp[n];
};

单词拆分

/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function (s, wordDict) {
  /* 
    字符串是一个背包，单词是一个物品，单词能否装满背包
    
    dp[j] 表示长度为j的字符串能否被字典组成
    
    讲究排列数
    */

  let dp = Array(s.length + 1).fill(false);
  dp[0] = true;
  for (let i = 0; i <= s.length; i++) {
    for (let j = 0; j < wordDict.length; j++) {
      if (i >= wordDict[j].length) {
        if (
          s.slice(i - wordDict[j].length, i) === wordDict[j] &&
          dp[i - wordDict[j].length]
        ) {
          dp[i] = true;
        }
      }
    }
  }
  return dp[s.length];
};

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
  /* 
    dp[i] 表示下标i之前的包含下标i 偷到的最大钱数是dp[i]
    
    两种状态，偷i和不偷i
    偷i最大值为dp[i-2]+nums[i]
    不偷最大值为 dp[i-1]
    */

  if (nums.length <= 2) {
    return Math.max(...nums);
  }

  const dp = new Array(nums.length).fill(0);
  dp[0] = nums[0];
  dp[1] = Math.max(nums[0], nums[1]);
  for (let i = 2; i < nums.length; i++) {
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
  }
  return dp[nums.length - 1]
};

打家劫舍二

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
  /* 
首尾相邻，不能偷相邻元素
三种情况：
1. 不考虑首和尾
2. 不考虑尾部
3. 不考虑首部

情况2和情况3包含情况1

dp[i] 表示偷下标i之前包含下标i所能偷到的最大金额
*/

  if (nums.length <= 3) {
    return Math.max(...nums);
  }

  const dp1 = new Array(nums.length - 1).fill(0);
  // 不考虑尾元素
  dp1[0] = nums[0];
  dp1[1] = Math.max(nums[0], nums[1]);
  for (let i = 2; i < nums.length - 1; i++) {
    dp1[i] = Math.max(dp1[i - 1], dp1[i - 2] + nums[i]);
  }
  // 不考虑首元素
  const dp2 = new Array(nums.length - 1).fill(0);
  dp2[0] = nums[1];
  dp2[1] = Math.max(nums[1], nums[2]);
  // 循环牵涉着dp2数组，所以不能变，变一下nums数组的取值
  for (let i = 2; i < nums.length - 1; i++) {
    dp2[i] = Math.max(dp2[i - 1], dp2[i - 2] + nums[i + 1]);
  }

  return Math.max(dp1[dp1.length - 1], dp2[dp2.length - 1]);
};

买卖股票的最佳时机一

右边的最大值减去左边的最小值

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
  持有不代表这一天买该股票
    dp[i][0] 表示持有该股票获得的最大金钱
    dp[i][1] 表示不持有该股票获得的最大金钱
     不持有不代表这一天卖出该股票
    */
  const len = prices.length;
  // 创建dp数组
  const dp = new Array(len).fill([0, 0]);
  // dp数组初始化
  dp[0] = [-prices[0], 0];
  for (let i = 1; i < len; i++) {
    // 更新dp[i]
    /*
    dp[i][0] 可以从前一天推出，也可以这一天买股票(-prices[i] 表示买这一天的股票)
    dp[i][1] 可以从前一天推出，也可以卖出这一天的股票
    */
    dp[i] = [
      Math.max(dp[i - 1][0], -prices[i]),
      Math.max(dp[i - 1][1], prices[i] + dp[i - 1][0]),
    ];
  }
  return dp[len - 1][1];
};

};

/**
 * @param {number} k
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (k, prices) {
  if (prices == null || prices.length < 2 || k == 0) {
    return 0;
  }
  let dp = Array.from(Array(prices.length), () => Array(2 * k + 1).fill(0));
  for (let j = 1; j < 2 * k; j += 2) {
    dp[0][j] = 0 - prices[0];
  }
  for (let i = 1; i < prices.length; i++) {
    for (let j = 0; j < 2 * k; j += 2) {
      dp[i][j + 1] = Math.max(dp[i - 1][j + 1], dp[i - 1][j] - prices[i]);
      dp[i][j + 2] = Math.max(dp[i - 1][j + 2], dp[i - 1][j + 1] + prices[i]);
    }
  }
  // console.log(dp[prices.length - 1][2 * k]);
  return dp[prices.length - 1][2 * k];
};

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
dp[i][0] 表示持有股票
dp[i][1] 表示卖出股票的状态
dp[i][2] 表示卖出股票(前一天卖出)
dp[i][3] 表示冷冻期
*/

  if (prices.length < 2) {
    return 0;
  } else if (prices.length < 3) {
    return Math.max(0, prices[1] - prices[0]);
  }
  let dp = Array.from(Array(prices.length), () => Array(4).fill(0));
  dp[0][0] = 0 - prices[0];
  for (i = 1; i < prices.length; i++) {
    dp[i][0] = Math.max(
      dp[i - 1][0],
      Math.max(dp[i - 1][1], dp[i - 1][3]) - prices[i]
    );
    dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][3]);
    dp[i][2] = dp[i - 1][0] + prices[i];
    dp[i][3] = dp[i - 1][2];
  }
  // console.log(
  //   Math.max(
  //     dp[prices.length - 1][1],
  //     dp[prices.length - 1][2],
  //     dp[prices.length - 1][3]
  //   )
  // );
  return Math.max(
    dp[prices.length - 1][1],
    dp[prices.length - 1][2],
    dp[prices.length - 1][3]
  );
};

/**
 * @param {number[]} prices
 * @param {number} fee
 * @return {number}
 */
var maxProfit = function (prices, fee) {
  /* 
    dp[i][0] 表示持有股票的最大金额
    dp[i][1] 表示不持有股票的最大金额
    */
  let dp = Array.from(Array(prices.length), () => Array(2).fill(0));
  dp[0][0] = 0 - prices[0];
  for (let i = 1; i < prices.length; i++) {
    dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] - prices[i]);
    dp[i][1] = Math.max(dp[i - 1][0] + prices[i] - fee, dp[i - 1][1]);
  }
  return Math.max(dp[prices.length - 1][0], dp[prices.length - 1][1]);
};

/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function (nums) {
  /* 
      dp[i] 表示以nums[i] 结尾的最长递增子序列的长度
      dp[i]=Math.max(dp[j]+1,dp[i])
  */
  const dp = new Array(nums.length).fill(1);
  for (let i = 0; i < nums.length; i++) {
    for (let j = 0; j < i; j++) {
      if (nums[i] > nums[j]) {
        dp[i] = Math.max(dp[j] + 1, dp[i]);
      }
    }
  }
  // 最终结果不一定是结尾，dp[i] 表示以num[i]结尾的最长递增子序列的长度
  return Math.max(...dp);
};

  let index = 0;

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findLength = function (nums1, nums2) {
  const length1 = nums1.length,
    length2 = nums2.length;
  const dp = new Array(length1 + 1)
    .fill(0)
    .map(() => new Array(length2 + 1).fill(0));
  let resMax = 0;
  for (let i = 1; i <= length1; i++) {
    for (let j = 1; j <= length2; j++) {
      if (nums1[i - 1] === nums2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
        resMax = Math.max(resMax, dp[i][j]);
      }
    }
  }
  return resMax;
};

  for (let i = 1; i <= text1.length; i++) {
    for (let j = 1; j <= text2.length; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;;

/**

 */
var maxUncrossedLines = function (nums1, nums2) {
  // 两个数组长度
  const [m, n] = [nums1.length, nums2.length];
  // 创建dp数组并都初始化为0
  const dp = new Array(m + 1).fill(0).map(x => new Array(n + 1).fill(0));
  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      // 根据两种情况更新dp[i][j]
      if (nums1[i - 1] === nums2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  // 返回dp数组中右下角的元素
  return dp[m][n];
};

/**

 */
var maxSubArray = function (nums) {
  // 如果和小于0，就从下一个开始
  let res = Number.MIN_SAFE_INTEGER;
  let cur = 0;
  if (nums.length === 1) {
    return nums[0]
  }
  for (let i = 0; i < nums.length; i++) {
    if (cur < 0) {
      cur = 0;
    }
    cur += nums[i]
    res = Math.max(res, cur)
  }
  return res;
};

判断子序列

/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isSubsequence = function (s, t) {
  /* 
  转换为两个数组，相同的arrS移除首元素，最后判断arrS的长度
  */

  const arrS = s.split("");
  const arrT = t.split("");

  if (arrS.length > arrT.length) {
    return false;
  }

  while (arrS.length && arrT.length) {
    if (arrS[0] === arrT[0]) {
      arrS.shift();
    }
    arrT.shift();
  }
  // console.log(arrS.length === 0 ? true : false);
  return arrS.length === 0 ? true : false;
};

/**
 * @param {string} s
 * @param {string} t
 * @return {number}
 */
var numDistinct = function (s, t) {
  /* 
    不要求连续
    dp[i][j] 表示以i-1为结尾的s中有以j-1为结尾的t的个数为dp[i][j]
    */
  let dp = Array.from(Array(s.length + 1), () => Array(t.length + 1).fill(0));
  for (let i = 0; i <= s.length; i++) {
    dp[i][0] = 1;
  }
  for (let i = 1; i <= s.length; i++) {
    for (let j = 1; j <= t.length; j++) {
      if (s[i - 1] === t[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + dp[i - 1][j];
      } else {
        dp[i][j] = dp[i - 1][j];
      }
    }
  }
  return dp[s.length][t.length];
};

/**
 * @param {string} s
 * @return {number}
 */
var countSubstrings = function (s) {
  const strLen = s.length;
  let numOfPalindromicStr = 0;
  let dp = Array.from(Array(strLen), () => Array(strLen).fill(false));

  for (let j = 0; j < strLen; j++) {
    for (let i = 0; i <= j; i++) {
      if (s[i] === s[j]) {
        if ((j - i) < 2) {
          dp[i][j] = true;
        } else {
          dp[i][j] = dp[i + 1][j - 1];
        }
        numOfPalindromicStr += dp[i][j] ? 1 : 0;
      }
    }
  }
  return numOfPalindromicStr;
};

/**
 * @param {string} s
 * @return {number}
 */
var longestPalindromeSubseq = function (s) {
  /* 
    可以不连续
    */
  const strLen = s.length;
  let dp = Array.from(Array(strLen), () => Array(strLen).fill(0));
  for (let i = 0; i < strLen; i++) {
    dp[i][i] = 1;
  }
  for (let i = strLen - 1; i >= 0; i--) {
    for (let j = i + 1; j < strLen; j++) {
      if (s[i] === s[j]) {
        dp[i][j] = dp[i + 1][j - 1] + 2;
      } else {
        dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
      }
    }
  }
  return dp[0][strLen - 1];
};
