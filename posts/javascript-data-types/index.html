JavaScript数据类型全面解析：分类、区别与类型校验

发表于2023|更新于2026|前端基础
|字数总计:2.9k|阅读时长:14分钟

JavaScript 的数据类型是面试必考知识点，也是日常开发中容易踩坑的地方。本文将全面梳理 JS 数据类型，并给出最完整的类型校验方案。

一、JS 数据类型分类

JavaScript 共有 8 种数据类型，分为两大类：

1.1 原始类型（Primitive Types）- 7种

类型 说明 示例
Number 数字（整数和浮点数） 1, 3.14, NaN, Infinity
String 字符串 'hello', "world"
Boolean 布尔值 true, false
undefined 未定义 undefined
null 空值 null
Symbol 符号（ES6新增） Symbol('desc')
BigInt 大整数（ES2020新增） 9007199254740991n

1.2 引用类型（Reference Types）- 1种

类型 说明 示例
Object 对象（包括普通对象、数组、函数等） {}, [], function(){} 
// 原始类型示例
let num = 42;
let str = 'hello';
let bool = true;
let undef = undefined;
let nul = null;
let sym = Symbol('id');
let big = 123n;

// 引用类型示例
let obj = { name: '张三' };
let arr = [1, 2, 3];
let fn = function() {};
let date = new Date();
let reg = /abc/;

二、原始类型与引用类型的区别

2.1 存储方式不同

// 原始类型：存储在栈（Stack）中，直接存储值
let a = 10;
let b = a;  // 复制值
b = 20;
console.log(a);  // 10（互不影响）

// 引用类型：存储在堆（Heap）中，变量存储的是内存地址
let obj1 = { name: '张三' };
let obj2 = obj1;  // 复制引用地址
obj2.name = '李四';
console.log(obj1.name);  // '李四'（互相影响）

2.2 比较方式不同

// 原始类型：比较值
console.log(1 === 1);           // true
console.log('hello' === 'hello'); // true

// 引用类型：比较内存地址
console.log({} === {});         // false（不同对象，地址不同）
console.log([] === []);         // false

const obj = {};
const ref = obj;
console.log(obj === ref);       // true（同一引用）

2.3 可变性不同

// 原始类型：不可变（immutable）
let str = 'hello';
str[0] = 'H';  // 尝试修改
console.log(str);  // 'hello'（无法修改）

// 引用类型：可变（mutable）
let arr = [1, 2, 3];
arr[0] = 100;
console.log(arr);  // [100, 2, 3]

三、特殊的 null 和 undefined

3.1 两者的区别

// undefined：表示"缺少值"，即此处应该有值，但还没有定义
let a;
console.log(a);  // undefined

function foo(x) {
  console.log(x);  // undefined（未传参）
}
foo();

const obj = {};
console.log(obj.name);  // undefined（属性不存在）

// null：表示"没有对象"，即此处不应该有值
let b = null;  // 主动设置为 null

// 典型场景：释放内存
let data = { name: 'test' };
data = null;  // 解除引用，便于垃圾回收

3.2 typeof null 的”bug”

console.log(typeof null);  // 'object'

// 原因：这是 JS 的历史遗留问题
// 在 JS 最初版本中，值使用 32 位存储
// 000 开头表示对象，而 null 的二进制表示全是 0
// 所以 typeof null 误判为 'object'

3.3 正确区分 null 和 undefined

// 方法一：严格相等
console.log(null === null);        // true
console.log(undefined === undefined);  // true
console.log(null === undefined);   // false

// 方法二：使用 typeof
console.log(typeof null);          // 'object'
console.log(typeof undefined);     // 'undefined'

// 方法三：使用 Object.prototype.toString
console.log(Object.prototype.toString.call(null));      // '[object Null]'
console.log(Object.prototype.toString.call(undefined)); // '[object Undefined]'

四、类型校验方法

4.1 typeof 操作符

// typeof 返回值：'number', 'string', 'boolean', 'undefined', 'object', 'function', 'symbol', 'bigint'

console.log(typeof 42);           // 'number'
console.log(typeof 'hello');      // 'string'
console.log(typeof true);         // 'boolean'
console.log(typeof undefined);    // 'undefined'
console.log(typeof null);         // 'object' ❌ 陷阱！
console.log(typeof {});           // 'object'
console.log(typeof []);           // 'object' ❌ 无法区分数组
console.log(typeof function(){}); // 'function'
console.log(typeof Symbol());     // 'symbol'
console.log(typeof 123n);         // 'bigint'

// typeof 的局限性
// 1. null 返回 'object'
// 2. 无法区分对象、数组、正则等

4.2 instanceof 操作符

// instanceof：判断对象是否是某个构造函数的实例
console.log([] instanceof Array);    // true
console.log([] instanceof Object);   // true（数组也是对象）
console.log({} instanceof Object);   // true
console.log(new Date() instanceof Date);  // true

// instanceof 的局限性
// 1. 只能用于对象，原始类型返回 false
console.log(1 instanceof Number);     // false
console.log('hello' instanceof String); // false

// 2. 跨 iframe 或跨窗口时会失效
// const iframe = document.createElement('iframe');
// document.body.appendChild(iframe);
// const iframeArray = iframe.contentWindow.Array;
// console.log([] instanceof iframeArray);  // false

4.3 constructor 属性

console.log([].constructor === Array);   // true
console.log({}.constructor === Object);  // true
console.log((function(){}).constructor === Function); // true

// constructor 的局限性
// 1. 可以被修改
let arr = [];
arr.constructor = Object;
console.log(arr.constructor === Array);  // false

// 2. null 和 undefined 没有 constructor
// console.log(null.constructor);  // 报错

4.4 Object.prototype.toString.call()（最通用）

// 这是最准确的类型判断方法
const toString = Object.prototype.toString;

console.log(toString.call(1));           // '[object Number]'
console.log(toString.call('hello'));     // '[object String]'
console.log(toString.call(true));        // '[object Boolean]'
console.log(toString.call(undefined));   // '[object Undefined]'
console.log(toString.call(null));        // '[object Null]'
console.log(toString.call({}));          // '[object Object]'
console.log(toString.call([]));          // '[object Array]'
console.log(toString.call(function(){})); // '[object Function]'
console.log(toString.call(new Date()));  // '[object Date]'
console.log(toString.call(/abc/));       // '[object RegExp]'
console.log(toString.call(Symbol()));    // '[object Symbol]'
console.log(toString.call(123n));        // '[object BigInt]'
console.log(toString.call(new Map()));   // '[object Map]'
console.log(toString.call(new Set()));   // '[object Set]'

4.5 Array.isArray()

// 专门判断数组
console.log(Array.isArray([]));        // true
console.log(Array.isArray({}));        // false
console.log(Array.isArray(new Array())); // true

// 比 instanceof 更可靠（跨 iframe 也正常工作）

五、其他特殊值判断

5.1 判断 NaN

// NaN 不等于自身！
console.log(NaN === NaN);  // false

// 方法一：Number.isNaN()（ES6推荐）
console.log(Number.isNaN(NaN));     // true
console.log(Number.isNaN('hello')); // false

// 方法二：isNaN()（不推荐，会先转换）
console.log(isNaN(NaN));      // true
console.log(isNaN('hello'));  // true ❌ 字符串被转为 NaN
console.log(isNaN('123'));    // false

// 方法三：自身比较
console.log(NaN !== NaN);  // true（只有 NaN 满足这个条件）
function isNaN(value) {
  return value !== value;
}

5.2 判断数组

// 方法一：Array.isArray()（推荐）
console.log(Array.isArray([]));  // true

// 方法二：Object.prototype.toString
console.log(Object.prototype.toString.call([]) === '[object Array]');  // true

// 方法三：instanceof（不推荐，跨 iframe 有问题）
console.log([] instanceof Array);  // true

5.3 判断纯对象

function isPlainObject(obj) {
  if (typeof obj !== 'object' || obj === null) {
    return false;
  }
  
  // 排除 Array、Date、RegExp 等
  const proto = Object.getPrototypeOf(obj);
  return proto === null || proto === Object.prototype;
}

console.log(isPlainObject({}));           // true
console.log(isPlainObject(new Object())); // true
console.log(isPlainObject([]));           // false
console.log(isPlainObject(new Date()));   // false
console.log(isPlainObject(null));         // false

六、完整的类型判断函数

6.1 通用类型判断函数

function getType(value) {
  const toString = Object.prototype.toString;
  const type = toString.call(value);
  
  // 提取类型名称 '[object Xxx]' -> 'xxx'
  return type.replace(/^\[object (\w+)\]$/, '$1').toLowerCase();
}

// 测试
console.log(getType(1));            // 'number'
console.log(getType('hello'));      // 'string'
console.log(getType(true));         // 'boolean'
console.log(getType(undefined));    // 'undefined'
console.log(getType(null));         // 'null'
console.log(getType({}));           // 'object'
console.log(getType([]));           // 'array'
console.log(getType(function(){})); // 'function'
console.log(getType(new Date()));   // 'date'
console.log(getType(/abc/));        // 'regexp'
console.log(getType(Symbol()));     // 'symbol'
console.log(getType(123n));         // 'bigint'
console.log(getType(new Map()));    // 'map'
console.log(getType(new Set()));    // 'set'
console.log(getType(document));     // 'htmldocument'

6.2 完整的类型校验工具库

const Type = {
  // 原始类型
  isNumber: (val) => typeof val === 'number' && !isNaN(val),
  isString: (val) => typeof val === 'string',
  isBoolean: (val) => typeof val === 'boolean',
  isUndefined: (val) => val === undefined,
  isNull: (val) => val === null,
  isSymbol: (val) => typeof val === 'symbol',
  isBigInt: (val) => typeof val === 'bigint',
  
  // 引用类型
  isObject: (val) => val !== null && typeof val === 'object',
  isArray: (val) => Array.isArray(val),
  isFunction: (val) => typeof val === 'function',
  isDate: (val) => getType(val) === 'date',
  isRegExp: (val) => getType(val) === 'regexp',
  isMap: (val) => getType(val) === 'map',
  isSet: (val) => getType(val) === 'set',
  isPlainObject: (val) => getType(val) === 'object',
  
  // 特殊值
  isNaN: (val) => Number.isNaN(val),
  isEmpty: (val) => {
    if (val === null || val === undefined) return true;
    if (Type.isArray(val) || Type.isString(val)) return val.length === 0;
    if (Type.isObject(val)) return Object.keys(val).length === 0;
    return false;
  },
  
  // 获取类型
  getType: (val) => getType(val)
};

function getType(value) {
  return Object.prototype.toString
    .call(value)
    .replace(/^\[object (\w+)\]$/, '$1')
    .toLowerCase();
}

// 使用示例
console.log(Type.isNumber(42));       // true
console.log(Type.isArray([1, 2, 3])); // true
console.log(Type.isEmpty({}));        // true
console.log(Type.isEmpty(''));        // true
console.log(Type.getType(new Date())); // 'date'

6.3 生产环境可用的类型判断库

/**
 * 完整的类型判断工具
 * 支持所有 JS 类型，包括 ES6+ 新类型
 */
const TypeChecker = (function() {
  const toString = Object.prototype.toString;
  
  // 类型映射表
  const typeMap = {
    '[object Number]': 'number',
    '[object String]': 'string',
    '[object Boolean]': 'boolean',
    '[object Undefined]': 'undefined',
    '[object Null]': 'null',
    '[object Object]': 'object',
    '[object Array]': 'array',
    '[object Function]': 'function',
    '[object Date]': 'date',
    '[object RegExp]': 'regexp',
    '[object Error]': 'error',
    '[object Symbol]': 'symbol',
    '[object BigInt]': 'bigint',
    '[object Map]': 'map',
    '[object Set]': 'set',
    '[object WeakMap]': 'weakmap',
    '[object WeakSet]': 'weakset',
    '[object Promise]': 'promise',
    '[object Generator]': 'generator',
    '[object GeneratorFunction]': 'generatorfunction',
    '[object ArrayBuffer]': 'arraybuffer',
    '[object DataView]': 'dataview',
    '[object Int8Array]': 'int8array',
    '[object Uint8Array]': 'uint8array',
    '[object Float32Array]': 'float32array'
  };
  
  return {
    // 获取类型
    getType(value) {
      const type = toString.call(value);
      return typeMap[type] || 'unknown';
    },
    
    // 原始类型判断
    isNumber(value) {
      return typeof value === 'number' && !Number.isNaN(value);
    },
    
    isString(value) {
      return typeof value === 'string';
    },
    
    isBoolean(value) {
      return typeof value === 'boolean';
    },
    
    isUndefined(value) {
      return value === undefined;
    },
    
    isNull(value) {
      return value === null;
    },
    
    isSymbol(value) {
      return typeof value === 'symbol';
    },
    
    isBigInt(value) {
      return typeof value === 'bigint';
    },
    
    // 引用类型判断
    isObject(value) {
      return value !== null && typeof value === 'object';
    },
    
    isArray(value) {
      return Array.isArray(value);
    },
    
    isFunction(value) {
      return typeof value === 'function';
    },
    
    isDate(value) {
      return this.getType(value) === 'date';
    },
    
    isRegExp(value) {
      return this.getType(value) === 'regexp';
    },
    
    isPlainObject(value) {
      if (!this.isObject(value)) return false;
      const proto = Object.getPrototypeOf(value);
      return proto === null || proto === Object.prototype;
    },
    
    // 特殊判断
    isNaN(value) {
      return Number.isNaN(value);
    },
    
    isEmpty(value) {
      if (value === null || value === undefined) return true;
      if (this.isArray(value) || this.isString(value)) return value.length === 0;
      if (this.isMap(value) || this.isSet(value)) return value.size === 0;
      if (this.isObject(value)) return Object.keys(value).length === 0;
      return false;
    },
    
    isMap(value) {
      return this.getType(value) === 'map';
    },
    
    isSet(value) {
      return this.getType(value) === 'set';
    },
    
    isPromise(value) {
      return this.getType(value) === 'promise' || 
             (this.isObject(value) && this.isFunction(value.then));
    },
    
    // 判断是否为原始类型
    isPrimitive(value) {
      return value === null || 
             (typeof value !== 'object' && typeof value !== 'function');
    },
    
    // 判断是否为可迭代对象
    isIterable(value) {
      return value !== null && 
             value !== undefined && 
             typeof value[Symbol.iterator] === 'function';
    }
  };
})();

// 导出（Node.js 环境）
if (typeof module !== 'undefined' && module.exports) {
  module.exports = TypeChecker;
}

七、总结

7.1 类型判断方法对比

方法 优点 缺点 适用场景
typeof 简单方便 null 返回 object，无法区分引用类型 原始类型判断
instanceof 可判断实例类型 跨 iframe 失效，只能用于对象 对象类型判断
constructor 简单直观 可被修改，null/undefined 无此属性 一般类型判断
Object.prototype.toString 最准确，支持所有类型 写法稍长 通用类型判断
Array.isArray 可靠判断数组 只能判断数组 数组判断
Number.isNaN 准确判断 NaN ES6+ NaN 判断

7.2 最佳实践

  1. 原始类型判断：使用 typeof
  2. 数组判断：使用 Array.isArray()
  3. NaN 判断：使用 Number.isNaN()
  4. 通用类型判断：使用 Object.prototype.toString.call()
  5. null 判断：使用 === null
  6. undefined 判断：使用 === undefined 或 typeof

7.3 面试高频问题

  1. typeof null 为什么返回 ‘object’？

    • JS 最初版本中，值用 32 位存储，低位存储类型标志
    • 对象类型以 000 开头，而 null 全是 0，被误判

  2. null 和 undefined 的区别？

    • null：表示”应该有值，但现在是空的”，主动设置
    • undefined：表示”缺少值”，系统默认

  3. 如何判断一个变量是数组？

    • Array.isArray()（推荐）
    • Object.prototype.toString.call(arr) === '[object Array]'
    • arr instanceof Array（不推荐）

  4. 原始类型和引用类型的区别？

    • 存储位置：栈 vs 堆
    • 复制方式：值复制 vs 引用复制
    • 比较方式：值比较 vs 引用比较
    • 可变性：不可变 vs 可变
