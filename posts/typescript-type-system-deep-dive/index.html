TypeScript类型系统深度解析：any、unknown、never与协变逆变

发表于2025|更新于2026|前端基础
|字数总计:5.2k|阅读时长:24分钟

前言

TypeScript 的类型系统是其核心特性，理解类型系统的工作原理对于编写高质量的 TypeScript 代码至关重要。本文将深入探讨 TypeScript 中的特殊类型（any、unknown、never）、类型系统的核心概念（协变与逆变），以及常见场景的类型标注方法，帮助你建立完整的类型系统知识体系。

一、any 类型：双刃剑

1.1 any 类型的本质

any 类型是 TypeScript 中的顶级类型（Top Type），它可以表示任何类型。使用 any 相当于告诉 TypeScript 编译器”关闭类型检查”。

let anything: any = 'hello';

anything = 42;          // ✓ 可以赋值为 number
anything = true;        // ✓ 可以赋值为 boolean
anything = { name: '张三' }; // ✓ 可以赋值为对象
anything = null;        // ✓ 可以赋值为 null
anything = undefined;   // ✓ 可以赋值为 undefined

// ⚠️ 危险：可以访问任何属性，不会报错
anything.foo.bar.baz;   // ✓ 编译通过，运行时可能报错

// ⚠️ 危险：可以调用任何方法
anything();             // ✓ 编译通过，运行时可能报错
anything.customMethod(); // ✓ 编译通过，运行时可能报错

1.2 any 的隐式推断

TypeScript 会在某些情况下隐式推断为 any 类型：

// 1. 未声明类型且无法推断
let implicitAny;  // 隐式 any

// 2. 函数参数未声明类型
function greet(name) {  // name 隐式 any
  return `Hello, ${name}`;
}

// 3. 对象属性未声明类型
const user = {
  name: '张三',  // 推断为 string
  age: 25,       // 推断为 number
  data           // 隐式 any（未初始化）
};

最佳实践： 在 tsconfig.json 中启用严格模式，禁止隐式 any：

{
  "compilerOptions": {
    "strict": true,
    "noImplicitAny": true
  }
}

1.3 any 的合理使用场景

虽然 any 可能导致类型安全问题，但在以下场景中合理使用：

1.3.1 渐进式迁移

// 从 JavaScript 迁移到 TypeScript 时
// 可以先用 any 标记未迁移的部分
function legacyCode(data: any) {
  // 旧代码，暂时保留 any
  return data.process();
}

1.3.2 第三方库缺少类型定义

// 第三方库没有类型定义
import someLibrary from 'some-untyped-library';

const result: any = someLibrary.doSomething();

1.3.3 类型断言和类型守卫

// 在类型断言中使用
const parsed: any = JSON.parse(jsonString);

// 配合类型守卫
function isString(value: any) {
  return typeof value === 'string';
}

1.4 any 的危险陷阱

陷阱 1：类型污染

let anyValue: any = 'hello';

// any 类型会污染其他变量
let stringType: string = anyValue;  // ✓ 编译通过，但运行时可能有问题
stringType.toUpperCase();            // ✓ 编译通过

// 修改 anyValue 的值
anyValue = 42;

// stringType 现在实际上是 number，但 TypeScript 不知道
console.log(stringType);  // 42
// stringType.toUpperCase(); // 运行时错误：toUpperCase is not a function

陷阱 2：丢失类型信息

interface User {
  name: string;
  age: number;
}

function processUser(user: User) {
  console.log(`${user.name} is ${user.age} years old`);
}

const anyUser: any = {
  name: '张三'
  // 缺少 age 属性
};

processUser(anyUser);  // ✓ 编译通过，但运行时 age 是 undefined

二、unknown 类型：安全的 any

2.1 unknown 的本质

unknown 类型是 TypeScript 3.0 引入的类型，它是类型安全的 any。unknown 可以表示任何值，但在使用前必须进行类型检查。

let unknownValue: unknown = 'hello';

unknownValue = 42;          // ✓ 可以赋值
unknownValue = true;        // ✓ 可以赋值
unknownValue = { name: '张三' }; // ✓ 可以赋值

// ✗ 错误：不能直接访问属性
// unknownValue.foo;

// ✗ 错误：不能直接调用方法
// unknownValue();

// ✗ 错误：不能赋值给其他类型
// let str: string = unknownValue;

2.2 unknown vs any

特性 any unknown
可以赋值任何类型 ✓ ✓
可以赋值给其他类型 ✓ ✗ (需类型断言或守卫)
可以访问任意属性 ✓ ✗
可以调用任意方法 ✓ ✗
类型安全 ✗ ✓ 
let anyValue: any = 'hello';
let unknownValue: unknown = 'hello';

// any: 不安全
anyValue.toUpperCase();  // ✓ 编译通过
let str1: string = anyValue;  // ✓ 编译通过

// unknown: 安全
// unknownValue.toUpperCase();  // ✗ 编译错误
// let str2: string = unknownValue;  // ✗ 编译错误

// 必须先进行类型检查
if (typeof unknownValue === 'string') {
  unknownValue.toUpperCase();  // ✓ 编译通过
  let str2: string = unknownValue;  // ✓ 编译通过
}

2.3 unknown 的正确使用方式

2.3.1 类型守卫

function processUnknown(value: unknown) {
  // 类型守卫：typeof
  if (typeof value === 'string') {
    return value.toUpperCase();  // ✓ value 是 string 类型
  }
  
  if (typeof value === 'number') {
    return value.toFixed(2);  // ✓ value 是 number 类型
  }
  
  // 类型守卫：instanceof
  if (value instanceof Date) {
    return value.toISOString();  // ✓ value 是 Date 类型
  }
  
  // 类型守卫：Array.isArray
  if (Array.isArray(value)) {
    return value.length;  // ✓ value 是 any[] 类型
  }
  
  // 自定义类型守卫
  if (isObject(value)) {
    return JSON.stringify(value);  // ✓ value 是 object 类型
  }
  
  return String(value);
}

// 自定义类型守卫
function isObject(value: unknown): value is object {
  return typeof value === 'object' && value !== null;
}

2.3.2 类型断言

let value: unknown = 'hello';

// 使用类型断言（谨慎使用）
let strLength = (value as string).length;

// 或者使用双重断言（更不安全，避免使用）
let num = value as unknown as number;

2.3.3 用户定义的类型守卫

interface User {
  id: number;
  name: string;
  email: string;
}

// 类型谓词：value is User
function isUser(value: unknown): value is User {
  return (
    typeof value === 'object' &&
    value !== null &&
    'id' in value &&
    'name' in value &&
    'email' in value &&
    typeof (value as User).id === 'number' &&
    typeof (value as User).name === 'string' &&
    typeof (value as User).email === 'string'
  );
}

function processUser(value: unknown) {
  if (isUser(value)) {
    // ✓ 在这个分支中，value 的类型是 User
    console.log(`User: ${value.name} (${value.email})`);
  } else {
    console.log('Invalid user data');
  }
}

2.4 unknown 的最佳实践

2.4.1 API 响应类型

async function fetchUser(id: number): Promise<unknown> {
  const response = await fetch(`/api/users/${id}`);
  return response.json();
}

async function handleUser(id: number) {
  const userData = await fetchUser(id);
  
  // 使用类型守卫验证数据
  if (isUser(userData)) {
    // ✓ 类型安全的 User 对象
    console.log(userData.name);
  } else {
    console.error('Invalid user data format');
  }
}

2.4.2 类型安全的 JSON 解析

function safeJSONParse(json: string): unknown {
  try {
    return JSON.parse(json);
  } catch {
    return null;
  }
}

const data = safeJSONParse('{"name": "张三", "age": 25}');

// 验证数据格式
if (isObject(data) && 'name' in data && 'age' in data) {
  console.log(data.name, data.age);
}

三、never 类型：不可能的类型

3.1 never 的本质

never 类型表示永不存在的值类型。它是 TypeScript 中的底部类型（Bottom Type），是所有类型的子类型。

// never 是所有类型的子类型
let neverValue: never;

// ✗ 错误：never 不能赋值任何值（除了 never 本身）
// neverValue = 'hello';  // 错误
// neverValue = 42;       // 错误
// neverValue = null;     // 错误
// neverValue = undefined; // 错误

3.2 never 的应用场景

3.2.1 永不返回的函数

// 1. 抛出错误的函数
function throwError(message: string): never {
  throw new Error(message);
}

// 2. 无限循环
function infiniteLoop(): never {
  while (true) {
    // 永不退出
  }
}

// 3. 死循环（类型推断）
function runForever(): never {
  while (true) {
    console.log('Running...');
  }
}

3.2.2 类型收窄和穷尽性检查

type Shape = 'circle' | 'square' | 'triangle';

function getArea(shape: Shape): number {
  switch (shape) {
    case 'circle':
      return Math.PI * 1 * 1;
    case 'square':
      return 1 * 1;
    case 'triangle':
      return 0.5 * 1 * 1;
    default:
      // 如果所有情况都已处理，shape 类型为 never
      const _exhaustiveCheck: never = shape;
      return _exhaustiveCheck;
  }
}

// 如果未来添加新的形状类型，但没有处理
// TypeScript 会在编译时报错，提醒你处理新类型

穷尽性检查工具函数：

function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`);
}

type Action = 
  | { type: 'INCREMENT' }
  | { type: 'DECREMENT' }
  | { type: 'RESET' };

function reducer(state: number, action: Action): number {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
    case 'RESET':
      return 0;
    default:
      return assertNever(action);  // 如果遗漏 case，编译错误
  }
}

3.2.3 条件类型中的过滤

// 从联合类型中移除 null 和 undefined
type NonNullable<T> = T extends null | undefined ? never : T;

type MaybeString = string | null | undefined;
type DefiniteString = NonNullable<MaybeString>; // string

// 从联合类型中移除特定类型
type Exclude<T, U> = T extends U ? never : T;

type EventNames = 'click' | 'scroll' | 'keypress' | 'mousemove';
type MouseEvents = Exclude<EventNames, 'keypress'>; // 'click' | 'scroll' | 'mousemove'

3.2.4 不可达代码标记

function process(value: string | number) {
  if (typeof value === 'string') {
    // value 是 string
    return value.toUpperCase();
  } else if (typeof value === 'number') {
    // value 是 number
    return value.toFixed(2);
  } else {
    // 这里 value 是 never，因为所有情况都已处理
    const _exhaustiveCheck: never = value;
    throw new Error('Unreachable code');
  }
}

3.3 never vs void

特性 void never
含义 没有返回值 永不返回
可以返回 ✓ 可以 return ✗ 不能 return
函数是否结束 ✓ 正常结束 ✗ 不会结束
作为类型 表示无返回值 表示不可能存在 
// void: 函数正常结束，但没有返回值
function log(message: string): void {
  console.log(message);
  // 正常结束，返回 undefined
}

// never: 函数永不结束
function fail(message: string): never {
  throw new Error(message);
  // 永不返回，因为抛出错误
}

四、协变与逆变

4.1 什么是协变与逆变？

协变和逆变描述了复合类型（如函数、数组、对象）之间如何根据其组成类型的关系而变化。

4.1.1 类型兼容性

// 基本类型关系
type Animal = { name: string };
type Dog = { name: string; breed: string };

// Dog 是 Animal 的子类型
const dog: Dog = { name: '旺财', breed: '金毛' };
const animal: Animal = dog;  // ✓ 子类型可以赋值给父类型

4.2 协变（Covariance）

协变：如果 Dog extends Animal，那么 Dog[] extends Animal[]。类型关系方向一致。

4.2.1 数组的协变

type Animal = { name: string };
type Dog = { name: string; breed: string };

// 数组是协变的
let dogs: Dog[] = [{ name: '旺财', breed: '金毛' }];
let animals: Animal[] = dogs;  // ✓ 协变：Dog[] 可以赋值给 Animal[]

// 但这可能导致类型安全问题
animals.push({ name: '小猫' });  // ✓ 编译通过（但不安全）
// dogs 现在包含一个不是 Dog 的对象

4.2.2 Promise 的协变

async function getAnimal(): Promise<Animal> {
  return { name: '动物' };
}

async function getDog(): Promise<Dog> {
  return { name: '旺财', breed: '金毛' };
}

// Promise 是协变的
let animalPromise: Promise<Animal> = getDog();  // ✓ 协变

4.2.3 对象属性的协变

interface AnimalShelter {
  animals: Animal[];
}

interface DogShelter {
  animals: Dog[];
}

let dogShelter: DogShelter = {
  animals: [{ name: '旺财', breed: '金毛' }]
};

// ✓ 对象属性是协变的
let animalShelter: AnimalShelter = dogShelter;

4.3 逆变（Contravariance）

逆变：如果 Dog extends Animal，那么 (animal: Animal) => void extends (dog: Dog) => void。类型关系方向相反。

4.3.1 函数参数的逆变

type Animal = { name: string };
type Dog = { name: string; breed: string };

// 函数参数是逆变的
let handleAnimal = (animal: Animal) => {
  console.log(`Animal: ${animal.name}`);
};

let handleDog = (dog: Dog) => {
  console.log(`Dog: ${dog.name}, breed: ${dog.breed}`);
};

// ✓ 逆变：可以赋值
let dogHandler: (dog: Dog) => void = handleAnimal;
// 因为 handleAnimal 可以处理 Animal，当然可以处理 Dog（Dog 是 Animal 的子类型）

// ✗ 错误：不能赋值
// let animalHandler: (animal: Animal) => void = handleDog;
// 因为 handleDog 需要 Dog 的 breed 属性，Animal 可能没有

为什么函数参数是逆变的？

type Animal = { name: string };
type Dog = { name: string; breed: string };

// 假设函数参数是协变的（实际上不是）
let printDog = (dog: Dog) => {
  console.log(dog.breed);  // 需要 breed 属性
};

let printAnimal: (animal: Animal) => void = printDog;  // 如果允许协变

const animal: Animal = { name: '小猫' };  // 没有 breed 属性
printAnimal(animal);  // 运行时错误：breed 不存在

4.3.2 函数返回值的协变

// 函数返回值是协变的
let createAnimal = (): Animal => ({ name: '动物' });
let createDog = (): Dog => ({ name: '旺财', breed: '金毛' });

// ✓ 返回值协变：Dog -> Animal
let animalFactory: () => Animal = createDog;

// ✗ 错误：返回值不能逆变
// let dogFactory: () => Dog = createAnimal;

类型层级：
Animal (父类型)
↑
|
Dog (子类型)

协变（方向一致）：
Animal[] (父类型)
↑
|
Dog[] (子类型)

逆变（方向相反）：
(dog: Dog) => void (父类型)
↑
|
(animal: Animal) => void (子类型)

五、常见类型标注

5.1 Class 类型标注

5.1.1 基本类定义

class User {
  // 属性类型标注
  name: string;
  age: number;
  private email: string;
  protected createdAt: Date;
  
  // 静态属性
  static species = '人类';
  
  // 构造函数
  constructor(name: string, age: number, email: string) {
    this.name = name;
    this.age = age;
    this.email = email;
    this.createdAt = new Date();
  }
  
  // 方法返回值类型标注
  greet(): string {
    return `Hello, I'm ${this.name}`;
  }
  
  // 异步方法
  async save(): Promise<void> {
    await fetch('/api/users', {
      method: 'POST',
      body: JSON.stringify(this)
    });
  }
}

5.1.2 类的类型（构造函数类型）

// 类的类型包含两部分：
// 1. 实例类型（使用 typeof instance 或 User）
// 2. 构造函数类型（使用 typeof ClassName）

class User {
  name: string;
  
  constructor(name: string) {
    this.name = name;
  }
  
  greet(): string {
    return `Hello, ${this.name}`;
  }
}

// 实例类型
type UserInstance = User;

// 构造函数类型
type UserConstructor = typeof User;

// 使用构造函数类型
function createUser(ctor: UserConstructor, name: string): UserInstance {
  return new ctor(name);
}

const user = createUser(User, '张三');

5.1.3 抽象类

abstract class Animal {
  name: string;
  
  constructor(name: string) {
    this.name = name;
  }
  
  // 抽象方法：必须由子类实现
  abstract makeSound(): void;
  
  // 具体方法：子类可以直接继承
  move(): void {
    console.log(`${this.name} is moving`);
  }
}

class Dog extends Animal {
  breed: string;
  
  constructor(name: string, breed: string) {
    super(name);
    this.breed = breed;
  }
  
  // 实现抽象方法
  makeSound(): void {
    console.log('汪汪汪！');
  }
}

// ✗ 错误：不能实例化抽象类
// const animal = new Animal('动物');

const dog = new Dog('旺财', '金毛');
dog.makeSound();  // 汪汪汪！
dog.move();       // 旺财 is moving

5.1.4 类实现接口

interface Serializable {
  serialize(): string;
}

interface Comparable {
  compareTo(other: this): number;
}

class User implements Serializable, Comparable {
  constructor(
    public name: string,
    public age: number
  ) {}
  
  serialize(): string {
    return JSON.stringify({ name: this.name, age: this.age });
  }
  
  compareTo(other: User): number {
    return this.age - other.age;
  }
}

5.1.5 泛型类

class Container<T> {
  private value: T;
  
  constructor(value: T) {
    this.value = value;
  }
  
  getValue(): T {
    return this.value;
  }
  
  setValue(value: T): void {
    this.value = value;
  }
  
  // 泛型方法
  map<U>(fn: (value: T) => U): Container<U> {
    return new Container(fn(this.value));
  }
}

const numberContainer = new Container<number>(42);
const stringContainer = numberContainer.map(n => n.toString());

5.2 Symbol 类型标注

5.2.1 基本使用

// Symbol 类型标注
const sym1: symbol = Symbol('description');
const sym2: symbol = Symbol('description');

console.log(sym1 === sym2);  // false，每个 Symbol 都是唯一的

// Symbol 作为对象属性键
const user = {
  name: '张三',
  [Symbol('id')]: 12345
};

5.2.2 Symbol 的类型

// 1. symbol 类型：所有 Symbol 的基类型
const sym: symbol = Symbol('test');

// 2. unique symbol 类型：特定的唯一 Symbol
const uniqueSym: unique symbol = Symbol('unique');
const anotherSym: typeof uniqueSym = uniqueSym;  // ✓ 类型相同

// unique symbol 的限制
const sym1: unique symbol = Symbol('a');
const sym2: unique symbol = Symbol('b');
// sym1 = sym2;  // ✗ 错误：不能赋值

5.2.3 Symbol 作为属性键

// 使用 Symbol 作为对象属性键
const idKey = Symbol('id');
const nameKey = Symbol('name');

interface User {
  [idKey]: number;  // Symbol 属性
  [nameKey]: string;
  email: string;    // 普通字符串属性
}

const user: User = {
  [idKey]: 1,
  [nameKey]: '张三',
  email: 'zhangsan@example.com'
};

console.log(user[idKey]);    // 1
console.log(user[nameKey]);  // 张三
console.log(user.email);     // zhangsan@example.com

// Symbol 属性不会被 Object.keys 等方法枚举
console.log(Object.keys(user));  // ['email']

5.2.4 内置 Symbol

TypeScript 提供了内置的 Symbol 值类型：

// Symbol.iterator
interface Iterable<T> {
  [Symbol.iterator](): Iterator<T>;
}

class NumberRange implements Iterable<number> {
  constructor(
    private start: number,
    private end: number
  ) {}
  
  *[Symbol.iterator](): Iterator<number> {
    for (let i = this.start; i <= this.end; i++) {
      yield i;
    }
  }
}

const range = new NumberRange(1, 5);
for (const num of range) {
  console.log(num);  // 1, 2, 3, 4, 5
}

// Symbol.toStringTag
class User {
  [Symbol.toStringTag] = 'User';
  
  constructor(public name: string) {}
}

const user = new User('张三');
console.log(Object.prototype.toString.call(user));  // '[object User]'

// Symbol.hasInstance
class ArrayLike {
  static [Symbol.hasInstance](instance: unknown): boolean {
    return Array.isArray(instance);
  }
}

console.log([] instanceof ArrayLike);  // true
console.log({} instanceof ArrayLike);  // false

5.2.5 Symbol 的实际应用

// 私有属性模拟
const privateKey = Symbol('private');

class Secret {
  constructor(private secret: string) {}
  
  // 通过 Symbol 提供受限访问
  [privateKey]() {
    return this.secret;
  }
  
  revealSecret(token: symbol): string | null {
    if (token === privateKey) {
      return this.secret;
    }
    return null;
  }
}

const secret = new Secret('密码123');
console.log(secret.revealSecret(privateKey));  // 密码123

// 元编程：修改对象行为
const customString = {
  [Symbol.toPrimitive](hint: 'string' | 'number' | 'default') {
    if (hint === 'string') {
      return '自定义字符串';
    }
    if (hint === 'number') {
      return 42;
    }
    return '默认值';
  }
};

console.log(String(customString));  // 自定义字符串
console.log(Number(customString));  // 42
console.log(customString + '');     // 默认值

六、类型安全最佳实践

6.1 避免使用 any

// ❌ 不好的做法
function process(data: any) {
  return data.value;
}

// ✓ 好的做法：使用 unknown
function processSafe(data: unknown) {
  if (typeof data === 'object' && data !== null && 'value' in data) {
    return data.value;
  }
  throw new Error('Invalid data format');
}

6.2 使用类型守卫

// 类型守卫函数
function isString(value: unknown): value is string {
  return typeof value === 'string';
}

function isNumber(value: unknown): value is number {
  return typeof value === 'number';
}

function processValue(value: unknown) {
  if (isString(value)) {
    return value.toUpperCase();  // ✓ 类型安全
  }
  if (isNumber(value)) {
    return value.toFixed(2);     // ✓ 类型安全
  }
  return String(value);
}

6.3 使用严格模式配置

{
  "compilerOptions": {
    "strict": true,
    "noImplicitAny": true,
    "strictNullChecks": true,
    "strictFunctionTypes": true,
    "strictBindCallApply": true,
    "strictPropertyInitialization": true,

  }

6.4 优先使用 unknown 而非 any

// ❌ 使用 any
function parse(json: string): any {
  return JSON.parse(json);
}

// ✓ 使用 unknown
function parseSafe(json: string): unknown {
  return JSON.parse(json);
}

// 使用时必须进行类型检查
const data = parseSafe('{"name": "张三"}');
if (isObject(data) && 'name' in data) {
  console.log(data.name);
}

6.5 使用类型推断

// ✓ 让 TypeScript 自动推断类型
const numbers = [1, 2, 3];  // 类型推断为 number[]
const sum = numbers.reduce((acc, num) => acc + num, 0);  // 类型推断为 number

// ✗ 不必要的显式类型标注
const numbers2: number[] = [1, 2, 3];
const sum2: number = numbers2.reduce((acc: number, num: number) => acc + num, 0);

七、总结

7.1 核心要点

  1. any 类型：关闭类型检查的双刃剑，应尽量避免使用
  2. unknown 类型：类型安全的 any，使用前必须进行类型检查
  3. never 类型：表示永不存在的值，用于不可达代码和穷尽性检查
  4. 协变：类型关系方向一致，如数组、Promise、函数返回值
  5. 逆变：类型关系方向相反，如函数参数

7.2 类型选择建议

场景 推荐类型
不确定类型但需要类型安全 unknown
不可能的值或永不出现在运行时 never
函数返回类型（无返回值） void
第三方库无类型定义 unknown + 类型守卫
JSON 解析结果 unknown + 类型验证

7.3 最佳实践总结

  1. 启用严格模式，禁止隐式 any
  2. 优先使用 unknown 而非 any
  3. 使用类型守卫和类型断言确保类型安全
  4. 利用 never 进行穷尽性检查
  5. 理解协变和逆变，正确处理函数类型兼容性
  6. 为 class 和 symbol 等提供正确的类型标注
