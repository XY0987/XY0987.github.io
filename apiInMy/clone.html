// 类型汇总
const mapTag = "[object Map]";
const setTag = "[object Set]";
const arrayTag = "[object Array]";
const objectTag = "[object Object]";
const argsTag = "[object Arguments]";
const boolTag = "[object Boolean]";
const dateTag = "[object Date]";
const numberTag = "[object Number]";
const stringTag = "[object String]";
const symbolTag = "[object Symbol]";
const errorTag = "[object Error]";
const regexpTag = "[object RegExp]";
const funcTag = "[object Function]";

const deepTag = [mapTag, setTag, arrayTag, objectTag, argsTag];

function getType(target) {
  return Object.prototype.toString.call(target)
}

function isObject(target) {
  const type = typeof target;
  return target !== null && (type === "object" || type === "function");
}

function initPrototype(target) {
  const Ctor = target.constructor;
  return new Ctor();
}

function cloneReg(target) {
  const reFlags = /\w*$/;
  const result = new target.constructor(target.source, reFlags.exec(target));
  result.lastIndex = target.lastIndex;
  return result;
}

function cloneSymbol(target) {
  return Object(Symbol.prototype.valueOf.call(target));
}

function cloneFunction(func) {
  // 匹配字符串，获取参数
  const bodyReg = /(?<={)(.|\n)+(?=})/m;
  const paramReg = /(?<=\().+(?=\)\s+{)/;
  const funcString = func.toString();
  // 判断是否是箭头函数，箭头函数没有原型
  if (func.prototype) {
    const param = paramReg.exec(funcString);
    const body = bodyReg.exec(funcString);
    if (body) {
      if (param) {
        const paramArr = param[0].split(",");
        return new Function(...paramArr, body[0]);
      } else {
        return new Function(body[0]);
      }
    } else {
      return null;
    }
  } else {
    return eval(funcString);
  }
}

function cloneOtherType(target, type) {
  const Ctor = target.constructor;
  switch (type) {
    case boolTag:
    case numberTag:
    case stringTag:
    case errorTag:
    case dateTag:
      return new Ctor(target);
    case regexpTag:
      return cloneReg(target);
    case symbolTag:
      return cloneSymbol(target);
    case funcTag:
      return cloneFunction(target);
    default:
      return null;
  }
}

function forAllEl(arr, callback) {
  let index = -1;
  const length = arr.length;
  while (++index < length) {
    callback(arr[index], index);
  }
  return arr;
}

function clone(target, map = new WeakMap()) {
  // 原始数据类型
  if (!isObject(target)) {
    return target
  }
  // 判断数据类型
  const type = getType(target)
  let cloneObj = null;
  // 是否是deepTag的类型(需要处理原型)
  if (deepTag.includes(type)) {
    cloneObj = initPrototype(target)
  } else {
    // clone其他类型
    return cloneOtherType(target, type)
  }
  // 查看是否clone过
  if (map.get(target)) {
    return map.get(target)
  }
  map.set(target, cloneObj)
  // clone set
  if (type === setTag) {
    // 遍历set的所有元素
    target.forEach((value) => {
      cloneObj.add(clone(value, map));
    });
    return cloneObj;
  }
  // clone map
  if (type === mapTag) {
    // 遍历map所有元素
    target.forEach((value, key) => {
      cloneObj.set(key, clone(value, map));
    });
    return cloneObj;
  }

  // 克隆对象和数组
  const keys = type === arrayTag ? undefined : Object.keys(target);// keys存在时表示对象的键值数组(不包括symbol属性)
  // 回调函数的第一个参数是值，第二个是下标
  forAllEl(keys || target, (value, key) => {
    // keys存在表示是对象
    if (keys) {
      key = value;
    }
    cloneObj[key] = clone(target[key], map);
  });
  const symbolKeys =
    type === objectTag ? Object.getOwnPropertySymbols(target) : undefined;
  forAllEl(symbolKeys, (value) => {
    cloneObj[value] = clone(target[value], map);
  });
  return cloneObj;
}

const sym = Symbol(111);
const obj = {
  [sym]: {},
  foo: function () {
    //函数不能使用简写
    console.log("1111");
  },
  baz: Symbol(1111122112),
};
const cloneObj = clone(obj);
console.log(cloneObj);
