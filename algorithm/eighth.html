打家劫舍

相邻房间不能偷

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
  /* 
    dp[i] 表示下标i之前的包含下标i 偷到的最大钱数是dp[i]
    
    两种状态，偷i和不偷i
    偷i最大值为dp[i-2]+nums[i]
    不偷最大值为 dp[i-1]
    */

  if (nums.length <= 2) {
    return Math.max(...nums);
  }

  const dp = new Array(nums.length).fill(0);
  dp[0] = nums[0];
  dp[1] = Math.max(nums[0], nums[1]);
  for (let i = 2; i < nums.length; i++) {
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
  }
  return dp[nums.length - 1]
};

打家劫舍二

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
  /* 
首尾相邻，不能偷相邻元素
三种情况：
1. 不考虑首和尾
2. 不考虑尾部
3. 不考虑首部

情况2和情况3包含情况1

dp[i] 表示偷下标i之前包含下标i所能偷到的最大金额
*/

  if (nums.length <= 3) {
    return Math.max(...nums);
  }

  const dp1 = new Array(nums.length - 1).fill(0);
  // 不考虑尾元素
  dp1[0] = nums[0];
  dp1[1] = Math.max(nums[0], nums[1]);
  for (let i = 2; i < nums.length - 1; i++) {
    dp1[i] = Math.max(dp1[i - 1], dp1[i - 2] + nums[i]);
  }
  // 不考虑首元素
  const dp2 = new Array(nums.length - 1).fill(0);
  dp2[0] = nums[1];
  dp2[1] = Math.max(nums[1], nums[2]);
  // 循环牵涉着dp2数组，所以不能变，变一下nums数组的取值
  for (let i = 2; i < nums.length - 1; i++) {
    dp2[i] = Math.max(dp2[i - 1], dp2[i - 2] + nums[i + 1]);
  }

  return Math.max(dp1[dp1.length - 1], dp2[dp2.length - 1]);
};

打家劫舍三

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var rob = function (root) {
    /* 
    dp数组是一维数组，只有两个元素，分别表示偷该节点和不偷该节点的最大值
    dp[0] 表示不偷
    dp[1] 表示偷

    后序遍历
    */
  
  // 后序遍历函数
  const postOrder = (node) => {
    // 递归出口
    if (!node) return [0, 0];
    // 遍历左子树
    const left = postOrder(node.left);
    // 遍历右子树
    const right = postOrder(node.right);
    // 不偷当前节点，左右子节点都可以偷或不偷，取最大值
    const DoNot = Math.max(left[0], left[1]) + Math.max(right[0], right[1]);
    // 偷当前节点，左右子节点只能不偷
    const Do = node.val + left[0] + right[0];
    // [不偷，偷]
    return [DoNot, Do];
  };
  const res = postOrder(root);
  // 返回最大值
  return Math.max(...res);
};

买卖股票的最佳时机一

右边的最大值减去左边的最小值
暴力循环解法

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
  持有不代表这一天买该股票
    dp[i][0] 表示持有该股票获得的最大金钱
    dp[i][1] 表示不持有该股票获得的最大金钱
     不持有不代表这一天卖出该股票
    */
  const len = prices.length;
  // 创建dp数组
  const dp = new Array(len).fill([0, 0]);
  // dp数组初始化
  dp[0] = [-prices[0], 0];
  for (let i = 1; i < len; i++) {
    // 更新dp[i]
    /*
    dp[i][0] 可以从前一天推出，也可以这一天买股票(-prices[i] 表示买这一天的股票)
    dp[i][1] 可以从前一天推出，也可以卖出这一天的股票
    */
    dp[i] = [
      Math.max(dp[i - 1][0], -prices[i]),
      Math.max(dp[i - 1][1], prices[i] + dp[i - 1][0]),
    ];
  }
  return dp[len - 1][1];
};

买卖股票的最佳时机二

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
  算出每天的利润(前一天买入，后一天卖出)，然后只取正数
  */
  let res = 0;
  for (let i = 0; i < prices.length; i++) {
    const num = prices[i] - prices[i - 1];
    res += num > 0 ? num : 0;
  }
  return res;
};

买卖股票的最佳时机三

至多买卖两次

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
    dp[i][0] 不操作
    dp[i][1] 第一次持有
    dp[i][2] 第一次不持有
    dp[i][3] 第二次持有
    dp[i][4] 第二次不持有
  */

  const len = prices.length;
  const dp = new Array(5).fill(0);
  dp[1] = -prices[0];
  dp[3] = -prices[0];
  for (let i = 1; i < len; i++) {
    dp[1] = Math.max(dp[1], dp[0] - prices[i]);
    dp[2] = Math.max(dp[2], dp[1] + prices[i]);
    dp[3] = Math.max(dp[3], dp[2] - prices[i]);
    dp[4] = Math.max(dp[4], dp[3] + prices[i]);
  }
  // console.log(dp[4]);
  return dp[4];
};

买卖股票的最佳时机四

至多可以买卖k次

/**
 * @param {number} k
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (k, prices) {
  if (prices == null || prices.length < 2 || k == 0) {
    return 0;
  }
  let dp = Array.from(Array(prices.length), () => Array(2 * k + 1).fill(0));
  for (let j = 1; j < 2 * k; j += 2) {
    dp[0][j] = 0 - prices[0];
  }
  for (let i = 1; i < prices.length; i++) {
    for (let j = 0; j < 2 * k; j += 2) {
      dp[i][j + 1] = Math.max(dp[i - 1][j + 1], dp[i - 1][j] - prices[i]);
      dp[i][j + 2] = Math.max(dp[i - 1][j + 2], dp[i - 1][j + 1] + prices[i]);
    }
  }
  // console.log(dp[prices.length - 1][2 * k]);
  return dp[prices.length - 1][2 * k];
};

买卖股票的最佳时机含冷冻期

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /* 
dp[i][0] 表示持有股票
dp[i][1] 表示卖出股票的状态
dp[i][2] 表示卖出股票(前一天卖出)
dp[i][3] 表示冷冻期
*/

  if (prices.length < 2) {
    return 0;
  } else if (prices.length < 3) {
    return Math.max(0, prices[1] - prices[0]);
  }
  let dp = Array.from(Array(prices.length), () => Array(4).fill(0));
  dp[0][0] = 0 - prices[0];
  for (i = 1; i < prices.length; i++) {
    dp[i][0] = Math.max(
      dp[i - 1][0],
      Math.max(dp[i - 1][1], dp[i - 1][3]) - prices[i]
    );
    dp[i][1] = Math.max(dp[i - 1][1], dp[i - 1][3]);
    dp[i][2] = dp[i - 1][0] + prices[i];
    dp[i][3] = dp[i - 1][2];
  }
  // console.log(
  //   Math.max(
  //     dp[prices.length - 1][1],
  //     dp[prices.length - 1][2],
  //     dp[prices.length - 1][3]
  //   )
  // );
  return Math.max(
    dp[prices.length - 1][1],
    dp[prices.length - 1][2],
    dp[prices.length - 1][3]
  );
};

买卖股票的最佳时机含手续费

/**
 * @param {number[]} prices
 * @param {number} fee
 * @return {number}
 */
var maxProfit = function (prices, fee) {
  /* 
    dp[i][0] 表示持有股票的最大金额
    dp[i][1] 表示不持有股票的最大金额
    */
  let dp = Array.from(Array(prices.length), () => Array(2).fill(0));
  dp[0][0] = 0 - prices[0];
  for (let i = 1; i < prices.length; i++) {
    dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] - prices[i]);
    dp[i][1] = Math.max(dp[i - 1][0] + prices[i] - fee, dp[i - 1][1]);
  }
  return Math.max(dp[prices.length - 1][0], dp[prices.length - 1][1]);
};

最长递增子序列

/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function (nums) {
  /* 
      dp[i] 表示以nums[i] 结尾的最长递增子序列的长度
      dp[i]=Math.max(dp[j]+1,dp[i])
  */
  const dp = new Array(nums.length).fill(1);
  for (let i = 0; i < nums.length; i++) {
    for (let j = 0; j < i; j++) {
      if (nums[i] > nums[j]) {
        dp[i] = Math.max(dp[j] + 1, dp[i]);
      }
    }
  }
  // 最终结果不一定是结尾，dp[i] 表示以num[i]结尾的最长递增子序列的长度
  return Math.max(...dp);
};
