/**
 * @param {character[]} s
 * @return {void} Do not return anything, modify s in-place instead.
 */
var reverseString = function (s) {
  let left = 0;
  let right = s.length - 1;
  while (left < right) {
    const tem = s[right]
    s[right] = s[left];
    s[left] = tem
    left++;
    right--
  }
  return s
};

反转字符串（二）

/**
 * @param {string} s
 * @param {number} k
 * @return {string}
 */
var reverseStr = function (s, k) {
  const len = s.length;
  let resArr = s.split("");
  // 反转后边的字符
  for (let i = 0; i < len; i += 2 * k) {  // 每隔 2k 个字符的前 k 个字符进行反转
    // 判断边界条件,不能溢出(如果剩余字符少于 k 个，则将剩余字符全部反转)
    let l = i - 1, r = i + k > len ? len : i + k;
    // 反转字符串
    while (++l < --r) {
      let tem = resArr[r]
      resArr[r] = resArr[l]
      resArr[l] = tem
    }
  }
  return resArr.join("");
};

解决字符串匹配问题
前缀：不包含尾字符
后缀：不包含首字符
最长相等前后缀，遇到不匹配的时候，回退到与前缀相等的一个位置
构建next数组

function getNext(next, s) {
  let j = -1;
  next[0] = j;
  for (let i = 1; i < s.length; i++) {
    // 不匹配一直回退(j>=0)，不代表只回退一次
    while (j >= 0 && s[i] !== s[j + 1]) {
      j = next[j];
    }
    // 配匹上，j++
    if (s[i] === s[j + 1]) {
      j++;
    }
    next[i] = j;
  }
}

var MyQueue = function () {
  this.stackIn = []
  this.stackOut = []
};

/** 
 * @param {number} x
 * @return {void}
 */
MyQueue.prototype.push = function (x) {
  this.stackIn.push(x)
};

/**
 * @return {number}
 */
MyQueue.prototype.pop = function () {
  // 判断出栈中是否有
  if (this.stackOut.length > 0) {
    return this.stackOut.pop()
  }
  while (this.stackIn.length > 0) {
    this.stackOut.push(this.stackIn.pop())
  }
  return this.stackOut.pop()
};

/**
 * @return {number}
 */
MyQueue.prototype.peek = function () {
  // 查看栈顶元素但是不用弹出
  const x = this.pop();
  this.stackOut.push(x);
  return x;
};

/**
 * @return {boolean}
 */
MyQueue.prototype.empty = function () {
  // 判断是否为空，空返回true
  return this.stackIn.length <= 0 && this.stackOut.length <= 0
};

有效括号

/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function (s) {
  // 长度为奇数
  if (s.length % 2 !== 0) {
    return false
  }
  // 用栈去匹配
  const stack = []
  for (let i = 0; i < s.length; i++) {
    if (s[i] === '(') {
      stack.push(')')
    }
    if (s[i] === '[') {
      stack.push(']')
    }
    if (s[i] === '{') {
      stack.push('}')
    }
    if (s[i] === ')' || s[i] === ']' || s[i] === '}') {
      if (stack[stack.length - 1] !== s[i]) {
        return false
      } else {
        stack.pop()
      }
    }
  }
  return stack.length == 0
};

删除字符串中所有相邻重复项

每次匹配栈顶元素，如果匹配上就移除

/**
 * @param {string} s
 * @return {string}
 */
var removeDuplicates = function (s) {
  const stack = []
  for (let i = 0; i < s.length; i++) {
    // 相同字母，移除
    if (stack.length > 0 && s[i] === stack[stack.length - 1]) {
      // 重复元素，栈顶弹出
      stack.pop()
    } else {
      stack.push(s[i])
    }
  }
  return stack.join('')
};

逆波兰表达式求值

遇见数字添加到栈里边，遇见运算符号从栈里边取出两个数进行运算，并把结果压入栈

/**
 * @param {string[]} tokens
 * @return {number}
 */
var evalRPN = function (tokens) {
  const stack = [];
  for (const token of tokens) {
    if (isNaN(Number(token))) { // 非数字
      // n2表示后边的数字
      const n2 = stack.pop(); // 出栈两个数字
      const n1 = stack.pop();
      switch (token) { // 判断运算符类型，算出新数入栈
        case "+":
          stack.push(n1 + n2);
          break;
        case "-":
          stack.push(n1 - n2);
          break;
        case "*":
          stack.push(n1 * n2);
          break;
        case "/":
          stack.push(n1 / n2 | 0);
          break;
      }
    } else { // 数字
      stack.push(Number(token));
    }
  }
  return stack[0]; // 因没有遇到运算符而待在栈中的结果
};

将数添加进队列中，如果添加的数比之前的大，就把之前的全部pop出去，出口一直保存着最大元素

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var maxSlidingWindow = function (nums, k) {
  class SignleStack {
    constructor() {
      this.queue = []
    }
    push(value) {
      let last = this.queue[this.queue.length - 1]
      while (last !== undefined && value > last) {
        this.queue.pop()
        last = this.queue[this.queue.length - 1]
      }
      this.queue.push(value)
    }
    refresh(value) {
      const max = this.getMax()
      if (max === value) {
        this.queue.shift()
      }
    }
    getMax() {
      return this.queue[0]
    }
  }
  const res = []
  const signleStack = new SignleStack()
  // 加入k个元素
  let i = 0
  while (i < k) {
    signleStack.push(nums[i])
    i++
  }
  let j = 0
  res.push(signleStack.getMax())
  while (i < nums.length) {
    signleStack.push(nums[i])
    // 保证单调栈中的元素永远是当前滑动窗口的中的值[1,-1],1的情况
    signleStack.refresh(nums[j])
    res.push(signleStack.getMax())
    i++
    j++
  }
  return res
};

var topKFrequent = function (nums, k) {
  const map = new Map()
  for (let i = 0; i < nums.length; i++) {
    map.set(nums[i], (map.get(nums[i]) || 0) + 1)

二叉树

二叉树基础

二叉树的种类：

  1. 满二叉树，节点数量为2^k-1 (k为节点深度)
  2. 完全二叉树，（其他层都是满的，底层是从左到右连续的二叉树，两个节点之间没有断层）
  3. 搜索二叉树，（左边的所有节点都小于中间节点，右边的所有节点都大于中间节点）
  4. 平衡搜索二叉树，（在搜索二叉树的基础上，左子树和右子树高度差不超过1）

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)

 *     this.right = (right===undefined ? null : right)

/**
 * @param {TreeNode} root
 * @return {number[]}
 */
//前序遍历
var preorderTraversal = function (root) {
  const res = []
  function getValue(node) {
    if (!node) {
      return
    }
    res.push(node.val)
    // 遍历左节点
    getValue(node.left)
    // 遍历右节点
    getValue(node.right)
  }
  getValue(root)
  return res
};

二叉树的层序遍历

借助队列，保存每一层的数据

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
var levelOrder = function (root) {
  // 使用队列去辅助，弹出原生
  const queue = []
  const res = []
  if (!root) {
    return res
  }
  queue.push(root)
  // 遍历对列
  while (queue.length > 0) {
    // 保存本层节点的数量(因为之后遍历的时候会添加下一层的数据)
    let len = queue.length
    const resItem = []
    // 将本层数据弹出，并保存进数组
    while (len > 0) {
      const node = queue.shift()
      resItem.push(node.val)
      // 将下一层的左右节点压入
      if (node.left) {
        queue.push(node.left)
      }
      if (node.right) {
        queue.push(node.right)
      }
      len--
    }
    res.push(resItem)
  }
  return res
};

反转二叉树

可以使用前序或者后序，中序比较麻烦

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var invertTree = function (root) {
  // 交换左右节点
  function invert(node) {
    if (!node) {
      return null;
    }
    // 交换两个节点
    const left = node.left
    node.left = node.right
    node.right = left
    invert(node.left)
    invert(node.right)
  }
  invert(root)
  return root
};
