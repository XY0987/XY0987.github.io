/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var findSubsequences = function (nums) {
  // 不能进行排序,因为排序回导致数据改变,导致一些不必要的答案
  const path = [];
  const res = [];
  function backtracking(startIndex) {
    if (path.length >= 2) {
      res.push([...path]);
    }
    if (startIndex >= nums.length) {
      return;
    }
    // 去重
    const set = new Set();
    for (let i = startIndex; i < nums.length; i++) {
      if (
        set.has(nums[i]) ||
        (path.length > 0 && nums[i] < path[path.length - 1])
      ) {
        continue;
      }
      set.add(nums[i]);
      // if (
      //   i > startIndex &&
      //   nums.indexOf(nums[i]) > 0 &&
      //   nums.indexOf(nums[i]) < i
      // ) {
      //   continue;
      // }
      path.push(nums[i]);
      backtracking(i + 1);
      path.pop();
    }
  }
  backtracking(0);
  return res;
};

全排列

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function (nums) {
  const res = [];
  const path = [];
  const used = new Array(nums.length).fill(0);

  function backtracking() {
    if (path.length === nums.length) {
      res.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) {
        continue;
      }
      path.push(nums[i]);
      used[i] = 1;
      backtracking();
      path.pop();
      used[i] = 0;
    }
  }
  backtracking();
  return res;
};

全排列二

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permuteUnique = function (nums) {
  const res = [];
  const path = [];
  const used = new Array(nums.length).fill(0);
  nums.sort((a, b) => {
    return a - b;
  });
  function backtracking() {
    if (path.length === nums.length) {
      res.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      /* 
    树层去重(前一个数没有用过，才需要进行树层去重)，否则不会拿到结果
    被用过了相当于树枝，树枝不用去重
    以该元素开头的，前边以该元素开头的一定包含该元素开头的所有情况
    */
      if (i > 0 && !used[i - 1] && nums[i] === nums[i - 1]) {
        continue;
      }
      // 当前数没有被用
      if (!used[i]) {
        path.push(nums[i]);
        used[i] = 1;
        backtracking();
        path.pop();
        used[i] = 0;
      }
    }
  }
  backtracking();
  return res;
};

N 皇后

/**
 * @param {number} n
 * @return {string[][]}
 */
var solveNQueens = function (n) {
  /* 
      每一层递归对应的是不同的行，循环种遍历该行的不同列
    */
  const result = [];
  // 判断是否符合规则（同一行，同一列，斜45度是否有皇后，能不能放皇后）
  function isValid(row, col, chessBoard, n) {
    for (let i = 0; i < row; i++) {
      if (chessBoard[i][col] === "Q") {
        return false;
      }
    }
    for (let i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
      if (chessBoard[i][j] === "Q") {
        return false;
      }
    }
    for (let i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
      if (chessBoard[i][j] === "Q") {
        return false;
      }
    }
    return true;
  }
  // 将棋盘转化为最终结果
  function transformChessBoard(chessBoard) {
    let chessBoardBack = [];
    chessBoard.forEach((row) => {
      let rowStr = "";
      row.forEach((value) => {
        rowStr += value;
      });
      chessBoardBack.push(rowStr);
    });

    return chessBoardBack;
  }
  function backtracing(row, chessBoard) {
    if (row === n) {
      result.push(transformChessBoard(chessBoard));
      return;
    }
    // 遍历该行的每一列
    for (let col = 0; col < n; col++) {
      // 判断是否符合规范
      if (isValid(row, col, chessBoard, n)) {
        chessBoard[row][col] = "Q";
        backtracing(row + 1, chessBoard);
        chessBoard[row][col] = ".";
      }
    }
  }
  let chessBoard = new Array(n).fill([]).map(() => new Array(n).fill("."));
  backtracing(0, chessBoard);
  return result;
};

/**
 * @param {character[][]} board
 * @return {void} Do not return anything, modify board in-place instead.
 */
var solveSudoku = function (board) {
  // 判断值是否符合规范
  function isValid(row, col, val, board) {
    let len = board.length;
    // 行不能重复
    for (let i = 0; i < len; i++) {
      if (board[row][i] === val) {
        return false;
      }
    }
    // 列不能重复
    for (let i = 0; i < len; i++) {
      if (board[i][col] === val) {
        return false;
      }
    }
    let startRow = Math.floor(row / 3) * 3;
    let startCol = Math.floor(col / 3) * 3;
    for (let i = startRow; i < startRow + 3; i++) {
      for (let j = startCol; j < startCol + 3; j++) {
        if (board[i][j] === val) {
          return false;
        }
      }
    }
    return true;
  }
  // 回溯
  function backTracking() {
    // 循环每一行
    for (let i = 0; i < board.length; i++) {
      // 循环每一列
      for (let j = 0; j < board[0].length; j++) {
        // 只处理值是点的
        if (board[i][j] !== ".") continue;
        // 将该位置存放1到9，是否符合规则
        for (let val = 1; val <= 9; val++) {
          if (isValid(i, j, `${val}`, board)) {
            board[i][j] = `${val}`;
            // 遇到结果，立即返回
            if (backTracking()) {
              return true;
            }
            board[i][j] = `.`;
          }
        }
        return false;
      }
    }
    return true;
  }
  backTracking(board);
  return board;
};

/**
 * @param {number[]} g
 * @param {number[]} s
 * @return {number}
 */
var findContentChildren = function (g, s) {
  g.sort((a, b) => {
    return a - b;
  });
  s.sort((a, b) => {
    return a - b;
  });
  let index = s.length - 1;
  let res = 0;
  for (let i = g.length - 1; i >= 0; i--) {
    // 先满足大胃口孩子
    if (index >= 0 && s[index] >= g[i]) {
      res++;
      index--;
    }
  }
  return res;
};

/**
 * @param {number[]} nums
 * @return {number}
 */
var wiggleMaxLength = function (nums) {
  /* 
1. 上下坡有平坡
2. 首尾元素
3. 单调坡度有平坡
*/

  if (nums.length === 1) {
    return 1;
  }
  // 相当于延长了前边
  let prediff = 0;
  let curdiff = 0;
  // 默认算上最后一个元素
  let res = 1;
  for (let i = 0; i < nums.length - 1; i++) {
    curdiff = nums[i + 1] - nums[i];
    if ((prediff >= 0 && curdiff < 0) || (prediff <= 0 && curdiff > 0)) {
      res++;
      prediff = curdiff;
    }
  }
  return res;
};

最大子序和

/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function (nums) {
  /* 
暴力解法：双重for循环，去遍历所有子数组，求和
贪心解法：不断的去求和，如果求得的连续和小于0了，那就从下一个数开始
*/
  let res = -Infinity;
  let count = 0;
  for (let i = 0; i < nums.length; i++) {
    count += nums[i];
    if (count > res) {
      res = count;
    }
    // 连续和为负数了，重新赋值为0，相当于从下一个数开始
    if (count < 0) {
      count = 0;
    }
  }
  return res;
};

买卖股票的最佳时机二

/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  /*
    尽可能的多卖，遇到正数相加，负数不要 
    算出每天的利润(前一天买入，后一天卖出)，然后只取正数
    */
  let res = 0;
  for (let i = 0; i < prices.length; i++) {
    const num = prices[i] - prices[i - 1];
    res += num > 0 ? num : 0;
  }
  return res;
};

跳跃游戏

/**
 * @param {number[]} nums
 * @return {boolean}
 */
var canJump = function (nums) {
  /* 
    查看每个点的覆盖范围，根据覆盖范围是否能到达终点去判断
  */
  let cover = 0;
  if (nums.length === 1) {
    return true;
  }
  // 只在覆盖范围内移动
  for (let i = 0; i <= cover; i++) {
    // 取最大的覆盖范围
    cover = Math.max(cover, i + nums[i]);
    // 覆盖范围能到达终点
    if (cover >= nums.length - 1) {
      return true;
    }
  }
  return false;
};

  //     nextIndex = Math.max(nextIndex, nums[i] + i)

  /* 
        先记录当前值的覆盖范围，如果当前覆盖范围没有到终点
        就启用下一步的覆盖范围，每次都是取最大值
    */
  let curIndex = 0;
  let nextIndex = 0;
  let steps = 0;
  for (let i = 0; i < nums.length - 1; i++) {
    nextIndex = Math.max(nums[i] + i, nextIndex);
    // 等于时还没有到达终点，启用下一个(每次都是保存最大的值)
    if (i === curIndex) {
      // 如果之前就走到终点了，后边的不会走到这里
      curIndex = nextIndex;
      steps++;
    }
  }

  return steps;
};

k 次取反后最大化的数组和

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var largestSumAfterKNegations = function (nums, k) {
  for (let i = 0; i < k; i++) {
    nums.sort((a, b) => {
      return a - b;
    });
    nums[0] = -nums[0];
  }
  return nums.reduce((prev, next) => {
    return prev + next;
  }, 0);
};

/**
 * @param {number[]} gas
 * @param {number[]} cost
 * @return {number}
 */
var canCompleteCircuit = function (gas, cost) {
  /* 
    统计每个站点剩余的油量，如果剩余的为负数，就将下一个站点作为起始位置
    */
  let curSum = 0;
  let totalSum = 0;
  let start = 0;
  for (let i = 0; i < gas.length; i++) {
    curSum += gas[i] - cost[i];
    totalSum += gas[i] - cost[i];
    if (curSum < 0) {
      start = i + 1;
      curSum = 0;
    }
  }
  if (totalSum < 0) {
    return -1;
  }

  return start;
};
